73 static void renameAccessesMaps(std::unordered_map<int, Extents>& accessesMap, int oldAccessID,
74  int newAccessID) {
75  for(auto it = accessesMap.begin(); it != accessesMap.end();) {
76  if(it->first == oldAccessID) {
77  accessesMap.emplace(newAccessID, it->second);
78  it = accessesMap.erase(it);
79  } else {
80  ++it;
81  }
82  }
83 }
