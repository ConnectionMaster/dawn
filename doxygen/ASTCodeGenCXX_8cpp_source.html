54 void ASTCodeGenCXX::visit(const std::shared_ptr<VarDeclStmt>& stmt) {
55  if(scopeDepth_ == 0)
56  ss_ << std::string(indent_, ' ');
57 
58  const auto& type = stmt->getType();
59  if(type.isConst())
60  ss_ << "const ";
61  if(type.isVolatile())
62  ss_ << "volatile ";
63 
64  if(type.isBuiltinType())
65  ss_ << ASTCodeGenCXX::builtinTypeIDToCXXType(type.getBuiltinTypeID(), true);
66  else
67  ss_ << type.getName();
68  ss_ << " " << getName(stmt);
69 
70  if(stmt->isArray())
71  ss_ << "[" << stmt->getDimension() << "]";
72 
73  if(stmt->hasInit()) {
74  ss_ << " " << stmt->getOp() << " ";
75  if(!stmt->isArray())
76  stmt->getInitList().front()->accept(*this);
77  else {
78  ss_ << "{";
79  int numInit = stmt->getInitList().size();
80  for(int i = 0; i < numInit; ++i) {
81  stmt->getInitList()[i]->accept(*this);
82  ss_ << ((i != (numInit - 1)) ? ", " : "");
83  }
84  ss_ << "}";
85  }
86  }
87  ss_ << ";\n";
88 }
