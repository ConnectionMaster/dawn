46 static void getAccessIDFromAssignment(const StencilInstantiation* instantiation,
47  AssignmentExpr* assignment, std::set<int>& LHSAccessIDs,
48  std::set<int>& RHSAccessIDs) {
49  auto computeAccessIDs = [&](const std::shared_ptr<Expr>& expr, std::set<int>& AccessIDs) {
50  AccessIDGetter getter{instantiation};
51  expr->accept(getter);
52  AccessIDs = std::move(getter.AccessIDs);
53  };
54 
55  computeAccessIDs(assignment->getLeft(), LHSAccessIDs);
56  computeAccessIDs(assignment->getRight(), RHSAccessIDs);
57 }

62  LoopOrderKind loopOrder) {

68 static void reportRaceCondition(const Statement& statement, StencilInstantiation* instantiation) {
69  DiagnosticsBuilder diag(DiagnosticsKind::Error, statement.ASTStmt->getSourceLocation());
70 
71  if(isa<IfStmt>(statement.ASTStmt.get())) {
72  diag << "unresolvable race-condition in body of if-statement";
73  } else {
74  diag << "unresolvable race-condition in statement";
75  }
76 
77  instantiation->getOptimizerContext()->getDiagnostics().report(diag);
78 
79  // Print stack trace of stencil calls
80  if(statement.StackTrace) {
81  std::vector<sir::StencilCall*>& stackTrace = *statement.StackTrace;
82  for(int i = stackTrace.size() - 1; i >= 0; --i) {
83  DiagnosticsBuilder note(DiagnosticsKind::Note, stackTrace[i]->Loc);
84  note << "detected during instantiation of stencil-call '" << stackTrace[i]->Callee << "'";
85  instantiation->getOptimizerContext()->getDiagnostics().report(note);
86  }
87  }
88 }

94 bool PassFieldVersioning::run(StencilInstantiation* stencilInstantiation) {
95  OptimizerContext* context = stencilInstantiation->getOptimizerContext();
96  numRenames_ = 0;
97 
98  for(auto& stencilPtr : stencilInstantiation->getStencils()) {
99  Stencil& stencil = *stencilPtr;
100 
101  // Iterate multi-stages backwards
102  int stageIdx = stencil.getNumStages() - 1;
103  for(auto multiStageRit = stencil.getMultiStages().rbegin(),
104  multiStageRend = stencil.getMultiStages().rend();
105  multiStageRit != multiStageRend; ++multiStageRit) {
106  MultiStage& multiStage = (**multiStageRit);
107  LoopOrderKind loopOrder = multiStage.getLoopOrder();
108 
109  std::shared_ptr<DependencyGraphAccesses> newGraph, oldGraph;
110  newGraph = std::make_shared<DependencyGraphAccesses>(stencilInstantiation);
111 
112  // Iterate stages bottom -> top
113  for(auto stageRit = multiStage.getStages().rbegin(),
114  stageRend = multiStage.getStages().rend();
115  stageRit != stageRend; ++stageRit) {
116  Stage& stage = (**stageRit);
117  DoMethod& doMethod = stage.getSingleDoMethod();
118 
119  // Iterate statements bottom -> top
120  for(int stmtIndex = doMethod.getStatementAccessesPairs().size() - 1; stmtIndex >= 0;
121  --stmtIndex) {
122  oldGraph = newGraph->clone();
123 
124  auto& stmtAccessesPair = doMethod.getStatementAccessesPairs()[stmtIndex];
125  newGraph->insertStatementAccessesPair(stmtAccessesPair);
126 
127  // Try to resolve race-conditions by using double buffering if necessary
128  auto rc =
129  fixRaceCondition(newGraph.get(), stencil, doMethod, loopOrder, stageIdx, stmtIndex);
130 
131  if(rc == RCKind::RK_Unresolvable)
132  // Nothing we can do ... bail out
133  return false;
134  else if(rc == RCKind::RK_Fixed) {
135  // We fixed a race condition (this means some fields have changed and our current graph
136  // is invalid)
137  newGraph = oldGraph;
138  newGraph->insertStatementAccessesPair(stmtAccessesPair);
139  }
140  }
141  }
142  stageIdx--;
143  }
144  }
145 
146  if(context->getOptions().ReportPassFieldVersioning && numRenames_ == 0)
147  std::cout << "\nPASS: " << getName() << ": " << stencilInstantiation->getName()
148  << ": no rename\n";
149  return true;
150 }

152 PassFieldVersioning::RCKind
153 PassFieldVersioning::fixRaceCondition(const DependencyGraphAccesses* graph, Stencil& stencil,
154  DoMethod& doMethod, LoopOrderKind loopOrder, int stageIdx,
155  int index) {
156  using Vertex = DependencyGraphAccesses::Vertex;
157  using Edge = DependencyGraphAccesses::Edge;
158 
159  Statement& statement = *doMethod.getStatementAccessesPairs()[index]->getStatement();
160 
161  StencilInstantiation* instantiation = stencil.getStencilInstantiation();
162  OptimizerContext* context = instantiation->getOptimizerContext();
163  int numRenames = 0;
164 
165  // Vector of strongly connected components with atleast one stencil access
166  auto stencilSCCs = make_unique<std::vector<std::set<int>>>();
167 
168  // Find all strongly connected components in the graph ...
169  auto SCCs = make_unique<std::vector<std::set<int>>>();
170  graph->findStronglyConnectedComponents(*SCCs);
171 
172  // ... and add those which have atleast one stencil access
173  for(std::set<int>& scc : *SCCs) {
174  bool isStencilSCC = false;
175 
176  for(int FromAccessID : scc) {
177  std::size_t FromVertexID = graph->getVertexIDFromID(FromAccessID);
178 
179  for(const Edge& edge : *graph->getAdjacencyList()[FromVertexID]) {
180  if(scc.count(graph->getIDFromVertexID(edge.ToVertexID)) &&
181  isHorizontalStencilOrCounterLoopOrderExtent(edge.Data, loopOrder)) {
182  isStencilSCC = true;
183  stencilSCCs->emplace_back(std::move(scc));
184  break;
185  }
186  }
187 
188  if(isStencilSCC)
189  break;
190  }
191  }
192 
193  if(stencilSCCs->empty()) {
194  // Check if we have self dependencies e.g `u = u(i+1)` (Our SCC algorithm does not capture SCCs
195  // of size one i.e single nodes)
196  for(const auto& AccessIDVertexPair : graph->getVertices()) {
197  const Vertex& vertex = AccessIDVertexPair.second;
198 
199  for(const Edge& edge : *graph->getAdjacencyList()[vertex.VertexID]) {
200  if(edge.FromVertexID == edge.ToVertexID &&
201  isHorizontalStencilOrCounterLoopOrderExtent(edge.Data, loopOrder)) {
202  stencilSCCs->emplace_back(std::set<int>{vertex.ID});
203  break;
204  }
205  }
206  }
207  }
208 
209  // If we only have non-stencil SCCs and there are no input and output fields (i.e we don't have a
210  // DAG) we have to break (by renaming) one of the SCCs to get a DAG. For example:
211  //
212  // field_a = field_b;
213  // field_b = field_a;
214  //
215  // needs to be renamed to
216  //
217  // field_a = field_b_1;
218  // field_b = field_a;
219  //
220  if(stencilSCCs->empty() && !SCCs->empty() && !graph->isDAG()) {
221  stencilSCCs->emplace_back(std::move(SCCs->front()));
222  }
223 
224  if(stencilSCCs->empty())
225  return RCKind::RK_Nothing;
226 
227  // Check whether our statement is an `ExprStmt` and contains an `AssignmentExpr`. If not,
228  // we cannot perform any double buffering (e.g if there is a problem inside an `IfStmt`, nothing
229  // we can do (yet ;))
230  AssignmentExpr* assignment = nullptr;
231  if(ExprStmt* stmt = dyn_cast<ExprStmt>(statement.ASTStmt.get()))
232  assignment = dyn_cast<AssignmentExpr>(stmt->getExpr().get());
233 
234  if(!assignment) {
235  if(context->getOptions().DumpRaceConditionGraph)
236  graph->toDot("rc_" + instantiation->getName() + ".dot");
237  reportRaceCondition(statement, instantiation);
238  return RCKind::RK_Unresolvable;
239  }
240 
241  // Get AccessIDs of the LHS and RHS
242  std::set<int> LHSAccessIDs, RHSAccessIDs;
243  getAccessIDFromAssignment(instantiation, assignment, LHSAccessIDs, RHSAccessIDs);
244 
245  DAWN_ASSERT_MSG(LHSAccessIDs.size() == 1, "left hand side should only have only one AccessID");
246  int LHSAccessID = *LHSAccessIDs.begin();
247 
248  // If the LHSAccessID is not part of the SCC, we cannot resolve the race-condition
249  for(std::set<int>& scc : *stencilSCCs) {
250  if(!scc.count(LHSAccessID)) {
251  if(context->getOptions().DumpRaceConditionGraph)
252  graph->toDot("rc_" + instantiation->getName() + ".dot");
253  reportRaceCondition(statement, instantiation);
254  return RCKind::RK_Unresolvable;
255  }
256  }
257 
258  DAWN_ASSERT_MSG(stencilSCCs->size() == 1, "only one strongly connected component can be handled");
259  std::set<int>& stencilSCC = (*stencilSCCs)[0];
260 
261  std::set<int> renameCandiates;
262  for(int AccessID : stencilSCC) {
263  if(RHSAccessIDs.count(AccessID))
264  renameCandiates.insert(AccessID);
265  }
266 
267  if(context->getOptions().ReportPassFieldVersioning)
268  std::cout << "\nPASS: " << getName() << ": " << instantiation->getName()
269  << ": rename:" << statement.ASTStmt->getSourceLocation().Line;
270 
271  // Create a new multi-versioned field and rename all occurences
272  for(int oldAccessID : renameCandiates) {
273  int newAccessID = instantiation->createVersionAndRename(oldAccessID, &stencil, stageIdx, index,
274  assignment->getRight(),
275  StencilInstantiation::RD_Above);
276 
277  if(context->getOptions().ReportPassFieldVersioning)
278  std::cout << (numRenames != 0 ? ", " : " ") << instantiation->getNameFromAccessID(oldAccessID)
279  << ":" << instantiation->getNameFromAccessID(newAccessID);
280 
281  numRenames++;
282  }
283 
284  if(context->getOptions().ReportPassFieldVersioning && numRenames > 0)
285  std::cout << "\n";
286 
287  numRenames_ += numRenames;
288  return RCKind::RK_Fixed;
289 }
