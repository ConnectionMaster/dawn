43 template <typename T>
44 unsigned computeEditDistance(ArrayRef<T> fromArray, ArrayRef<T> toArray,
45  bool allowReplacements = true, unsigned maxEditDistance = 0) {
46  // The algorithm implemented below is the "classic" dynamic-programming algorithm for computing
47  // the Levenshtein distance, which is described here:
48  //
49  // http://en.wikipedia.org/wiki/Levenshtein_distance
50  //
51  // Although the algorithm is typically described using an m x n array, only one row plus one
52  // element are used at a time, so this implementation just keeps one vector for the row. To update
53  // one entry, only the entries to the left, top, and top-left are needed. The left entry is in
54  // Row[x-1], the top entry is what's in Row[x] from the last iteration, and the top-left entry is
55  // stored in Previous.
56  typename ArrayRef<T>::size_type m = fromArray.size();
57  typename ArrayRef<T>::size_type n = toArray.size();
58 
59  const unsigned SmallBufferSize = 64;
60  unsigned SmallBuffer[SmallBufferSize];
61  std::unique_ptr<unsigned[]> Allocated;
62  unsigned* Row = SmallBuffer;
63  if(n + 1 > SmallBufferSize) {
64  Row = new unsigned[n + 1];
65  Allocated.reset(Row);
66  }
67 
68  for(unsigned i = 1; i <= n; ++i)
69  Row[i] = i;
70 
71  for(typename ArrayRef<T>::size_type y = 1; y <= m; ++y) {
72  Row[0] = y;
73  unsigned BestThisRow = Row[0];
74 
75  unsigned Previous = y - 1;
76  for(typename ArrayRef<T>::size_type x = 1; x <= n; ++x) {
77  int OldRow = Row[x];
78  if(allowReplacements) {
79  Row[x] = std::min(Previous + (fromArray[y - 1] == toArray[x - 1] ? 0u : 1u),
80  std::min(Row[x - 1], Row[x]) + 1);
81  } else {
82  if(fromArray[y - 1] == toArray[x - 1])
83  Row[x] = Previous;
84  else
85  Row[x] = std::min(Row[x - 1], Row[x]) + 1;
86  }
87  Previous = OldRow;
88  BestThisRow = std::min(BestThisRow, Row[x]);
89  }
90 
91  if(maxEditDistance && BestThisRow > maxEditDistance)
92  return maxEditDistance + 1;
93  }
94 
95  unsigned Result = Row[n];
96  return Result;
97 }
