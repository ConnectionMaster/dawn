372 class FieldAccessExpr : public Expr {
373  std::string name_;
374 
375  // The offset known so far. If we have directional or offset arguments, we have to perform a
376  // lazy evaluation to compute the real offset once we know the mapping of the directions (and
377  // offsets) to the actual arguments of the stencil function.
378  Array3i offset_;
379 
380  // Mapping of the directional and offset arguments of the stencil function.
381  // The `argumentMap` stores an index to the argument list of the stencil function with -1
382  // indicating this argument is unused. The `argumentOffset` holds the parsed offsets of the
383  // direction (or offset).
384  //
385  // Consider the following example (given in the gridtools_clang DSL) which implements an average
386  // stencil function :
387  //
388  // stencil_function avg {
389  // storage in;
390  // direction dir;
391  //
392  // Do {
393  // return in(dir+1) + in;
394  // }
395  // };
396  //
397  // The `in(dir+1)` FieldAccess would have the following members:
398  // - name_ : "in"
399  // - offset_ : {0, 0, 0} // We don't have any i,j or k accesses
400  // - argumentMap_ : {1, -1, -1} // `dir` maps to the 1st argument of `avg` (0 based)
401  // - argumentOffset_ : {1, 0, 0} // `dir+1` has an offset `+1`
402  //
403  Array3i argumentMap_;
404  Array3i argumentOffset_;
405 
406  // Negate the offset (this allows writing `in(-off)`)
407  bool negateOffset_;
408 
409 public:
412  FieldAccessExpr(const std::string& name, Array3i offset = Array3i{{0, 0, 0}},
413  Array3i argumentMap = Array3i{{-1, -1, -1}},
414  Array3i argumentOffset = Array3i{{0, 0, 0}}, bool negateOffset = false,
415  SourceLocation loc = SourceLocation());
416  FieldAccessExpr(const FieldAccessExpr& expr);
417  FieldAccessExpr& operator=(FieldAccessExpr expr);
418  virtual ~FieldAccessExpr();
420 
422  bool hasArguments() const {
423  return (argumentMap_[0] != -1 || argumentMap_[1] != -1 || argumentMap_[2] != -1);
424  }
425 
429  void setPureOffset(const Array3i& offset);
430 
431  const std::string& getName() const { return name_; }
432  std::string& getName() { return name_; }
433 
434  const Array3i& getOffset() const { return offset_; }
435  Array3i& getOffset() { return offset_; }
436 
437  const Array3i& getArgumentMap() const { return argumentMap_; }
438  Array3i& getArgumentMap() { return argumentMap_; }
439 
440  const Array3i& getArgumentOffset() const { return argumentOffset_; }
441  Array3i& getArgumentOffset() { return argumentOffset_; }
442 
443  bool negateOffset() const { return negateOffset_; }
444 
445  virtual std::shared_ptr<Expr> clone() const override;
446  virtual bool equals(const Expr* other) const override;
447  virtual void accept(ASTVisitor& visitor) override;
448  static bool classof(const Expr* expr) { return expr->getKind() == EK_FieldAccessExpr; }
