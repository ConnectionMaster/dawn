374 class FieldAccessExpr : public Expr {
375  std::string name_;
376 
377  // The offset known so far. If we have directional or offset arguments, we have to perform a
378  // lazy evaluation to compute the real offset once we know the mapping of the directions (and
379  // offsets) to the actual arguments of the stencil function.
380  Array3i offset_;
381 
382  // Mapping of the directional and offset arguments of the stencil function.
383  // The `argumentMap` stores an index to the argument list of the stencil function with -1
384  // indicating this argument is unused. The `argumentOffset` holds the parsed offsets of the
385  // direction (or offset).
386  //
387  // Consider the following example (given in the gridtools_clang DSL) which implements an average
388  // stencil function :
389  //
390  // stencil_function avg {
391  // storage in;
392  // direction dir;
393  //
394  // Do {
395  // return in(dir+1) + in;
396  // }
397  // };
398  //
399  // The `in(dir+1)` FieldAccess would have the following members:
400  // - name_ : "in"
401  // - offset_ : {0, 0, 0} // We don't have any i,j or k accesses
402  // - argumentMap_ : {1, -1, -1} // `dir` maps to the 1st argument of `avg` (0 based)
403  // - argumentOffset_ : {1, 0, 0} // `dir+1` has an offset `+1`
404  //
405  Array3i argumentMap_;
406  Array3i argumentOffset_;
407 
408  // Negate the offset (this allows writing `in(-off)`)
409  bool negateOffset_;
410 
411 public:
414  FieldAccessExpr(const std::string& name, Array3i offset = Array3i{{0, 0, 0}},
415  Array3i argumentMap = Array3i{{-1, -1, -1}},
416  Array3i argumentOffset = Array3i{{0, 0, 0}}, bool negateOffset = false,
417  SourceLocation loc = SourceLocation());
418  FieldAccessExpr(const FieldAccessExpr& expr);
419  FieldAccessExpr& operator=(FieldAccessExpr expr);
420  virtual ~FieldAccessExpr();
422 
424  bool hasArguments() const {
425  return (argumentMap_[0] != -1 || argumentMap_[1] != -1 || argumentMap_[2] != -1);
426  }
427 
431  void setPureOffset(const Array3i& offset);
432 
433  const std::string& getName() const { return name_; }
434  std::string& getName() { return name_; }
435 
436  const Array3i& getOffset() const { return offset_; }
437  Array3i& getOffset() { return offset_; }
438 
439  const Array3i& getArgumentMap() const { return argumentMap_; }
440  Array3i& getArgumentMap() { return argumentMap_; }
441 
442  const Array3i& getArgumentOffset() const { return argumentOffset_; }
443  Array3i& getArgumentOffset() { return argumentOffset_; }
444 
445  bool negateOffset() const { return negateOffset_; }
446 
447  virtual std::shared_ptr<Expr> clone() const override;
448  virtual bool equals(const Expr* other) const override;
449  virtual void accept(ASTVisitor& visitor) override;
