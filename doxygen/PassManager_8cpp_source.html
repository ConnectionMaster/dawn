23 bool PassManager::runAllPassesOnStecilInstantiation(StencilInstantiation* instantiation) {
24  std::vector<std::string> passesRan;
25 
26  for(auto& pass : passes_) {
27  for(const auto& dependency : pass->getDependencies())
28  if(std::find(passesRan.begin(), passesRan.end(), dependency) == passesRan.end()) {
29  DiagnosticsBuilder diag(DiagnosticsKind::Error);
30  diag << "invalid pass registration: optimizer pass '" << pass->getName() << "' depends on '"
31  << dependency << "'";
32  instantiation->getOptimizerContext()->getDiagnostics().report(diag);
33  return false;
34  }
35 
36  if(!runPassOnStecilInstantiation(instantiation, pass.get()))
37  return false;
38 
39  passesRan.emplace_back(pass->getName());
40  }
41  return true;
42 }
