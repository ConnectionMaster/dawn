26 bool PassSSA::run(StencilInstantiation* stencilInstantiation) {
27  OptimizerContext* context = stencilInstantiation->getOptimizerContext();
28 
29  if(context->getOptions().SSA) {
30 
31  for(auto& stencilPtr : stencilInstantiation->getStencils()) {
32  Stencil& stencil = *stencilPtr;
33 
34  std::shared_ptr<DependencyGraphAccesses> DAG =
35  std::make_shared<DependencyGraphAccesses>(stencilInstantiation);
36 
37  std::unordered_set<int> tochedAccessIDs;
38 
39  // Iterate each statement of the stencil (top -> bottom)
40  for(int stageIdx = 0; stageIdx < stencil.getNumStages(); ++stageIdx) {
41  std::shared_ptr<Stage> stagePtr = stencil.getStage(stageIdx);
42 
43  DoMethod& doMethod = stagePtr->getSingleDoMethod();
44  for(int stmtIdx = 0; stmtIdx < doMethod.getStatementAccessesPairs().size(); ++stmtIdx) {
45 
46  std::shared_ptr<StatementAccessesPair> stmtAccessesPair =
47  doMethod.getStatementAccessesPairs()[stmtIdx];
48 
49  AssignmentExpr* assignment = nullptr;
50  if(ExprStmt* stmt = dyn_cast<ExprStmt>(stmtAccessesPair->getStatement()->ASTStmt.get()))
51  assignment = dyn_cast<AssignmentExpr>(stmt->getExpr().get());
52 
53  std::vector<int> AccessIDsToRename;
54 
55  for(const std::pair<int, Extents>& readAccess :
56  stmtAccessesPair->getAccesses()->getReadAccesses()) {
57  int AccessID = readAccess.first;
58  if(!tochedAccessIDs.count(AccessID))
59  tochedAccessIDs.insert(AccessID);
60  }
61 
62  // Every write to a field which was has been touched (read/written) will get a new version
63  for(const std::pair<int, Extents>& writeAccess :
64  stmtAccessesPair->getAccesses()->getWriteAccesses()) {
65 
66  int AccessID = writeAccess.first;
67 
68  if(tochedAccessIDs.count(AccessID))
69  // Field was already written to, rename it in all remaining occurences
70  AccessIDsToRename.push_back(AccessID);
71 
72  tochedAccessIDs.insert(AccessID);
73  }
74 
75  for(int AccessID : AccessIDsToRename)
76  tochedAccessIDs.insert(stencilInstantiation->createVersionAndRename(
77  AccessID, &stencil, stageIdx, stmtIdx, assignment->getLeft(),
78  StencilInstantiation::RD_Below));
79 
80  DAG->insertStatementAccessesPair(stmtAccessesPair);
81  }
82  }
83  }
84  }
85 
86  return true;
87 }
