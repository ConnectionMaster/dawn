210  bool isValid() const {
211  // Nullary twines always have Empty on the RHS.
212  if(isNullary() && getRHSKind() != EmptyKind)
213  return false;
214 
215  // Null should never appear on the RHS.
216  if(getRHSKind() == NullKind)
217  return false;
218 
219  // The RHS cannot be non-empty if the LHS is empty.
220  if(getRHSKind() != EmptyKind && getLHSKind() == EmptyKind)
221  return false;
222 
223  // A twine child should always be binary.
224  if(getLHSKind() == TwineKind && !lhs_.twine->isBinary())
225  return false;
226  if(getRHSKind() == TwineKind && !rhs_.twine->isBinary())
227  return false;
228 
229  return true;
230  }

461 inline Twine Twine::concat(const Twine& Suffix) const {
462  // Concatenation with null is null.
463  if(isNull() || Suffix.isNull())
464  return Twine(NullKind);
465 
466  // Concatenation with empty yields the other side.
467  if(isEmpty())
468  return Suffix;
469  if(Suffix.isEmpty())
470  return *this;
471 
472  // Otherwise we need to create a new node, taking care to fold in unary
473  // twines.
474  Child NewLHS, NewRHS;
475  NewLHS.twine = this;
476  NewRHS.twine = &Suffix;
477  NodeKind NewLHSKind = TwineKind, NewRHSKind = TwineKind;
478  if(isUnary()) {
479  NewLHS = lhs_;
480  NewLHSKind = getLHSKind();
481  }
482  if(Suffix.isUnary()) {
483  NewRHS = Suffix.lhs_;
484  NewRHSKind = Suffix.getLHSKind();
485  }
486 
487  return Twine(NewLHS, NewLHSKind, NewRHS, NewRHSKind);
488 }
