145  inline int compare(StringRef RHS) const {
146  // Check the prefix for a mismatch.
147  if(int Res = compareMemory(Data, RHS.Data, std::min(Length, RHS.Length)))
148  return Res < 0 ? -1 : 1;
149 
150  // Otherwise the prefixes match, so we only need to check the lengths.
151  if(Length == RHS.Length)
152  return 0;
153  return Length < RHS.Length ? -1 : 1;
154  }

237  inline size_t find(char C, size_t From = 0) const {
238  size_t FindBegin = std::min(From, Length);
239  if(FindBegin < Length) { // Avoid calling memchr with nullptr.
240  // Just forward to memchr, which is faster than a hand-rolled loop.
241  if(const void* P = ::memchr(Data + FindBegin, C, Length - FindBegin))
242  return static_cast<const char*>(P) - Data;
243  }
244  return npos;
245  }

256  inline size_t find_if(function_ref<bool(char)> F, size_t From = 0) const {
257  StringRef S = drop_front(From);
258  while(!S.empty()) {
259  if(F(S.front()))
260  return size() - S.size();
261  S = S.drop_front();
262  }
263  return npos;
264  }

385  long long LLVal;

392  template <typename T>
393  typename std::enable_if<!std::numeric_limits<T>::is_signed, bool>::type
394  getAsInteger(unsigned Radix, T& Result) const {
395  unsigned long long ULLVal;
396  // The additional cast to unsigned long long is required to avoid the
397  // Visual C++ warning C4805: '!=' : unsafe mix of type 'bool' and type
398  // 'unsigned __int64' when instantiating getAsInteger with T = bool.
399  if(getAsUnsignedInteger(*this, Radix, ULLVal) ||
400  static_cast<unsigned long long>(static_cast<T>(ULLVal)) != ULLVal)
401  return true;
402  Result = ULLVal;
403  return false;
404  }
