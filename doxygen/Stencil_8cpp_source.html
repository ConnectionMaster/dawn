77 bool Stencil::Lifetime::overlaps(const Stencil::Lifetime& other) const {
78  // Note: same stage but different Do-Method are treated as overlapping!
79 
80  bool lowerBoundOverlap = false;
81  if(Begin.StagePos == other.End.StagePos && Begin.DoMethodIndex != other.End.DoMethodIndex)
82  lowerBoundOverlap = true;
83  else
84  lowerBoundOverlap = Begin <= other.End;
85 
86  bool upperBoundOverlap = false;
87  if(other.Begin.StagePos == End.StagePos && other.Begin.DoMethodIndex != End.DoMethodIndex)
88  upperBoundOverlap = true;
89  else
90  upperBoundOverlap = other.Begin <= End;
91 
92  return lowerBoundOverlap && upperBoundOverlap;
93 }

213 Stencil::StagePosition Stencil::getPositionFromStageIndex(int stageIdx) const {
214  DAWN_ASSERT(!multistages_.empty());
215  if(stageIdx == -1)
216  return StagePosition(0, -1);
217 
218  int curIdx = 0, multiStageIdx = 0;
219  for(const auto& MS : multistages_) {
220 
221  // Is our stage in this multi-stage?
222  int numStages = MS->getStages().size();
223  if((curIdx + numStages) <= stageIdx) {
224  curIdx += numStages;
225  multiStageIdx++;
226  continue;
227  } else {
228  int stageOffset = stageIdx - curIdx;
229  DAWN_ASSERT_MSG(stageOffset < numStages, "invalid stage index");
230  return StagePosition(multiStageIdx, stageOffset);
231  }
232  }
233  dawn_unreachable("invalid stage index");
234 }

236 int Stencil::getStageIndexFromPosition(const Stencil::StagePosition& position) const {
237  auto curMSIt = multistages_.begin();
238  std::advance(curMSIt, position.MultiStageIndex);
239 
240  // Count the number of stages in the multistages before our current MS
241  int numStagesInMSBeforeCurMS =
242  std::accumulate(multistages_.begin(), curMSIt, int(0),
243  [&](int numStages, const std::shared_ptr<MultiStage>& MS) {
244  return numStages + MS->getStages().size();
245  });
246 
247  // Add the current stage offset
248  return numStagesInMSBeforeCurMS + position.StageOffset;
249 }

251 const std::shared_ptr<Stage>& Stencil::getStage(const StagePosition& position) const {
252  // Get the multi-stage ...
253  DAWN_ASSERT_MSG(position.MultiStageIndex < multistages_.size(), "invalid multi-stage index");
254  auto msIt = multistages_.begin();
255  std::advance(msIt, position.MultiStageIndex);
256  const auto& MS = *msIt;
257 
258  // ... and the requested stage inside the given multi-stage
259  DAWN_ASSERT_MSG(position.StageOffset == -1 || position.StageOffset < MS->getStages().size(),
260  "invalid stage offset");
261  auto stageIt = MS->getStages().begin();
262  std::advance(stageIt, position.StageOffset == -1 ? 0 : position.StageOffset);
263  return *stageIt;
264 }

266 const std::shared_ptr<Stage>& Stencil::getStage(int stageIdx) const {
267  int curIdx = 0;
268  for(const auto& MS : multistages_) {
269 
270  // Is our stage in this multi-stage?
271  int numStages = MS->getStages().size();
272 
273  if((curIdx + numStages) <= stageIdx) {
274  // No... continue
275  curIdx += numStages;
276  continue;
277  } else {
278  // Yes... advance to our stage
279  int stageOffset = stageIdx - curIdx;
280 
281  DAWN_ASSERT_MSG(stageOffset < MS->getStages().size(), "invalid stage index");
282  auto stageIt = MS->getStages().begin();
283  std::advance(stageIt, stageOffset);
284 
285  return *stageIt;
286  }
287  }
288  dawn_unreachable("invalid stage index");
289 }

291 void Stencil::insertStage(const StagePosition& position, const std::shared_ptr<Stage>& stage) {
292 
293  // Get the multi-stage ...
294  DAWN_ASSERT_MSG(position.MultiStageIndex < multistages_.size(), "invalid multi-stage index");
295  auto msIt = multistages_.begin();
296  std::advance(msIt, position.MultiStageIndex);
297  const auto& MS = *msIt;
298 
299  // ... and the requested stage inside the given multi-stage
300  DAWN_ASSERT_MSG(position.StageOffset == -1 || position.StageOffset < MS->getStages().size(),
301  "invalid stage offset");
302  auto stageIt = MS->getStages().begin();
303 
304  // A stage offset of -1 indicates *before* the first element (thus nothing to do).
305  // Otherwise we advance one beyond the requested stage as we insert *after* the specified
306  // stage and `std::list::insert` inserts *before*.
307  if(position.StageOffset != -1) {
308  std::advance(stageIt, position.StageOffset);
309  if(stageIt != MS->getStages().end())
310  stageIt++;
311  }
312 
313  MS->getStages().insert(stageIt, stage);
314 }

344 std::unordered_map<int, Stencil::Lifetime>
345 Stencil::getLifetime(const std::unordered_set<int>& AccessIDs) const {
346  std::unordered_map<int, StatementPosition> Begin;
347  std::unordered_map<int, StatementPosition> End;
348 
349  int multiStageIdx = 0;
350  for(const auto& multistagePtr : multistages_) {
351 
352  int stageOffset = 0;
353  for(const auto& stagePtr : multistagePtr->getStages()) {
354 
355  int doMethodIndex = 0;
356  for(const auto& doMethodPtr : stagePtr->getDoMethods()) {
357  DoMethod& doMethod = *doMethodPtr;
358 
359  for(int statementIdx = 0; statementIdx < doMethod.getStatementAccessesPairs().size();
360  ++statementIdx) {
361  const Accesses& accesses =
362  *doMethod.getStatementAccessesPairs()[statementIdx]->getAccesses();
363 
364  auto processAccessMap = [&](const std::unordered_map<int, Extents>& accessMap) {
365  for(const auto& AccessIDExtentPair : accessMap) {
366  int AccessID = AccessIDExtentPair.first;
367 
368  if(AccessIDs.count(AccessID)) {
369  StatementPosition pos(StagePosition(multiStageIdx, stageOffset), doMethodIndex,
370  statementIdx);
371 
372  if(!Begin.count(AccessID))
373  Begin.emplace(AccessID, pos);
374  End[AccessID] = pos;
375  }
376  }
377  };
378 
379  processAccessMap(accesses.getWriteAccesses());
380  processAccessMap(accesses.getReadAccesses());
381  }
382 
383  doMethodIndex++;
384  }
385 
386  stageOffset++;
387  }
388 
389  multiStageIdx++;
390  }
391 
392  std::unordered_map<int, Lifetime> lifetimeMap;
393  for(int AccessID : AccessIDs) {
394  auto& begin = Begin[AccessID];
395  auto& end = End[AccessID];
396 
397  lifetimeMap.emplace(AccessID, Lifetime(begin, end));
398  }
399 
400  return lifetimeMap;
401 }
