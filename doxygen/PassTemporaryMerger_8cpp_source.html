28 bool PassTemporaryMerger::run(StencilInstantiation* stencilInstantiation) {
29  using Edge = DependencyGraphAccesses::Edge;
30  using Vertex = DependencyGraphAccesses::Vertex;
31 
32  OptimizerContext* context = stencilInstantiation->getOptimizerContext();
33  bool merged = false;
34 
35  bool stencilNeedsMergePass = false;
36  for(const auto& stencilPtr : stencilInstantiation->getStencils())
37  stencilNeedsMergePass |=
38  stencilPtr->getSIRStencil()->Attributes.has(sir::Attr::AK_MergeTemporaries);
39 
40  if(!(context->getOptions().MergeTemporaries || stencilNeedsMergePass))
41  return true;
42 
43  // Pair of nodes to visit and AccessID of the last temporary (or -1 if no temporary has been
44  // processed yet)
45  std::vector<std::pair<std::size_t, int>> nodesToVisit;
46  std::unordered_set<std::size_t> visitedNodes;
47 
48  int stencilIdx = 0;
49  for(const auto& stencilPtr : stencilInstantiation->getStencils()) {
50  Stencil& stencil = *stencilPtr;
51 
52  // Build the dependency graph of the stencil (merge all dependency graphs of the multi-stages)
53  DependencyGraphAccesses AccessesDAG(stencilInstantiation);
54  for(const auto& multiStagePtr : stencilPtr->getMultiStages()) {
55  MultiStage& multiStage = *multiStagePtr;
56  AccessesDAG.merge(multiStage.getDependencyGraphOfAxis().get());
57  }
58  const auto& adjacencyList = AccessesDAG.getAdjacencyList();
59 
60  // Build the dependency graph of the temporaries
61  DependencyGraphAccesses TemporaryDAG(stencilInstantiation);
62  int AccessIDOfLastTemporary = -1;
63  for(std::size_t VertexID : AccessesDAG.getOutputVertexIDs()) {
64  nodesToVisit.clear();
65  visitedNodes.clear();
66 
67  nodesToVisit.emplace_back(VertexID, AccessIDOfLastTemporary = -1);
68  while(!nodesToVisit.empty()) {
69 
70  // Process the current node
71  std::size_t FromVertexID = nodesToVisit.back().first;
72  int FromAccessID = AccessesDAG.getIDFromVertexID(FromVertexID);
73  AccessIDOfLastTemporary = nodesToVisit.back().second;
74  nodesToVisit.pop_back();
75 
76  if(stencilInstantiation->isTemporaryField(FromAccessID)) {
77  TemporaryDAG.insertNode(FromAccessID);
78  AccessIDOfLastTemporary = FromAccessID;
79  }
80 
81  // Check if we already visited this node
82  if(visitedNodes.count(FromVertexID))
83  continue;
84  else
85  visitedNodes.insert(FromVertexID);
86 
87  // Follow edges of the current node and update the node extents
88  for(const Edge& edge : *adjacencyList[FromVertexID]) {
89  std::size_t ToVertexID = edge.ToVertexID;
90  int ToAccessID = AccessesDAG.getIDFromVertexID(ToVertexID);
91  int newAccessIDOfLastTemporary = AccessIDOfLastTemporary;
92 
93  if(stencilInstantiation->isTemporaryField(ToAccessID) && AccessIDOfLastTemporary != -1) {
94  TemporaryDAG.insertEdge(AccessIDOfLastTemporary, ToAccessID);
95  newAccessIDOfLastTemporary = ToAccessID;
96  }
97 
98  nodesToVisit.emplace_back(ToVertexID, newAccessIDOfLastTemporary);
99  }
100  }
101  }
102 
103  std::cout << TemporaryDAG.toDot() << std::endl;
104 
105  if(TemporaryDAG.empty())
106  continue;
107 
108  // Add dependencies due to overlapping lifetime of the temporaries. The dependencies will assure
109  // that these temporaries can not be merged into one i.e have different colors.
110  std::unordered_set<int> temporaries;
111  std::for_each(TemporaryDAG.getVertices().begin(), TemporaryDAG.getVertices().end(),
112  [&](const std::pair<int, Vertex>& vertexPair) {
113  temporaries.emplace(vertexPair.second.ID);
114  });
115  auto LifeTimeMap = stencil.getLifetime(temporaries);
116 
117  for(const auto& FromAccessIDLifetimePair : LifeTimeMap) {
118  const int FromAccessID = FromAccessIDLifetimePair.first;
119  const Stencil::Lifetime& FromLifetime = FromAccessIDLifetimePair.second;
120 
121  for(const auto& ToAccessIDLifetimePair : LifeTimeMap) {
122  const int ToAccessID = ToAccessIDLifetimePair.first;
123  const Stencil::Lifetime& ToLifetime = ToAccessIDLifetimePair.second;
124 
125  if(FromAccessID == ToAccessID)
126  continue;
127 
128  if(FromLifetime.overlaps(ToLifetime)) {
129  TemporaryDAG.insertEdge(FromAccessID, ToAccessID);
130  }
131  }
132  }
133 
134  if(context->getOptions().DumpTemporaryGraphs)
135  TemporaryDAG.toDot(format("tmp_stencil_%i.dot", stencilIdx));
136 
137  // Color the temporary graph
138  std::unordered_map<int, int> coloring;
139  TemporaryDAG.greedyColoring(coloring);
140 
141  // Compute the candiates for renaming. We will rename all fields of the same color to the same
142  // AccessID.
143  std::unordered_map<int, std::vector<int>> colorToAccessIDOfRenameCandidatesMap;
144  for(const auto& colorAccessIDPair : coloring) {
145  int AccessID = colorAccessIDPair.first;
146  int color = colorAccessIDPair.second;
147  auto it = colorToAccessIDOfRenameCandidatesMap.find(color);
148  if(it == colorToAccessIDOfRenameCandidatesMap.end())
149  colorToAccessIDOfRenameCandidatesMap.emplace(color, std::vector<int>{AccessID});
150  else
151  colorToAccessIDOfRenameCandidatesMap[color].push_back(AccessID);
152  }
153 
154  for(const auto& colorRenameCandidatesPair : colorToAccessIDOfRenameCandidatesMap) {
155  const std::vector<int>& AccessIDOfRenameCandiates = colorRenameCandidatesPair.second;
156 
157  // Print the rename candiates in alphabetical order
158  if(context->getOptions().ReportPassTemporaryMerger && AccessIDOfRenameCandiates.size() >= 2) {
159  std::vector<std::string> renameCandiatesNames;
160  for(int AccessID : AccessIDOfRenameCandiates)
161  renameCandiatesNames.emplace_back(stencilInstantiation->getNameFromAccessID(AccessID));
162  std::sort(renameCandiatesNames.begin(), renameCandiatesNames.end());
163  std::cout << "\nPASS: " << getName() << ": " << stencilInstantiation->getName()
164  << ": merging: " << RangeToString(", ", "", "\n")(renameCandiatesNames);
165  }
166 
167  int newAccessID = AccessIDOfRenameCandiates[0];
168 
169  // Rename all other fields of the color to the AccessID of the first field (note that it
170  // wouldn't matter which AccessID we choose)
171  for(int i = 1; i < AccessIDOfRenameCandiates.size(); ++i) {
172  merged = true;
173  int oldAccessID = AccessIDOfRenameCandiates[i];
174  stencilInstantiation->renameAllOccurrences(stencilPtr.get(), oldAccessID, newAccessID);
175  }
176  }
177 
178  stencilIdx++;
179  }
180 
181  if(context->getOptions().ReportPassTemporaryMerger && !merged)
182  std::cout << "\nPASS: " << getName() << ": " << stencilInstantiation->getName()
183  << ": no merge\n";
184 
185  return true;
186 }
