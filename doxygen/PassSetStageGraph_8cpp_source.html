25 static bool depends(const Stage& fromStage, const Stage& toStage) {
26  if(!fromStage.overlaps(toStage))
27  return false;
28 
29  bool intervalsMatch = fromStage.getEnclosingInterval() == toStage.getEnclosingInterval();
30 
31  for(const Field& fromField : fromStage.getFields()) {
32  for(const Field& toField : toStage.getFields()) {
33 
34  if(fromField.AccessID != toField.AccessID)
35  continue;
36 
37  Field::IntendKind fromFieldIntend = fromField.Intend;
38  Field::IntendKind toFieldIntend = toField.Intend;
39 
40  switch(fromFieldIntend) {
41  case Field::IK_Output:
42  if(!intervalsMatch || toFieldIntend == Field::IK_Input ||
43  toFieldIntend == Field::IK_InputOutput)
44  return true;
45  break;
46  case Field::IK_InputOutput:
47  return true;
48  case Field::IK_Input:
49  if(toFieldIntend == Field::IK_Output || toFieldIntend == Field::IK_InputOutput)
50  return true;
51  break;
52  }
53  }
54  }
55  return false;
56 }

62 bool PassSetStageGraph::run(StencilInstantiation* stencilInstantiation) {
63  OptimizerContext* context = stencilInstantiation->getOptimizerContext();
64 
65  int stencilIdx = 0;
66  for(auto& stencilPtr : stencilInstantiation->getStencils()) {
67  Stencil& stencil = *stencilPtr;
68  int numStages = stencil.getNumStages();
69 
70  auto stageDAG = std::make_shared<DependencyGraphStage>(stencilInstantiation);
71 
72  // Build DAG of stages (backward sweep)
73  for(int i = numStages - 1; i >= 0; --i) {
74  auto fromStagePtr = stencil.getStage(i);
75  stageDAG->insertNode(fromStagePtr->getStageID());
76  int curStageID = fromStagePtr->getStageID();
77 
78  for(int j = i - 1; j >= 0; --j) {
79  auto toStagePtr = stencil.getStage(j);
80  if(depends(*fromStagePtr, *toStagePtr))
81  stageDAG->insertEdge(curStageID, toStagePtr->getStageID());
82  }
83  }
84 
85  if(context->getOptions().DumpStageGraph)
86  stageDAG->toDot("stage_" + stencilInstantiation->getName() + "_s" +
87  std::to_string(stencilIdx) + ".dot");
88 
89  stencil.setStageDependencyGraph(std::move(stageDAG));
90  }
91 
92  return true;
93 }
