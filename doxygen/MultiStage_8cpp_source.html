28 std::vector<std::shared_ptr<MultiStage>>
29 MultiStage::split(std::deque<MultiStage::SplitIndex>& splitterIndices,
30  LoopOrderKind lastLoopOrder) {
31 
32  std::vector<std::shared_ptr<MultiStage>> newMultiStages;
33 
34  int curStageIndex = 0;
35  auto curStageIt = stages_.begin();
36  std::deque<int> curStageSplitterIndices;
37 
38  newMultiStages.push_back(std::make_shared<MultiStage>(stencilInstantiation_, lastLoopOrder));
39 
40  for(std::size_t i = 0; i < splitterIndices.size(); ++i) {
41  SplitIndex& splitIndex = splitterIndices[i];
42 
43  if(splitIndex.StageIndex == curStageIndex) {
44 
45  curStageSplitterIndices.push_back(splitIndex.StmtIndex);
46  newMultiStages.push_back(
47  std::make_shared<MultiStage>(stencilInstantiation_, splitIndex.LowerLoopOrder));
48  lastLoopOrder = splitIndex.LowerLoopOrder;
49  }
50 
51  if(i == (splitterIndices.size() - 1) || splitIndex.StageIndex != curStageIndex) {
52  if(!curStageSplitterIndices.empty()) {
53 
54  // Split the current stage (we assume the graphs are assigned in the stage splitter pass)
55  auto newStages = (**curStageIt).split(curStageSplitterIndices, nullptr);
56 
57  // Move the new stages to the new MultiStage
58  auto newMultiStageRIt = newMultiStages.rbegin();
59  for(auto newStagesRIt = newStages.rbegin(); newStagesRIt != newStages.rend();
60  ++newStagesRIt, ++newMultiStageRIt)
61  (*newMultiStageRIt)->getStages().push_back(std::move(*newStagesRIt));
62 
63  curStageSplitterIndices.clear();
64  } else {
65  // No split in this stage, just move it to the current multi-stage
66  newMultiStages.back()->getStages().push_back(std::move(*curStageIt));
67  }
68 
69  if(i != (splitterIndices.size() - 1))
70  newMultiStages.push_back(
71  std::make_shared<MultiStage>(stencilInstantiation_, lastLoopOrder));
72 
73  // Handle the next stage
74  curStageIndex++;
75  curStageIt++;
76  }
77  }
78 
79  return newMultiStages;
80 }

85  std::for_each(stages_.begin(), stages_.end(), [&](const std::shared_ptr<Stage>& stagePtr) {

87  std::for_each(stagePtr->getDoMethods().begin(), stagePtr->getDoMethods().end(),

128 std::unordered_map<int, Field> MultiStage::getFields() const {
129  std::unordered_map<int, Field> fields;
130 
131  for(const auto& stagePtr : stages_) {
132  for(const Field& field : stagePtr->getFields()) {
133  auto it = fields.find(field.AccessID);
134  if(it != fields.end()) {
135  // Adjust the Intend
136  if(it->second.Intend == Field::IK_Input && field.Intend == Field::IK_Output)
137  it->second.Intend = Field::IK_InputOutput;
138  else if(it->second.Intend == Field::IK_Output && field.Intend == Field::IK_Input)
139  it->second.Intend = Field::IK_InputOutput;
140 
141  // Merge the Extent
142  it->second.Extent.merge(field.Extent);
143  } else
144  fields.emplace(field.AccessID, field);
145  }
146  }
147 
148  return fields;
149 }
