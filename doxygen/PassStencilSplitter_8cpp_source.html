31 static int mergePossible(const std::set<int>& fields, const Stage* stage, int maxNumFields) {
32  int numFields = fields.size();
33 
34  for(const Field& field : stage->getFields())
35  if(!fields.count(field.AccessID))
36  numFields++;
37 
38  // Inserting the stage would further increase the number of fields
39  if(fields.size() > maxNumFields)
40  return numFields == fields.size();
41 
42  return numFields <= maxNumFields;
43 }

49 bool PassStencilSplitter::run(StencilInstantiation* stencilInstantiation) {
50  OptimizerContext* context = stencilInstantiation->getOptimizerContext();
51 
52  if(context->getOptions().SplitStencils) {
53 
54  // If we split a stencil, we need to recompute the stage graphs
55  bool rerunPassSetStageGraph = false;
56 
57  for(auto stencilIt = stencilInstantiation->getStencils().begin();
58  stencilIt != stencilInstantiation->getStencils().end(); ++stencilIt) {
59  Stencil& stencil = **stencilIt;
60 
61  // New stencils which serve as a replacement for `stencil`
62  std::vector<std::shared_ptr<Stencil>> newStencils;
63 
64  // If a stencil exceeds the threshold, we need to split it
65  if(stencil.getFields().size() > MaxFieldPerStencil) {
66  rerunPassSetStageGraph = true;
67 
68  newStencils.emplace_back(std::make_shared<Stencil>(
69  stencilInstantiation, stencil.getSIRStencil(), stencilInstantiation->nextUID()));
70  std::shared_ptr<Stencil> newStencil = newStencils.back();
71 
72  std::set<int> fieldsInNewStencil;
73 
74  // Iterate the multi-stage of the old `stencil` and insert its stages into `newStencil`
75  for(const auto& multiStagePtr : stencil.getMultiStages()) {
76  MultiStage& multiStage = *multiStagePtr;
77 
78  // Create an empty multi-stage in the current stencil with the same parameter as
79  // `multiStage`
80  newStencil->getMultiStages().push_back(
81  std::make_shared<MultiStage>(stencilInstantiation, multiStage.getLoopOrder()));
82 
83  for(std::shared_ptr<Stage>& stagePtr : multiStage.getStages()) {
84  if(newStencil->isEmpty() ||
85  mergePossible(fieldsInNewStencil, stagePtr.get(), MaxFieldPerStencil)) {
86 
87  // We can safely insert the stage into the current multi-stage of the `newStencil`
88  newStencil->getMultiStages().back()->getStages().push_back(stagePtr);
89 
90  // Update fields of the `newStencil`. Note that the indivudual stages do not need to
91  // update their fields as they remain the same.
92  for(const Field& field : stagePtr->getFields())
93  fieldsInNewStencil.insert(field.AccessID);
94 
95  } else {
96  // Make a new stencil
97  newStencils.emplace_back(std::make_shared<Stencil>(
98  stencilInstantiation, stencil.getSIRStencil(), stencilInstantiation->nextUID()));
99  newStencil = newStencils.back();
100  fieldsInNewStencil.clear();
101 
102  // Re-create the current multi-stage in the `newStencil` and insert the stage
103  newStencil->getMultiStages().push_back(
104  std::make_shared<MultiStage>(stencilInstantiation, multiStage.getLoopOrder()));
105  newStencil->getMultiStages().back()->getStages().push_back(stagePtr);
106  }
107  }
108  }
109  }
110 
111  // Replace the current stencil with the new stencils and update the stencil description AST
112  if(!newStencils.empty()) {
113 
114  // Repair broken references to temporaries i.e promote them to real fields
115  PassTemporaryType::fixTemporariesSpanningMultipleStencils(stencilInstantiation,
116  newStencils);
117 
118  // Remove empty multi-stages within the stencils
119  for(auto& s : newStencils) {
120  for(auto msIt = s->getMultiStages().begin(); msIt != s->getMultiStages().end();)
121  if((*msIt)->isEmpty())
122  msIt = s->getMultiStages().erase(msIt);
123  else
124  ++msIt;
125  }
126 
127  // Update the fields of the stencil (this is not strictly necessary but it might be a source
128  // of error in the future when updateFields also changes data-structures in the
129  // Multi-Stage/Stencil)
130  for(auto& s : newStencils)
131  s->updateFields();
132 
133  // Update the calls to this stencil in the stencil description AST
134  std::vector<int> newStencilIDs;
135  for(const auto& s : newStencils)
136  newStencilIDs.push_back(s->getStencilID());
137 
138  replaceStencilCalls(stencilInstantiation, stencil.getStencilID(), newStencilIDs);
139 
140  // Erase the old stencil ...
141  stencilIt = stencilInstantiation->getStencils().erase(stencilIt);
142 
143  // ... and insert the new ones
144  stencilIt = stencilInstantiation->getStencils().insert(stencilIt, newStencils.begin(),
145  newStencils.end());
146  std::advance(stencilIt, newStencils.size() - 1);
147  }
148  }
149 
150  // Recompute the stage graph of each stencil
151  if(rerunPassSetStageGraph) {
152  PassSetStageGraph pass;
153  pass.run(stencilInstantiation);
154  }
155  }
156 
157  return true;
158 }
