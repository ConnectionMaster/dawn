29 class UnusedFieldVisitor : public ASTVisitorForwarding {
30  int AccessID_;
31  bool fieldIsUnused_;
32  StencilInstantiation* instantiation_;
33  std::stack<StencilFunctionInstantiation*> functionInstantiationStack_;
34 
35 public:
36  UnusedFieldVisitor(int AccessID, StencilInstantiation* instantiation)
37  : AccessID_(AccessID), fieldIsUnused_(false), instantiation_(instantiation) {}
38 
39  StencilFunctionInstantiation*
40  getStencilFunctionInstantiation(const std::shared_ptr<StencilFunCallExpr>& expr) {
41  if(!functionInstantiationStack_.empty())
42  return functionInstantiationStack_.top()->getStencilFunctionInstantiation(expr);
43  return instantiation_->getStencilFunctionInstantiation(expr);
44  }
45 
46  void visit(const std::shared_ptr<StencilFunCallExpr>& expr) override {
47  StencilFunctionInstantiation* funCall = getStencilFunctionInstantiation(expr);
48 
49  functionInstantiationStack_.push(funCall);
50  fieldIsUnused_ |= funCall->isFieldUnused(AccessID_);
51 
52  // Follow the AST of the stencil function, it maybe unused in a nested stencil function
53  funCall->getAST()->accept(*this);
54 
55  // Visit arguments
56  ASTVisitorForwarding::visit(expr);
57  functionInstantiationStack_.pop();
58  }
59 
60  bool isFieldUnused() const { return fieldIsUnused_; }
61 };

67 bool PassTemporaryFirstAccess::run(StencilInstantiation* stencilInstantiation) {
68  OptimizerContext* context = stencilInstantiation->getOptimizerContext();
69 
70  for(auto& stencilPtr : stencilInstantiation->getStencils()) {
71  std::vector<Stencil::FieldInfo> fields = stencilPtr->getFields();
72  std::set<int> temporaryFields;
73 
74  for(int i = 0; i < fields.size(); ++i)
75  if(fields[i].IsTemporary)
76  temporaryFields.insert(stencilInstantiation->getAccessIDFromName(fields[i].Name));
77 
78  // {AccesID : (isFirstAccessWrite, Stmt)}
79  std::unordered_map<int, std::pair<bool, std::shared_ptr<Stmt>>> accessMap;
80 
81  for(auto& multiStagePtr : stencilPtr->getMultiStages()) {
82  for(auto& stagePtr : multiStagePtr->getStages()) {
83  DoMethod& doMethod = stagePtr->getSingleDoMethod();
84 
85  for(int i = 0; i < doMethod.getStatementAccessesPairs().size(); ++i) {
86  const auto& accesses = doMethod.getStatementAccessesPairs()[i]->getAccesses();
87  const auto& astStatement =
88  doMethod.getStatementAccessesPairs()[i]->getStatement()->ASTStmt;
89 
90  for(const auto& writeAccess : accesses->getWriteAccesses())
91  if(temporaryFields.count(writeAccess.first))
92  accessMap.emplace(writeAccess.first, std::make_pair(true, astStatement));
93 
94  for(const auto& readAccess : accesses->getReadAccesses())
95  if(temporaryFields.count(readAccess.first))
96  accessMap.emplace(readAccess.first, std::make_pair(false, astStatement));
97  }
98  }
99  }
100 
101  for(const auto& accessPair : accessMap) {
102 
103  // Is first access to the temporary a read?
104  if(accessPair.second.first == false) {
105  int AccessID = accessPair.first;
106  const std::shared_ptr<Stmt>& stmt = accessPair.second.second;
107 
108  // Check if the statment contains a stencil function call in which the field is unused.
109  // If a field is unused, we still have to add it as an input field of the stencil function.
110  // Thus, the uninitialized temporary anaylsis reports a false positive on this temporary
111  // field!
112  UnusedFieldVisitor visitor(AccessID, stencilInstantiation);
113  stmt->accept(visitor);
114  if(visitor.isFieldUnused())
115  continue;
116 
117  // Report the error
118  auto nameLocPair =
119  stencilInstantiation->getOriginalNameAndLocationsFromAccessID(AccessID, stmt);
120  DiagnosticsBuilder diagError(DiagnosticsKind::Error, nameLocPair.second[0]);
121 
122  diagError << "access to uninitialized temporary storage '" << nameLocPair.first << "'";
123  context->getDiagnostics().report(diagError);
124 
125  // Report notes where the temporary is referenced
126  for(int i = 1; i < nameLocPair.second.size(); ++i) {
127  DiagnosticsBuilder diagNote(DiagnosticsKind::Note, nameLocPair.second[i]);
128  diagNote << "'" << nameLocPair.first << "' referenced here";
129  context->getDiagnostics().report(diagNote);
130  }
131 
132  return false;
133  }
134  }
135  }
136 
137  return true;
138 }
