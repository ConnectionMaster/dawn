32 inline std::ostream& indent(int level, std::stringstream& ss) {
33  switch(level) {
34  case 0:
35  return (ss << MakeIndent<0>::value);
36  case 1:
37  return (ss << MakeIndent<1>::value);

39  return (ss << MakeIndent<2>::value);

143  ~ClassName() { \

311  class ValueType = typename std::decay<Sequence>::type::value_type>
312  MemberFunction& addArgs(const Twine& typeName, Sequence&& sequence,
313  StrinfigyFunctor&& stringifier) {
314  DAWN_ASSERT(!IsBodyDeclared);
315  std::string typeNameStr = internal::twineToStr(typeName);
316  ss() << RangeToString(", ", (NumArgs++ == 0 ? "(" : ", "), "")(
317  sequence,

319  return *this;
320  }

324  return addArgs(typeName, std::forward<Sequence>(sequence),
325  RangeToString::DefaultStringifyFunctor());
326  }

330  MemberFunction& startBody() {
331  if(!IsBodyDeclared) {
332  finishArgs();
333  ss() << "{\n";
334  IsBodyDeclared = true;
335  }
336  return *this;
337  }
338 
340  MemberFunction& finishArgs() {
341  DAWN_ASSERT(!IsBodyDeclared);
342  if(!AreArgsFinished) {
343  ss() << (NumArgs == 0 ? "() " : ") ");
344  if(IsConst)

376  template <class BlockBodyFunType>
377  MemberFunction& addBlockStatement(const Twine& init, BlockBodyFunType&& blockBodyFun) {
378  startBody();

407  void commitImpl() {
408  // If there is a body, check if we added one or add an empty one `{}`
409  if(CanHaveBody) {
410  if(!IsBodyDeclared) {
411  if(NumInits == 0)
412  ss() << ") ";
413  ss() << "{}";
414  } else
415  indentImpl(IndentLevel) << "}";
416  }
417  }
