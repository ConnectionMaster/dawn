33 // Define a template that can be specialized by smart pointers to reflect the fact that they are
34 // automatically dereferenced, and are not involved with the template selection process... the
35 // default implementation is a noop.
36 template <typename From>
37 struct simplify_type {
38  typedef From SimpleType; // The real type this represents...
39 
40  // An accessor to get the real value...
41  static SimpleType& getSimplifiedValue(From& Val) { return Val; }
42 };
43 
44 template <typename From>
45 struct simplify_type<const From> {
46  typedef typename simplify_type<From>::SimpleType NonConstSimpleType;
47  typedef typename add_const_past_pointer<NonConstSimpleType>::type SimpleType;
48  typedef typename add_lvalue_reference_if_not_pointer<SimpleType>::type RetType;
49  static RetType getSimplifiedValue(const From& Val) {
50  return simplify_type<From>::getSimplifiedValue(const_cast<From&>(Val));
51  }
52 };

54 // The core of the implementation of isa<X> is here; To and From should be the names of classes.
55 // This template can be specialized to customize the implementation of isa<> without rewriting it
56 // from scratch.
57 template <typename To, typename From, typename Enabler = void>
58 struct isa_impl {
59  static inline bool doit(const From& Val) { return To::classof(&Val); }
60 };

62 // Always allow upcasts, and perform no dynamic check for them.
63 template <typename To, typename From>
64 struct isa_impl<To, From, typename std::enable_if<std::is_base_of<To, From>::value>::type> {
65  static inline bool doit(const From&) { return true; }
66 };

110 template <typename To, typename From, typename SimpleFrom>
111 struct isa_impl_wrap {
112  // When From != SimplifiedType, we can simplify the type some more by using the simplify_type
113  // template.
114  static bool doit(const From& Val) {
115  return isa_impl_wrap<To, SimpleFrom, typename simplify_type<SimpleFrom>::SimpleType>::doit(
116  simplify_type<const From>::getSimplifiedValue(Val));
117  }

123  static bool doit(const FromTy& Val) { return isa_impl_cl<To, FromTy>::doit(Val); }

189 // Ensure the non-simple values are converted using the simplify_type template that may be
190 // specialized by smart pointers...
191 template <class To, class From, class SimpleFrom>
192 struct cast_convert_val {
193  // This is not a simple type, use the template to simplify it...
194  static typename cast_retty<To, From>::ret_type doit(From& Val) {
195  return cast_convert_val<To, SimpleFrom, typename simplify_type<SimpleFrom>::SimpleType>::doit(
196  simplify_type<From>::getSimplifiedValue(Val));
197  }

201 struct cast_convert_val<To, FromTy, FromTy> {
202  // This _is_ a simple type, just cast it.
203  static typename cast_retty<To, FromTy>::ret_type doit(const FromTy& Val) {
204  typename cast_retty<To, FromTy>::ret_type Res2 =
