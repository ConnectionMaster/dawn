28 template <class InstantiationType>
29 static std::string toStringImpl(const StatementAccessesPair* pair,
30  const InstantiationType* instantiation, std::size_t initialIndent) {
31  std::stringstream ss;
32 
33  std::string initialIndentStr = std::string(initialIndent, ' ');
34  std::size_t curIndent = initialIndent + DAWN_PRINT_INDENT;
35  std::string curIndentStr = std::string(curIndent, ' ');
36 
37  ss << initialIndentStr << "[\n" << curIndentStr << "Statement:\n";
38  ss << "\e[1m"
39  << ASTStringifer::toString(pair->getStatement()->ASTStmt, curIndent + DAWN_PRINT_INDENT)
40  << "\e[0m\n";
41 
42  if(pair->getCallerAccesses()) {
43  ss << curIndentStr << "CallerAccesses:\n";
44  ss << pair->getCallerAccesses()->toString(instantiation, curIndent + DAWN_PRINT_INDENT) << "\n";
45  }
46 
47  if(pair->getCalleeAccesses()) {
48  ss << curIndentStr << "CalleeAccesses:\n";
49  ss << pair->getCalleeAccesses()->toString(instantiation, curIndent + DAWN_PRINT_INDENT) << "\n";
50  }
51 
52  if(!pair->getChildren().empty()) {
53  ss << curIndentStr << "Children:\n";
54  for(auto& child : pair->getChildren())
55  ss << child->toString(instantiation, curIndent);
56  }
57  ss << initialIndentStr << "]\n";
58 
59  return ss.str();
60 }
