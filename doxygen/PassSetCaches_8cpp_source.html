29 enum FirstAccessKind {
30  FK_Unknown = 0,
31  FK_WriteOnly = 1,
32  FK_ReadOnly = 2,
33  FK_ReadAndWrite = 4
34 };

40 static FirstAccessKind getFirstAccessKind(const MultiStage& MS, int AccessID) {
41 
42  // Get the stage of the first access
43  for(const auto& stagePtr : MS.getStages()) {
44  Stage& stage = *stagePtr;
45  if(std::find_if(stage.getFields().begin(), stage.getFields().end(), [&](const Field& field) {
46  return field.AccessID == AccessID;
47  }) != stage.getFields().end()) {
48 
49  auto getFirstAccessKindFromDoMethod = [&](const DoMethod* doMethod) -> FirstAccessKind {
50  for(const auto& statementAccesssPair : doMethod->getStatementAccessesPairs()) {
51  const Accesses& accesses = *statementAccesssPair->getAccesses();
52 
53  for(const auto& writeAccessIDExtentPair : accesses.getWriteAccesses()) {
54  // Storage has a write access, does it have a read access as well?
55  if(writeAccessIDExtentPair.first == AccessID) {
56  for(const auto& readAccessIDExtentPair : accesses.getReadAccesses())
57  // Yes!
58  if(readAccessIDExtentPair.first == AccessID)
59  return FK_ReadAndWrite;
60  // No!
61  return FK_WriteOnly;
62  }
63  }
64 
65  for(const auto& readAccessIDExtentPair : accesses.getReadAccesses())
66  // First access is read-only
67  if(readAccessIDExtentPair.first == AccessID)
68  return FK_ReadOnly;
69  }
70  // Storage not referenced in this Do-Method
71  return FK_Unknown;
72  };
73 
74  // We need to check all Do-Methods
75  if(MS.getLoopOrder() == LoopOrderKind::LK_Parallel) {
76  FirstAccessKind firstAccess = FK_Unknown;
77  for(const auto& doMethodPtr : stage.getDoMethods())
78  firstAccess = static_cast<FirstAccessKind>(
79  firstAccess | getFirstAccessKindFromDoMethod(doMethodPtr.get()));
80  return firstAccess;
81 
82  } else {
83  // Forward and backward loop orders
84  std::vector<DoMethod*> doMethods;
85  std::transform(stage.getDoMethods().begin(), stage.getDoMethods().end(),
86  std::back_inserter(doMethods),
87  [](const std::unique_ptr<DoMethod>& D) { return D.get(); });
88 
89  if(MS.getLoopOrder() == LoopOrderKind::LK_Forward) {
90  // Sort the Do-Method in ascending order (i.e lowest interval is first)
91  std::sort(doMethods.begin(), doMethods.end(), [](DoMethod* D1, DoMethod* D2) {
92  return D1->getInterval().lowerBound() < D2->getInterval().lowerBound();
93  });
94  } else {
95  // Sort the Do-Method in descending order (i.e top interval is first)
96  std::sort(doMethods.begin(), doMethods.end(), [](DoMethod* D1, DoMethod* D2) {
97  return D1->getInterval().upperBound() > D2->getInterval().upperBound();
98  });
99  }
100 
101  for(DoMethod* D : doMethods) {
102  FirstAccessKind firstAccess = getFirstAccessKindFromDoMethod(D);
103  if(firstAccess != FK_Unknown)
104  return firstAccess;
105  }
106  }
107  }
108  }
109  dawn_unreachable("MultiStage does not contain the given field");
110 }

129 Cache::CacheIOPolicy combinePolicy(Cache::CacheIOPolicy MS1Policy, Cache::CacheIOPolicy MS2Policy) {
130  if(MS1Policy == Cache::local) {
131  if(MS2Policy == Cache::flush || MS2Policy == Cache::epflush)
132  return MS2Policy; // flush, epflush
133  else if(MS2Policy == Cache::local)
134  return MS2Policy; // local
135  } else if(MS1Policy == Cache::fill || MS1Policy == Cache::bpfill) {
136  if(MS2Policy == Cache::flush || MS2Policy == Cache::epflush)
137  return Cache::fill_and_flush;
138  else if(MS2Policy == Cache::local)
139  return MS1Policy; // fill, pbfill
140  }
141  dawn_unreachable("invalid policy combination");
142 }

148 bool PassSetCaches::run(StencilInstantiation* instantiation) {
149  OptimizerContext* context = instantiation->getOptimizerContext();
150 
151  for(const auto& stencilPtr : instantiation->getStencils()) {
152  const Stencil& stencil = *stencilPtr;
153 
154  // Set IJ-Caches
155  int msIdx = 0;
156  for(const auto& multiStagePtr : stencil.getMultiStages()) {
157  MultiStage& MS = *multiStagePtr;
158 
159  std::set<int> outputFields;
160 
161  auto isOutput = [](const Field& field) {
162  return field.Intend == Field::IK_Output || field.Intend == Field::IK_InputOutput;
163  };
164 
165  for(const auto& stagePtr : multiStagePtr->getStages()) {
166  for(const Field& field : stagePtr->getFields()) {
167 
168  // Field is already cached, skip
169  if(MS.isCached(field.AccessID))
170  continue;
171 
172  // Field has vertical extents, can't be ij-cached
173  if(!field.Extent.isVerticalPointwise())
174  continue;
175 
176  // Currently we only cache temporaries!
177  if(!instantiation->isTemporaryField(field.AccessID))
178  continue;
179 
180  // Cache the field
181  if(field.Intend == Field::IK_Input && outputFields.count(field.AccessID) &&
182  !field.Extent.isHorizontalPointwise()) {
183 
184  Cache& cache = multiStagePtr->setCache(Cache::IJ, Cache::local, field.AccessID);
185 
186  if(context->getOptions().ReportPassSetCaches) {
187  std::cout << "\nPASS: " << getName() << ": " << instantiation->getName() << ": MS"
188  << msIdx << ": "
189  << instantiation->getOriginalNameFromAccessID(field.AccessID) << ":"
190  << cache.getCacheTypeAsString() << ":" << cache.getCacheIOPolicyAsString()
191  << std::endl;
192  }
193  }
194 
195  if(isOutput(field))
196  outputFields.insert(field.AccessID);
197  }
198  }
199  msIdx++;
200  }
201 
202  // Set K-Caches
203  if(context->getOptions().UseKCaches ||
204  stencil.getSIRStencil()->Attributes.has(sir::Attr::AK_UseKCaches)) {
205 
206  // Get the fields of all Multi-Stages
207  std::vector<std::unordered_map<int, Field>> fields;
208  std::transform(stencil.getMultiStages().begin(), stencil.getMultiStages().end(),
209  std::back_inserter(fields),
210  [](const std::shared_ptr<MultiStage>& MSPtr) { return MSPtr->getFields(); });
211 
212  int numMS = fields.size();
213  for(int MSIndex = 0; MSIndex < numMS; ++MSIndex) {
214  for(const auto& AccessIDFieldPair : fields[MSIndex]) {
215  MultiStage& MS = *stencil.getMultiStageFromMultiStageIndex(MSIndex);
216  const Field& field = AccessIDFieldPair.second;
217 
218  // Field is already cached, skip
219  if(MS.isCached(field.AccessID))
220  continue;
221 
222  // Field has horizontal extents, can't be k-cached
223  if(!field.Extent.isHorizontalPointwise())
224  continue;
225 
226  // Currently we only cache temporaries!
227  // if(!instantiation->isTemporaryField(field.AccessID))
228  // continue;
229 
230  if(!instantiation->isTemporaryField(field.AccessID) &&
231  (field.Intend == Field::IK_Output ||
232  (field.Intend == Field::IK_Input && field.Extent.isPointwise())))
233  continue;
234 
235  // Determine if we need to fill the cache by analyzing the current multi-stage
236  Cache::CacheIOPolicy policy = Cache::unknown;
237  if(field.Intend == Field::IK_Input) {
238  policy = Cache::fill;
239  } else if(field.Intend == Field::IK_Output)
240  policy = Cache::local;
241  else if(field.Intend == Field::IK_InputOutput) {
242  // Do we compute the first levels or do we need to access main memory (i.e fill the
243  // fist accesses)?
244  FirstAccessKind firstAccess = getFirstAccessKind(MS, field.AccessID);
245  if(firstAccess == FK_WriteOnly)
246  policy = Cache::local;
247  else {
248  // Do we have a read in counter loop order or pointwise access?
249  if(field.Extent.getVerticalLoopOrderAccesses(MS.getLoopOrder()).CounterLoopOrder ||
250  field.Extent.isVerticalPointwise())
251  policy = Cache::fill;
252  else
253  policy = Cache::bpfill;
254  }
255  }
256 
257  // Determine if we need to flush the cache by analyzing the next multi-stage
258  if(MSIndex != (numMS - 1) && fields[MSIndex + 1].count(field.AccessID)) {
259 
260  const MultiStage& nextMS = *stencil.getMultiStageFromMultiStageIndex(MSIndex + 1);
261  const Field& fieldInNextMS = fields[MSIndex + 1].find(field.AccessID)->second;
262 
263  if(fieldInNextMS.Intend == Field::IK_Input)
264  policy = combinePolicy(policy, Cache::flush);
265  else if(fieldInNextMS.Intend == Field::IK_InputOutput) {
266  // Do we compute the first levels or do we need to access main memory (i.e flush
267  // the last accesses)?
268  FirstAccessKind firstAccess = getFirstAccessKind(nextMS, fieldInNextMS.AccessID);
269  if(firstAccess == FK_WriteOnly)
270  policy = combinePolicy(policy, Cache::local);
271  else {
272  if(fieldInNextMS.Extent.getVerticalLoopOrderAccesses(nextMS.getLoopOrder())
273  .CounterLoopOrder ||
274  fieldInNextMS.Extent.isVerticalPointwise())
275  policy = combinePolicy(policy, Cache::flush);
276  else
277  policy = combinePolicy(policy, Cache::epflush);
278  }
279  }
280  } else {
281  if(!instantiation->isTemporaryField(field.AccessID) && field.Intend != Field::IK_Input)
282  policy = combinePolicy(policy, Cache::flush);
283  }
284 
285  // Set the cache
286  Cache& cache = MS.setCache(Cache::K, policy, field.AccessID);
287 
288  if(context->getOptions().ReportPassSetCaches) {
289  std::cout << "\nPASS: " << getName() << ": " << instantiation->getName() << ": MS"
290  << MSIndex << ": "
291  << instantiation->getOriginalNameFromAccessID(field.AccessID) << ":"
292  << cache.getCacheTypeAsString() << ":" << cache.getCacheIOPolicyAsString()
293  << std::endl;
294  }
295  }
296  }
297  }
298  }
299 
300  return true;
301 }
