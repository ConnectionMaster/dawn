43  for(size_t I = 0; I < Length; ++I) {
44  unsigned char LHC = ascii_tolower(LHS[I]);
45  unsigned char RHC = ascii_tolower(RHS[I]);

74 int StringRef::compare_numeric(StringRef RHS) const {
75  for(size_t I = 0, E = std::min(Length, RHS.Length); I != E; ++I) {
76  // Check for sequences of digits.
77  if(ascii_isdigit(Data[I]) && ascii_isdigit(RHS.Data[I])) {
78  // The longer sequence of numbers is considered larger.
79  // This doesn't really handle prefixed zeros well.
80  size_t J;
81  for(J = I + 1; J != E + 1; ++J) {
82  bool ld = J < Length && ascii_isdigit(Data[J]);
83  bool rd = J < RHS.Length && ascii_isdigit(RHS.Data[J]);
84  if(ld != rd)
85  return rd ? -1 : 1;
86  if(!rd)
87  break;
88  }
89  // The two number sequences have the same length (J-I), just memcmp them.
90  if(int Res = compareMemory(Data + I, RHS.Data + I, J - I))
91  return Res < 0 ? -1 : 1;
92  // Identical number sequences, continue search after the numbers.
93  I = J - 1;
94  continue;
95  }
96  if(Data[I] != RHS.Data[I])
97  return (unsigned char)Data[I] < (unsigned char)RHS.Data[I] ? -1 : 1;
98  }
99  if(Length == RHS.Length)
100  return 0;
101  return Length < RHS.Length ? -1 : 1;
102 }

135 size_t StringRef::find(StringRef Str, size_t From) const {
136  if(From > Length)
137  return npos;
138 
139  const char* Start = Data + From;
140  size_t Size = Length - From;
141 
142  const char* Needle = Str.data();
143  size_t N = Str.size();
144  if(N == 0)
145  return From;
146  if(Size < N)
147  return npos;
148  if(N == 1) {
149  const char* Ptr = (const char*)::memchr(Start, Needle[0], Size);
150  return Ptr == nullptr ? npos : Ptr - Data;
151  }
152 
153  const char* Stop = Start + (Size - N + 1);
154 
155  // For short haystacks or unsupported needles fall back to the naive algorithm
156  if(Size < 16 || N > 255) {
157  do {
158  if(std::memcmp(Start, Needle, N) == 0)
159  return Start - Data;
160  ++Start;
161  } while(Start < Stop);
162  return npos;
163  }
164 
165  // Build the bad char heuristic table, with uint8_t to reduce cache thrashing.
166  uint8_t BadCharSkip[256];
167  std::memset(BadCharSkip, N, 256);
168  for(unsigned i = 0; i != N - 1; ++i)
169  BadCharSkip[(uint8_t)Str[i]] = N - 1 - i;
170 
171  do {
172  uint8_t Last = Start[N - 1];
173  if(DAWN_BUILTIN_UNLIKELY(Last == (uint8_t)Needle[N - 1]))
174  if(std::memcmp(Start, Needle, N - 1) == 0)
175  return Start - Data;
176 
177  // Otherwise skip the appropriate number of bytes.
178  Start += BadCharSkip[Last];
179  } while(Start < Stop);
180 
181  return npos;
182 }

290 void StringRef::split(SmallVectorImpl<StringRef>& A, StringRef Separator, int MaxSplit,
291  bool KeepEmpty) const {
292  StringRef S = *this;
293 
294  // Count down from MaxSplit. When MaxSplit is -1, this will just split
295  // "forever". This doesn't support splitting more than 2^31 times
296  // intentionally; if we ever want that we can make MaxSplit a 64-bit integer
297  // but that seems unlikely to be useful.
298  while(MaxSplit-- != 0) {
299  size_t Idx = S.find(Separator);
300  if(Idx == npos)
301  break;
302 
303  // Push this split.
304  if(KeepEmpty || Idx > 0)
305  A.push_back(S.slice(0, Idx));
306 
307  // Jump forward.
308  S = S.slice(Idx + Separator.size(), npos);
309  }
310 
311  // Push the tail.
312  if(KeepEmpty || !S.empty())
313  A.push_back(S);
314 }

357 static unsigned GetAutoSenseRadix(StringRef& Str) {
358  if(Str.empty())
359  return 10;
360 
361  if(Str.startswith("0x") || Str.startswith("0X")) {
362  Str = Str.substr(2);

384 bool consumeUnsignedInteger(StringRef& Str, unsigned Radix, unsigned long long& Result) {
385  // Autosense radix if not specified.
386  if(Radix == 0)
387  Radix = GetAutoSenseRadix(Str);
388 
389  // Empty strings (after the radix autosense) are invalid.
390  if(Str.empty())
391  return true;
392 
393  // Parse all the bytes of the string given this radix. Watch for overflow.
394  StringRef Str2 = Str;
395  Result = 0;
396  while(!Str2.empty()) {
397  unsigned CharVal;
398  if(Str2[0] >= '0' && Str2[0] <= '9')
399  CharVal = Str2[0] - '0';
400  else if(Str2[0] >= 'a' && Str2[0] <= 'z')
401  CharVal = Str2[0] - 'a' + 10;
402  else if(Str2[0] >= 'A' && Str2[0] <= 'Z')
403  CharVal = Str2[0] - 'A' + 10;
404  else
405  break;
406 
407  // If the parsed value is larger than the integer radix, we cannot
408  // consume any more characters.
409  if(CharVal >= Radix)
410  break;
411 
412  // Add in this character.
413  unsigned long long PrevResult = Result;
414  Result = Result * Radix + CharVal;
415 
416  // Check for overflow by shifting back and seeing if bits were lost.
417  if(Result / Radix < PrevResult)
418  return true;
419 
420  Str2 = Str2.substr(1);
421  }
422 
423  // We consider the operation a failure if no characters were consumed
424  // successfully.
425  if(Str.size() == Str2.size())
426  return true;
427 
428  Str = Str2;
429  return false;
430 }

432 bool consumeSignedInteger(StringRef& Str, unsigned Radix, long long& Result) {
433  unsigned long long ULLVal;
434 
435  // Handle positive strings first.
436  if(Str.empty() || Str.front() != '-') {
437  if(consumeUnsignedInteger(Str, Radix, ULLVal) ||
438  // Check for value so large it overflows a signed value.
439  (long long)ULLVal < 0)
440  return true;
441  Result = ULLVal;
442  return false;
443  }
444 
445  // Get the positive part of the value.
446  StringRef Str2 = Str.drop_front(1);
447  if(consumeUnsignedInteger(Str2, Radix, ULLVal) ||
448  // Reject values so large they'd overflow as negative signed, but allow
449  // "-0". This negates the unsigned so that the negative isn't undefined
450  // on signed overflow.
451  (long long)-ULLVal > 0)
452  return true;
453 
454  Str = Str2;
455  Result = -ULLVal;
456  return false;
457 }
