99  void insertEdge(int IDFrom, int IDTo, EdgeData data = EdgeData()) {
100 
101  // Create `IDTo` node (We shift the burden to the `insertNode` to take appropriate actions
102  // if the node does already exist)
103  static_cast<Derived*>(this)->insertNode(IDTo);
104 
105  // Traverse the edge-list of node `IDFrom` to check if we already have such an edge
106  auto& edgeList = adjacencyList_[getVertexIDFromID(IDFrom)];
107  const Edge edge{data, getVertexIDFromID(IDFrom), getVertexIDFromID(IDTo)};
108 
109  auto it = std::find_if(edgeList->begin(), edgeList->end(), [&edge](const Edge& e) {
110  return e.FromVertexID == edge.FromVertexID && e.ToVertexID == edge.ToVertexID;
111  });
112 
113  if(it != edgeList->end())
114  static_cast<Derived*>(this)->edgeAlreadyExists(it->Data, edge.Data);
115  else
116  edgeList->push_back(edge);
117  }

193  void toDotImpl(StreamType& os) const {
194  std::unordered_set<std::string> nodeStrs;
195  std::unordered_set<std::string> edgeStrs;
196 
197  for(std::size_t VertexID = 0; VertexID < adjacencyList_.size(); ++VertexID) {
198  const std::string FromVertexName =
199  static_cast<const Derived*>(this)->getVertexNameByVertexID(VertexID);
200 
201  // Convert node to dot
202  nodeStrs.emplace(std::string("node [shape = ") +
203  static_cast<const Derived*>(this)->getDotShape() + "] \"" + FromVertexName +
204  "\"");
205 
206  // Convert edge to dot
207  for(const Edge& edge : *(adjacencyList_[VertexID]))
208  edgeStrs.emplace(
209  "\"" + FromVertexName + "\" -> \"" +
210  static_cast<const Derived*>(this)->getVertexNameByVertexID(edge.ToVertexID) + "\"" +
211  static_cast<const Derived*>(this)->edgeDataToDot(edge.Data));
212  }
213 
214  os << "digraph G {\n"
215  << "rankdir=LR;\n"
216  << "size=\"8.5\" \n";
217  std::copy(nodeStrs.begin(), nodeStrs.end(), std::ostream_iterator<std::string>(os, ";\n"));
218  std::copy(edgeStrs.begin(), edgeStrs.end(), std::ostream_iterator<std::string>(os, ";\n"));
219  os << "}\n";
220  }
