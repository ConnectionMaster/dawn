91 bool PassTemporaryType::run(StencilInstantiation* instantiation) {
92  OptimizerContext* context = instantiation->getOptimizerContext();
93 
94  std::unordered_map<int, Temporary> temporaries;
95  std::unordered_set<int> AccessIDs;
96 
97  // Fix temporaries which span over multiple stencils and promote them to 3D allocated fields
98 
99  // Fix the temporaries within the same stencil
100  for(const auto& stencilPtr : instantiation->getStencils()) {
101  temporaries.clear();
102  AccessIDs.clear();
103 
104  // Loop over all accesses
105  for(auto& multiStagePtr : stencilPtr->getMultiStages()) {
106  for(auto& stagePtr : multiStagePtr->getStages()) {
107  for(auto& doMethodPtr : stagePtr->getDoMethods()) {
108  for(const auto& statementAccessesPair : doMethodPtr->getStatementAccessesPairs()) {
109 
110  auto processAccessMap = [&](const std::unordered_map<int, Extents>& accessMap) {
111  for(const auto& AccessIDExtentPair : accessMap) {
112  int AccessID = AccessIDExtentPair.first;
113  const Extents& extent = AccessIDExtentPair.second;
114 
115  // Is it a temporary?
116  bool isTemporaryField = instantiation->isTemporaryField(AccessID);
117  if(isTemporaryField || (!instantiation->isGlobalVariable(AccessID) &&
118  instantiation->isVariable(AccessID))) {
119 
120  auto it = temporaries.find(AccessID);
121  if(it != temporaries.end()) {
122  // If we already registered it, update the extent
123  it->second.Extent.merge(extent);
124  } else {
125  // Register the temporary
126  AccessIDs.insert(AccessID);
127  temporaries.emplace(AccessID,
128  Temporary(AccessID,
129  isTemporaryField ? Temporary::TT_Field
130  : Temporary::TT_LocalVariable,
131  extent));
132  }
133  }
134  }
135  };
136 
137  processAccessMap(statementAccessesPair->getAccesses()->getWriteAccesses());
138  processAccessMap(statementAccessesPair->getAccesses()->getReadAccesses());
139  }
140  }
141  }
142  }
143 
144  auto LifetimeMap = stencilPtr->getLifetime(AccessIDs);
145  std::for_each(LifetimeMap.begin(), LifetimeMap.end(),
146  [&](const std::pair<int, Stencil::Lifetime>& lifetimePair) {
147  temporaries[lifetimePair.first].Lifetime = lifetimePair.second;
148  });
149 
150  // Process each temporary
151  for(const auto& AccessIDTemporaryPair : temporaries) {
152  int AccessID = AccessIDTemporaryPair.first;
153  const Temporary& temporary = AccessIDTemporaryPair.second;
154 
155  auto report = [&](const char* action) {
156  std::cout << "\nPASS: " << getName() << ": " << instantiation->getName() << ": " << action
157  << ":" << instantiation->getOriginalNameFromAccessID(AccessID) << std::endl;
158  };
159 
160  if(temporary.Type == Temporary::TT_LocalVariable) {
161  // If the variable is accessed in multiple Do-Methods, we need to promote it to a field!
162  if(!temporary.Lifetime.Begin.inSameDoMethod(temporary.Lifetime.End)) {
163 
164  if(context->getOptions().ReportPassTemporaryType)
165  report("promote");
166 
167  instantiation->promoteLocalVariableToTemporaryField(stencilPtr.get(), AccessID,
168  temporary.Lifetime);
169  }
170  } else {
171  // If the field is only accessed within the same Do-Method, does not have an extent and is
172  // not argument to a stencil function, we can demote it to a local variable
173  if(temporary.Lifetime.Begin.inSameDoMethod(temporary.Lifetime.End) &&
174  temporary.Extent.isPointwise() &&
175  !usedAsArgumentInStencilFun(stencilPtr.get(), AccessID)) {
176 
177  if(context->getOptions().ReportPassTemporaryType)
178  report("demote");
179 
180  instantiation->demoteTemporaryFieldToLocalVariable(stencilPtr.get(), AccessID,
181  temporary.Lifetime);
182  }
183  }
184  }
185  }
186 
187  return true;
188 }

190 void PassTemporaryType::fixTemporariesSpanningMultipleStencils(
191  StencilInstantiation* instantiation, const std::vector<std::shared_ptr<Stencil>>& stencils) {
192  if(stencils.size() <= 1)
193  return;
194 
195  for(int i = 0; i < stencils.size(); ++i) {
196  for(const Stencil::FieldInfo& fieldi : stencils[i]->getFields()) {
197 
198  // Is fieldi a temporary?
199  if(fieldi.IsTemporary) {
200 
201  // Is it referenced in another stencil?
202  for(int j = i + 1; j < stencils.size(); ++j) {
203  for(const Stencil::FieldInfo& fieldj : stencils[j]->getFields()) {
204 
205  // Yes and yes ... promote it to a real storage
206  if(fieldi.AccessID == fieldj.AccessID && fieldj.IsTemporary) {
207  instantiation->promoteTemporaryFieldToAllocatedField(fieldi.AccessID);
208  }
209  }
210  }
211  }
212  }
213  }
214 }
