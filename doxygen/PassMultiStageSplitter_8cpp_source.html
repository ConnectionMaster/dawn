32 bool PassMultiStageSplitter::run(StencilInstantiation* stencilInstantiation) {
33  OptimizerContext* context = stencilInstantiation->getOptimizerContext();
34 
35  int numSplit = 0;
36  std::deque<MultiStage::SplitIndex> splitterIndices;
37  DependencyGraphAccesses graph(stencilInstantiation);
38 
39  // Iterate over all stages in all multistages of all stencils
40  for(auto& stencil : stencilInstantiation->getStencils()) {
41 
42  int multiStageIndex = 0;
43  for(auto multiStageIt = stencil->getMultiStages().begin();
44  multiStageIt != stencil->getMultiStages().end(); ++multiStageIndex) {
45  MultiStage& multiStage = (**multiStageIt);
46 
47  splitterIndices.clear();
48  graph.clear();
49 
50  // Loop order specified by the user in the vertical region (e.g {k_start, k_end} is forward)
51  auto userSpecifiedLoopOrder = multiStage.getLoopOrder();
52 
53  // If not proven otherwise, we assume a parralel loop order
54  auto curLoopOrder = LoopOrderKind::LK_Parallel;
55 
56  // Build the Dependency graph (bottom --> top i.e iterate the stages backwards)
57  int stageIndex = multiStage.getStages().size() - 1;
58  for(auto stageIt = multiStage.getStages().rbegin(); stageIt != multiStage.getStages().rend();
59  ++stageIt, --stageIndex) {
60  Stage& stage = (**stageIt);
61  DoMethod& doMethod = stage.getSingleDoMethod();
62 
63  // Iterate statements backwards
64  for(int stmtIndex = doMethod.getStatementAccessesPairs().size() - 1; stmtIndex >= 0;
65  --stmtIndex) {
66  auto& stmtAccessesPair = doMethod.getStatementAccessesPairs()[stmtIndex];
67  graph.insertStatementAccessesPair(stmtAccessesPair);
68 
69  // Check for read-before-write conflicts in the loop order and counter loop order.
70  // Conflicts in the loop order will assure us that the multi-stage can't be
71  // parallel. A conflict in the counter loop order is more severe and needs the current
72  // multistage be splitted!
73  auto conflict = hasVerticalReadBeforeWriteConflict(&graph, userSpecifiedLoopOrder);
74 
75  if(conflict.CounterLoopOrderConflict) {
76 
77  // The loop order of the lower part is what we recoreded in the last steps
78  splitterIndices.emplace_front(
79  MultiStage::SplitIndex{stageIndex, stmtIndex, curLoopOrder});
80 
81  if(context->getOptions().ReportPassMultiStageSplit)
82  std::cout << "\nPASS: " << getName() << ": " << stencilInstantiation->getName()
83  << ": split:"
84  << doMethod.getStatementAccessesPairs()[stmtIndex]
85  ->getStatement()
86  ->ASTStmt->getSourceLocation()
87  .Line
88  << " looporder:" << curLoopOrder << "\n";
89 
90  if(context->getOptions().DumpSplitGraphs)
91  graph.toDot(format("stmt_vd_ms%i_%02i.dot", multiStageIndex, numSplit));
92 
93  // Clear the graph ...
94  graph.clear();
95  curLoopOrder = LoopOrderKind::LK_Parallel;
96 
97  // ... and process the current statement again
98  graph.insertStatementAccessesPair(stmtAccessesPair);
99 
100  numSplit++;
101 
102  } else if(conflict.LoopOrderConflict)
103  // We have a conflict in the loop order, the multi-stage cannot be executed in parallel
104  // and we use the loop order specified by the user
105  curLoopOrder = userSpecifiedLoopOrder;
106  }
107  }
108 
109  if(context->getOptions().DumpSplitGraphs)
110  graph.toDot(format("stmt_vd_m%i_%02i.dot", multiStageIndex, numSplit));
111 
112  if(!splitterIndices.empty()) {
113  auto newMultiStages = multiStage.split(splitterIndices, curLoopOrder);
114  multiStageIt = stencil->getMultiStages().erase(multiStageIt);
115  stencil->getMultiStages().insert(multiStageIt,
116  std::make_move_iterator(newMultiStages.begin()),
117  std::make_move_iterator(newMultiStages.end()));
118  } else {
119  ++multiStageIt;
120 
121  // Not split needed, however we still may have detected forward or backward loop order
122  multiStage.setLoopOrder(curLoopOrder);
123  }
124  }
125  }
126 
127  if(context->getOptions().ReportPassMultiStageSplit && !numSplit)
128  std::cout << "\nPASS: " << getName() << ": " << stencilInstantiation->getName()
129  << ": no split\n";
130 
131  return true;
132 }
