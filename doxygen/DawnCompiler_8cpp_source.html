55 template <>
56 struct ComputeEditDistance<std::string> {
57  static std::string getSuggestion(const std::string& value,
58  const std::vector<std::string>& possibleValues) {
59  if(possibleValues.empty())
60  return "";
61 
62  std::vector<unsigned> editDistances;
63  std::transform(possibleValues.begin(), possibleValues.end(), std::back_inserter(editDistances),
64  [&](const std::string& val) { return computeEditDistance(value, val); });
65 
66  // Find minimum edit distance
67  unsigned minEditDistance = editDistances[0], minEditDistanceIdx = 0;
68  for(unsigned i = 1; i < editDistances.size(); ++i)
69  if(editDistances[i] < minEditDistance) {
70  minEditDistance = editDistances[i];
71  minEditDistanceIdx = i;
72  }
73 
74  return minEditDistance <= 2 ? "did you mean '" + possibleValues[minEditDistanceIdx] + "' ?"
75  : "";
76  }
77 };

80 template <class T>
81 DiagnosticsBuilder buildDiag(const std::string& option, const T& value, std::string reason,
82  std::vector<T> possibleValues = std::vector<T>{}) {
83  DiagnosticsBuilder diag(DiagnosticsKind::Error, SourceLocation());
84  diag << "invalid value '" << value << "' of option '" << option << "'";
85 
86  if(!reason.empty()) {
87  diag << ", " << reason;
88  } else {
89  auto suggestion = ComputeEditDistance<T>::getSuggestion(value, possibleValues);
90 
91  if(!suggestion.empty())
92  diag << ", " << suggestion;
93  else if(!possibleValues.empty())
94  diag << ", possible values " << RangeToString()(possibleValues);
95  }
96  return diag;
97 }

105 std::unique_ptr<TranslationUnit> DawnCompiler::compile(const SIR* SIR, CodeGenKind codeGen) {
106  diagnostics_->clear();
107  diagnostics_->setFilename(SIR->Filename);
108 
109  // Check if options are valid
110 
111  // -max-halo
112  if(options_->MaxHaloPoints < 0) {
113  diagnostics_->report(buildDiag("-max-halo", options_->MaxHaloPoints,
114  "maximum number of allowed halo points must be >= 0"));
115  return nullptr;
116  }
117 
118  // -inline
119  using InlineStrategyKind = PassInlining::InlineStrategyKind;
120  InlineStrategyKind inlineStrategy = StringSwitch<InlineStrategyKind>(options_->InlineStrategy)
121  .Case("none", InlineStrategyKind::IK_None)
122  .Case("cof", InlineStrategyKind::IK_ComputationOnTheFly)
123  .Case("pc", InlineStrategyKind::IK_Precomputation)
124  .Default(InlineStrategyKind::IK_Unknown);
125 
126  if(inlineStrategy == InlineStrategyKind::IK_Unknown) {
127  diagnostics_->report(buildDiag("-inline", options_->InlineStrategy, "", {"none", "cof", "pc"}));
128  return nullptr;
129  }
130 
131  // -reorder
132  using ReorderStrategyKind = ReorderStrategy::ReorderStrategyKind;
133  ReorderStrategyKind reorderStrategy = StringSwitch<ReorderStrategyKind>(options_->ReorderStrategy)
134  .Case("none", ReorderStrategyKind::RK_None)
135  .Case("greedy", ReorderStrategyKind::RK_Greedy)
136  .Case("scut", ReorderStrategyKind::RK_Partitioning)
137  .Default(ReorderStrategyKind::RK_Unknown);
138 
139  if(reorderStrategy == ReorderStrategyKind::RK_Unknown) {
140  diagnostics_->report(
141  buildDiag("-reorder", options_->ReorderStrategy, "", {"none", "greedy", "scut"}));
142  return nullptr;
143  }
144 
145  // Initialize optimizer
146  auto optimizer = make_unique<OptimizerContext>(this, SIR);
147  PassManager& passManager = optimizer->getPassManager();
148 
149  // Setup pass interface
150  passManager.pushBackPass<PassInlining>(inlineStrategy);
151  passManager.pushBackPass<PassTemporaryFirstAccess>();
152  passManager.pushBackPass<PassFieldVersioning>();
153  passManager.pushBackPass<PassSSA>();
154  passManager.pushBackPass<PassMultiStageSplitter>();
155  passManager.pushBackPass<PassStageSplitter>();
156  passManager.pushBackPass<PassPrintStencilGraph>();
157  passManager.pushBackPass<PassTemporaryType>();
158  passManager.pushBackPass<PassSetStageName>();
159  passManager.pushBackPass<PassSetStageGraph>();
160  passManager.pushBackPass<PassStageReordering>(reorderStrategy);
161  passManager.pushBackPass<PassStageMerger>();
162  passManager.pushBackPass<PassStencilSplitter>();
163  passManager.pushBackPass<PassTemporaryType>();
164  passManager.pushBackPass<PassTemporaryMerger>();
165  passManager.pushBackPass<PassSetCaches>();
166  passManager.pushBackPass<PassDataLocalityMetric>();
167 
168  // Run optimization passes
169  for(auto& stencil : optimizer->getStencilInstantiationMap()) {
170  StencilInstantiation* instantiation = stencil.second.get();
171  DAWN_LOG(INFO) << "Starting Optimization and Analysis passes for `" << instantiation->getName()
172  << "` ...";
173 
174  if(!passManager.runAllPassesOnStecilInstantiation(instantiation))
175  return nullptr;
176 
177  DAWN_LOG(INFO) << "Done with Optimization and Analysis passes for `" << instantiation->getName()
178  << "`";
179  }
180 
181  if(diagnostics_->hasErrors()) {
182  DAWN_LOG(INFO) << "Errors occured. Skipping code generation.";
183  return nullptr;
184  }
185 
186  // Generate code
187  std::unique_ptr<CodeGen> CG;
188  switch(codeGen) {
189  case CodeGenKind::CG_GTClang:
190  CG = make_unique<GTClangCodeGen>(optimizer.get());
191  break;
192  case CodeGenKind::CG_GTClangNaiveCXX:
193  CG = make_unique<GTClangNaiveCXXCodeGen>(optimizer.get());
194  break;
195  }
196  return CG->generateCode();
197 }
