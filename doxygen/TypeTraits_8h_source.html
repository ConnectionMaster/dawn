27 template <typename T>

29 
30 // std::is_trivially_copyable is available in libc++ with clang, libstdc++
31 // that comes with GCC 5.
32 #if(__has_feature(is_trivially_copyable) && defined(_LIBCPP_VERSION)) || \
33  (defined(__GNUC__) && __GNUC__ >= 5)
34  // If the compiler supports the is_trivially_copyable trait use it, as it
35  // matches the definition of isPodLike closely.
36  static const bool value = std::is_trivially_copyable<T>::value;
37 #elif __has_feature(is_trivially_copyable)
38  // Use the internal name if the compiler supports is_trivially_copyable but we
39  // don't know if the standard library does. This is the case for clang in
40  // conjunction with libstdc++ from GCC 4.x.
41  static const bool value = __is_trivially_copyable(T);
42 #else
43  // If we don't know anything else, we can (at least) assume that all non-class
44  // types are PODs.
45  static const bool value = !std::is_class<T>::value;
46 #endif
47 };
