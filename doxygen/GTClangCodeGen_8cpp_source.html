36 GTClangCodeGen::IntervalDefinitions::IntervalDefinitions(const Stencil& stencil)
37  : Intervals(stencil.getIntervals()), Axis(*Intervals.begin()) {
38  DAWN_ASSERT(!Intervals.empty());
39 
40  // Add intervals for the stencil functions
41  for(const auto& stencilFun :
42  stencil.getStencilInstantiation()->getStencilFunctionInstantiations())
43  Intervals.insert(stencilFun->getInterval());
44 
45  // Compute axis and populate the levels
46  Axis = *Intervals.begin();
47  for(const Interval& interval : Intervals) {
48  Levels.insert(interval.lowerLevel());
49  Levels.insert(interval.upperLevel());
50  Axis.merge(interval);
51  }
52 
53  // Generate the name of the enclosing intervals of each multi-stage (required by the K-Caches)
54  for(auto multiStagePtr : stencil.getMultiStages()) {
55  Interval interval = multiStagePtr->getEnclosingInterval();
56  if(!IntervalToNameMap.count(interval))
57  IntervalToNameMap.emplace(interval, Interval::makeCodeGenName(interval));
58  }
59 
60  // Compute the intervals required by each stage. Note that each stage needs to have Do-Methods
61  // for the entire axis, this means we may need to add empty Do-Methods
62  // See https://github.com/eth-cscs/gridtools/issues/330
63  int numStages = stencil.getNumStages();
64  for(int i = 0; i < numStages; ++i) {
65  const std::shared_ptr<Stage>& stagePtr = stencil.getStage(i);
66 
67  auto iteratorSuccessPair = StageIntervals.emplace(
68  stagePtr, Interval::computeGapIntervals(Axis, stagePtr->getIntervals()));
69  DAWN_ASSERT(iteratorSuccessPair.second);
70  std::vector<Interval>& DoMethodIntervals = iteratorSuccessPair.first->second;
71 
72  // Generate unique names for the intervals
73  for(const Interval& interval : DoMethodIntervals)
74  IntervalToNameMap.emplace(interval, Interval::makeCodeGenName(interval));
75  }
76 
77  // Make sure the intervals for the stencil functions exist
78  for(const auto& stencilFun :
79  stencil.getStencilInstantiation()->getStencilFunctionInstantiations())
80  IntervalToNameMap.emplace(stencilFun->getInterval(),
81  Interval::makeCodeGenName(stencilFun->getInterval()));
82 }

84 std::string
85 GTClangCodeGen::generateStencilInstantiation(const StencilInstantiation* stencilInstantiation) {
86  using namespace codegen;
87 
88  std::stringstream ssSW, ssMS, tss;
89  auto gt = []() { return Twine("gridtools::"); };
90  auto gtc = []() { return Twine("gridtools::clang::"); };
91  auto gt_enum = []() { return Twine("gridtools::enumtype::"); };
92 
93  // K-Cache branch changes the signature of Do-Methods
94  const char* DoMethodArg = "Evaluation& eval";
95 
96  Class StencilWrapperClass(stencilInstantiation->getName(), ssSW);
97  StencilWrapperClass.changeAccessibility(
98  "public"); // The stencils should technically be private but nvcc doesn't like it ...
99 
100  bool isEmpty = true;
101 
102  // Generate stencils
103  auto& stencils = stencilInstantiation->getStencils();
104  for(std::size_t stencilIdx = 0; stencilIdx < stencils.size(); ++stencilIdx) {
105  const Stencil& stencil = *stencilInstantiation->getStencils()[stencilIdx];
106 
107  if(stencil.isEmpty())
108  continue;
109 
110  isEmpty = false;
111  Structure StencilClass = StencilWrapperClass.addStruct(Twine("stencil_") + Twine(stencilIdx));
112  std::string StencilName = StencilClass.getName();
113 
114  //
115  // Interval typedefs
116  //
117  StencilClass.addComment("Intervals");
118  IntervalDefinitions intervalDefinitions(stencil);
119 
120  std::size_t maxLevel = intervalDefinitions.Levels.size() - 1;
121 
122  auto makeLevelName = [&](int level, int offset) {
123  clear(tss);
124  tss << "gridtools::level<"
125  << (level == sir::Interval::End ? maxLevel
126  : std::distance(intervalDefinitions.Levels.begin(),
127  intervalDefinitions.Levels.find(level)))
128  << ", " << (offset <= 0 ? offset - 1 : offset) << ">";
129  return tss.str();
130  };
131 
132  // Generate typedefs for the individual intervals
133  for(const auto& intervalNamePair : intervalDefinitions.IntervalToNameMap)
134  StencilClass.addTypeDef(intervalNamePair.second)
135  .addType(gt() + "interval")
136  .addTemplates(makeArrayRef({makeLevelName(intervalNamePair.first.lowerLevel(),
137  intervalNamePair.first.lowerOffset()),
138  makeLevelName(intervalNamePair.first.upperLevel(),
139  intervalNamePair.first.upperOffset())}));
140 
141  ASTCodeGenGTClangStencilBody stencilBodyCGVisitor(stencilInstantiation,
142  intervalDefinitions.IntervalToNameMap);
143 
144  // Generate typedef for the axis
145  const Interval& axis = intervalDefinitions.Axis;
146  StencilClass.addTypeDef(Twine("axis_") + StencilName)
147  .addType(gt() + "interval")
148  .addTemplates(makeArrayRef(
149  {makeLevelName(axis.lowerLevel(), axis.lowerOffset() - 1),
150  makeLevelName(axis.upperLevel(),
151  (axis.upperOffset() + 1) == 0 ? 1 : (axis.upperOffset() + 1))}));
152 
153  // Generate typedef of the grid
154  StencilClass.addTypeDef(Twine("grid_") + StencilName)
155  .addType(gt() + "grid")
156  .addTemplate(Twine("axis_") + StencilName);
157 
158  // Generate code for members of the stencil
159  StencilClass.addComment("Members");
160  StencilClass.addMember("std::shared_ptr< gridtools::stencil >", "m_stencil");
161  StencilClass.addMember(Twine("std::unique_ptr< grid_") + StencilName + ">", "m_grid");
162 
163  //
164  // Generate stencil functions code for stencils instantiated by this stencil
165  //
166  std::unordered_set<std::string> generatedStencilFun;
167  for(const auto& stencilFun : stencilInstantiation->getStencilFunctionInstantiations()) {
168  std::string stencilFunName = StencilFunctionInstantiation::makeCodeGenName(*stencilFun);
169  if(generatedStencilFun.emplace(stencilFunName).second) {
170  Structure StencilFunStruct = StencilClass.addStruct(stencilFunName);
171 
172  // Field declaration
173  const auto& fields = stencilFun->getCalleeFields();
174  std::vector<std::string> arglist;
175 
176  if(fields.empty() && !stencilFun->hasReturn()) {
177  DiagnosticsBuilder diag(DiagnosticsKind::Error, stencilFun->getStencilFunction()->Loc);
178  diag << "no storages referenced in stencil function '" << stencilFun->getName()
179  << "', this would result in invalid gridtools code";
180  context_->getDiagnostics().report(diag);
181  return "";
182  }
183 
184  // If we have a return argument, we generate a special `__out` field
185  int accessorID = 0;
186  if(stencilFun->hasReturn()) {
187  StencilFunStruct.addStatement("using __out = gridtools::accessor<0, "
188  "gridtools::enumtype::inout, gridtools::extent<0, 0, 0, 0, "
189  "0, 0>>");
190  arglist.push_back("__out");
191  accessorID++;
192  }
193  // Generate field declarations
194  for(std::size_t m = 0; m < fields.size(); ++m, ++accessorID) {
195  std::string paramName = stencilFun->getOriginalNameFromCallerAccessID(fields[m].AccessID);
196 
197  // Generate parameter of stage
198  codegen::Type extent(gt() + "extent", clear(tss));
199  for(auto& e : fields[m].Extent.getExtents())
200  extent.addTemplate(Twine(e.Minus) + ", " + Twine(e.Plus));
201 
202  StencilFunStruct.addTypeDef(paramName)
203  .addType(gt() + "accessor")
204  .addTemplate(Twine(accessorID))
205  .addTemplate(gt_enum() + (fields[m].Intend == Field::IK_Input ? "in" : "inout"))
206  .addTemplate(extent);
207 
208  arglist.push_back(std::move(paramName));
209  }
210 
211  // Generate arglist
212  StencilFunStruct.addTypeDef("arg_list").addType("boost::mpl::vector").addTemplates(arglist);
213  mplContainerMaxSize_ = std::max(mplContainerMaxSize_, arglist.size());
214 
215  // Generate Do-Method
216  auto DoMethod = StencilFunStruct.addMemberFunction("GT_FUNCTION static void", "Do",
217  "typename Evaluation");
218  auto interveralIt = intervalDefinitions.IntervalToNameMap.find(stencilFun->getInterval());
219  DAWN_ASSERT_MSG(intervalDefinitions.IntervalToNameMap.end() != interveralIt,
220  "non-existing interval");
221  DoMethod.addArg(DoMethodArg);
222  DoMethod.addArg(interveralIt->second);
223  DoMethod.startBody();
224 
225  stencilBodyCGVisitor.setCurrentStencilFunction(stencilFun.get());
226  stencilBodyCGVisitor.setIndent(DoMethod.getIndent());
227  for(const auto& statementAccessesPair : stencilFun->getStatementAccessesPairs()) {
228  statementAccessesPair->getStatement()->ASTStmt->accept(stencilBodyCGVisitor);
229  DoMethod.indentStatment();
230  DoMethod << stencilBodyCGVisitor.getCodeAndResetStream();
231  }
232 
233  DoMethod.commit();
234  }
235  }
236 
237  // Done generating stencil functions ...
238  stencilBodyCGVisitor.setCurrentStencilFunction(nullptr);
239 
240  std::vector<std::string> makeComputation;
241 
242  //
243  // Generate code for stages and assemble the `make_computation`
244  //
245  std::size_t multiStageIdx = 0;
246  for(auto multiStageIt = stencil.getMultiStages().begin(),
247  multiStageEnd = stencil.getMultiStages().end();
248  multiStageIt != multiStageEnd; ++multiStageIt, ++multiStageIdx) {
249  const MultiStage& multiStage = **multiStageIt;
250 
251  // Generate `make_multistage`
252  ssMS << "gridtools::make_multistage(gridtools::enumtype::execute<gridtools::enumtype::";
253  if(!context_->getOptions().UseParallelEP &&
254  multiStage.getLoopOrder() == LoopOrderKind::LK_Parallel)
255  ssMS << LoopOrderKind::LK_Forward << " /*parallel*/ ";
256  else
257  ssMS << multiStage.getLoopOrder();
258  ssMS << ">(),";
259 
260  // Add the MultiStage caches
261  if(!multiStage.getCaches().empty()) {
262  ssMS << RangeToString(", ", "gridtools::define_caches(", "),")(
263  multiStage.getCaches(),
264  [&](const std::pair<int, Cache>& AccessIDCachePair) -> std::string {
265  return (gt() + "cache<" +
266  // Type: IJ or K
267  gt() + AccessIDCachePair.second.getCacheTypeAsString() + ", " +
268  // IOPolicy: local, fill, bpfill, flush, epflush or flush_and_fill
269  gt() + "cache_io_policy::" +
270  AccessIDCachePair.second.getCacheIOPolicyAsString() +
271  // Interval: if IOPolicy is not local, we need to provide the interval
272  (AccessIDCachePair.second.getCacheIOPolicy() != Cache::local
273  ? ", " +
274  intervalDefinitions
275  .IntervalToNameMap[multiStage.getEnclosingInterval()]
276  : std::string()) +
277  // Placeholder which will be cached
278  ">(p_" + stencilInstantiation->getNameFromAccessID(AccessIDCachePair.first) +
279  "())")
280  .str();
281  });
282  }
283 
284  std::size_t stageIdx = 0;
285  for(auto stageIt = multiStage.getStages().begin(), stageEnd = multiStage.getStages().end();
286  stageIt != stageEnd; ++stageIt, ++stageIdx) {
287  const auto& stagePtr = *stageIt;
288  const Stage& stage = *stagePtr;
289 
290  Structure StageStruct =
291  StencilClass.addStruct(Twine("stage_") + Twine(multiStageIdx) + "_" + Twine(stageIdx));
292 
293  ssMS << "gridtools::make_stage<" << StageStruct.getName() << ">(";
294 
295  // Field declaration
296  const auto& fields = stage.getFields();
297  std::vector<std::string> arglist;
298 
299  if(fields.empty()) {
300  DiagnosticsBuilder diag(DiagnosticsKind::Error,
301  stencilInstantiation->getSIRStencil()->Loc);
302  diag << "no storages referenced in stencil '" << stencilInstantiation->getName()
303  << "', this would result in invalid gridtools code";
304  context_->getDiagnostics().report(diag);
305  return "";
306  }
307 
308  std::size_t accessorIdx = 0;
309  for(; accessorIdx < fields.size(); ++accessorIdx) {
310  const auto& field = fields[accessorIdx];
311  std::string paramName = stencilInstantiation->getNameFromAccessID(field.AccessID);
312 
313  // Generate parameter of stage
314  codegen::Type extent(gt() + "extent", clear(tss));
315  for(auto& e : field.Extent.getExtents())
316  extent.addTemplate(Twine(e.Minus) + ", " + Twine(e.Plus));
317 
318  StageStruct.addTypeDef(paramName)
319  .addType(gt() + "accessor")
320  .addTemplate(Twine(accessorIdx))
321  .addTemplate(gt_enum() + (field.Intend == Field::IK_Input ? "in" : "inout"))
322  .addTemplate(extent);
323 
324  // Generate placeholder mapping of the field in `make_stage`
325  ssMS << "p_" << paramName << "()"
326  << ((stage.getGlobalVariables().empty() && (accessorIdx == fields.size() - 1))
327  ? ""
328  : ", ");
329 
330  arglist.push_back(std::move(paramName));
331  }
332 
333  // Global accessor declaration
334  std::size_t maxAccessors = fields.size() + stage.getGlobalVariables().size();
335  for(int AccessID : stage.getGlobalVariables()) {
336  std::string paramName = stencilInstantiation->getNameFromAccessID(AccessID);
337 
338  StageStruct.addTypeDef(paramName)
339  .addType(gt() + "global_accessor")
340  .addTemplate(Twine(accessorIdx))
341  .addTemplate(gt_enum() + "inout");
342  accessorIdx++;
343 
344  // Generate placeholder mapping of the field in `make_stage`
345  ssMS << "p_" << paramName << "()" << (accessorIdx == maxAccessors ? "" : ", ");
346 
347  arglist.push_back(std::move(paramName));
348  }
349 
350  ssMS << ")" << ((stageIdx != multiStage.getStages().size() - 1) ? "," : ")");
351 
352  // Generate arglist
353  StageStruct.addTypeDef("arg_list").addType("boost::mpl::vector").addTemplates(arglist);
354  mplContainerMaxSize_ = std::max(mplContainerMaxSize_, arglist.size());
355 
356  // Generate Do-Method
357  for(const auto& doMethodPtr : stage.getDoMethods()) {
358  const DoMethod& doMethod = *doMethodPtr;
359 
360  auto DoMethodCodeGen =
361  StageStruct.addMemberFunction("GT_FUNCTION static void", "Do", "typename Evaluation");
362  DoMethodCodeGen.addArg(DoMethodArg);
363  DoMethodCodeGen.addArg(intervalDefinitions.IntervalToNameMap[doMethod.getInterval()]);
364  DoMethodCodeGen.startBody();
365 
366  stencilBodyCGVisitor.setIndent(DoMethodCodeGen.getIndent());
367  for(const auto& statementAccessesPair : doMethod.getStatementAccessesPairs()) {
368  statementAccessesPair->getStatement()->ASTStmt->accept(stencilBodyCGVisitor);
369  DoMethodCodeGen << stencilBodyCGVisitor.getCodeAndResetStream();
370  }
371  }
372 
373  // Generate empty Do-Methods
374  // See https://github.com/eth-cscs/gridtools/issues/330
375  const auto& stageIntervals = stage.getIntervals();
376  for(const auto& interval : intervalDefinitions.StageIntervals[stagePtr])
377  if(std::find(stageIntervals.begin(), stageIntervals.end(), interval) ==
378  stageIntervals.end())
379  StageStruct.addMemberFunction("GT_FUNCTION static void", "Do", "typename Evaluation")
380  .addArg(DoMethodArg)
381  .addArg(intervalDefinitions.IntervalToNameMap[interval]);
382  }
383 
384  makeComputation.push_back(ssMS.str());
385  clear(ssMS);
386  }
387 
388  //
389  // Generate constructor/destructor and methods of the stencil
390  //
391  std::vector<Stencil::FieldInfo> StencilFields = stencil.getFields();
392  std::vector<std::string> StencilGlobalVariables = stencil.getGlobalVariables();
393  std::size_t numFields = StencilFields.size();
394 
395  mplContainerMaxSize_ = std::max(mplContainerMaxSize_, numFields);
396 
397  std::vector<std::string> StencilConstructorTemplates;
398  int numTemporaries = 0;
399  for(int i = 0; i < numFields; ++i)
400  if(StencilFields[i].IsTemporary)
401  numTemporaries += 1;
402  else
403  StencilConstructorTemplates.push_back("S" + std::to_string(i + 1 - numTemporaries));
404 
405  // Generate constructor
406  auto StencilConstructor = StencilClass.addConstructor(RangeToString(", ", "", "")(
407  StencilConstructorTemplates, [](const std::string& str) { return "class " + str; }));
408 
409  StencilConstructor.addArg("const gridtools::clang::domain& dom");
410  for(int i = 0; i < numFields; ++i)
411  if(!StencilFields[i].IsTemporary)
412  StencilConstructor.addArg(StencilConstructorTemplates[i - numTemporaries] + "& " +
413  StencilFields[i].Name);
414 
415  StencilConstructor.startBody();
416 
417  // Generate domain
418  StencilConstructor.addComment("Domain");
419  int accessorIdx = 0;
420 
421  for(; accessorIdx < numFields; ++accessorIdx)
422  // Fields
423  StencilConstructor.addTypeDef("p_" + StencilFields[accessorIdx].Name)
424  .addType(gt() + (StencilFields[accessorIdx].IsTemporary ? "tmp_arg" : "arg"))
425  .addTemplate(Twine(accessorIdx))
426  .addTemplate(StencilFields[accessorIdx].IsTemporary
427  ? "storage_t"
428  : StencilConstructorTemplates[accessorIdx - numTemporaries]);
429 
430  for(; accessorIdx < (numFields + StencilGlobalVariables.size()); ++accessorIdx) {
431  // Global variables
432  const auto& varname = StencilGlobalVariables[accessorIdx - numFields];
433  StencilConstructor.addTypeDef("p_" + StencilGlobalVariables[accessorIdx - numFields])
434  .addType(gt() + "arg")
435  .addTemplate(Twine(accessorIdx))
436  .addTemplate("typename std::decay<decltype(globals::get()." + varname +
437  ".as_global_parameter())>::type");
438  }
439 
440  std::vector<std::string> ArglistPlaceholders;
441  for(const auto& field : StencilFields)
442  ArglistPlaceholders.push_back("p_" + field.Name);
443  for(const auto& var : StencilGlobalVariables)
444  ArglistPlaceholders.push_back("p_" + var);
445 
446  StencilConstructor.addTypeDef("domain_arg_list")
447  .addType("boost::mpl::vector")
448  .addTemplates(ArglistPlaceholders);
449 
450  // Placeholders to map the real storages to the placeholders (no temporaries)
451  std::vector<std::string> DomainMapPlaceholders;
452  std::transform(StencilFields.begin() + numTemporaries, StencilFields.end(),
453  std::back_inserter(DomainMapPlaceholders),
454  [](const Stencil::FieldInfo& field) { return field.Name; });
455  for(const auto& var : StencilGlobalVariables)
456  DomainMapPlaceholders.push_back("globals::get()." + var + ".as_global_parameter()");
457 
458  // This is a memory leak.. but nothing we can do ;)
459  // https://github.com/eth-cscs/gridtools/issues/621
460  StencilConstructor.addStatement("auto gt_domain = new " + gt() +
461  "aggregator_type<domain_arg_list>(" +
462  RangeToString(", ", "", ")")(DomainMapPlaceholders));
463 
464  // Generate grid
465  StencilConstructor.addComment("Grid");
466  StencilConstructor.addStatement("unsigned int di[5] = {dom.iminus(), dom.iminus(), "
467  "dom.iplus(), dom.isize() - 1 - dom.iplus(), dom.isize()}");
468  StencilConstructor.addStatement("unsigned int dj[5] = {dom.jminus(), dom.jminus(), "
469  "dom.jplus(), dom.jsize() - 1 - dom.jplus(), dom.jsize()}");
470  StencilConstructor.addStatement("m_grid = std::unique_ptr<grid_stencil_" +
471  std::to_string(stencilIdx) + ">(new grid_stencil_" +
472  std::to_string(stencilIdx) + "(di, dj))");
473 
474  auto getLevelSize = [](int level) -> std::string {
475  switch(level) {
476  case sir::Interval::Start:
477  return "dom.kminus()";
478  case sir::Interval::End:
479  return "dom.ksize() == 0 ? 0 : dom.ksize() - dom.kplus() - 1";
480  default:
481  return std::to_string(level);
482  }
483  };
484 
485  int levelIdx = 0;
486  for(auto it = intervalDefinitions.Levels.begin(), end = intervalDefinitions.Levels.end();
487  it != end; ++it, ++levelIdx)
488  StencilConstructor.addStatement("m_grid->value_list[" + std::to_string(levelIdx) + "] = " +
489  getLevelSize(*it));
490 
491  // Generate make_computation
492  StencilConstructor.addComment("Computation");
493  StencilConstructor.addStatement(
494  Twine("m_stencil = gridtools::make_computation<gridtools::clang::backend_t>(*gt_domain, "
495  "*m_grid") +
496  RangeToString(", ", ", ", ")")(makeComputation));
497 
498  StencilConstructor.commit();
499 
500  // Generate destructor
501  auto dtor = StencilClass.addDestructor();
502  dtor.addStatement("m_stencil->finalize()");
503  dtor.commit();
504 
505  // Generate stencil getter
506  StencilClass.addMemberFunction("gridtools::stencil*", "get_stencil")
507  .addStatement("return m_stencil.get()");
508  }
509 
510  if(isEmpty) {
511  DiagnosticsBuilder diag(DiagnosticsKind::Error, stencilInstantiation->getSIRStencil()->Loc);
512  diag << "empty stencil '" << stencilInstantiation->getName()
513  << "', this would result in invalid gridtools code";
514  context_->getDiagnostics().report(diag);
515  return "";
516  }
517 
518  //
519  // Generate constructor/destructor and methods of the stencil wrapper
520  //
521  StencilWrapperClass.addComment("Members");
522 
523  // Define allocated memebers if necessary
524  if(stencilInstantiation->hasAllocatedFields()) {
525  StencilWrapperClass.addMember(gtc() + "meta_data_t", "m_meta_data");
526 
527  for(int AccessID : stencilInstantiation->getAllocatedFieldAccessIDs())
528  StencilWrapperClass.addMember(gtc() + "storage_t",
529  "m_" + stencilInstantiation->getNameFromAccessID(AccessID));
530  }
531 
532  // Stencil members
533  std::vector<std::string> stencilMembers;
534  for(std::size_t i = 0; i < stencils.size(); ++i) {
535  StencilWrapperClass.addMember("stencil_" + Twine(i), "m_stencil_" + Twine(i));
536  stencilMembers.emplace_back("m_stencil_" + std::to_string(i));
537  }
538 
539  StencilWrapperClass.addMember("static constexpr const char* s_name =",
540  Twine("\"") + StencilWrapperClass.getName() + Twine("\""));
541 
542  StencilWrapperClass.changeAccessibility("public");
543  StencilWrapperClass.addCopyConstructor(Class::Deleted);
544 
545  // Generate stencil wrapper constructor
546  auto SIRFieldsWithoutTemps = stencilInstantiation->getSIRStencil()->Fields;
547  for(auto it = SIRFieldsWithoutTemps.begin(); it != SIRFieldsWithoutTemps.end();)
548  if((*it)->IsTemporary)
549  it = SIRFieldsWithoutTemps.erase(it);
550  else
551  ++it;
552 
553  std::vector<std::string> StencilWrapperConstructorTemplates;
554  for(int i = 0; i < SIRFieldsWithoutTemps.size(); ++i)
555  StencilWrapperConstructorTemplates.push_back("S" + std::to_string(i + 1));
556 
557  auto StencilWrapperConstructor = StencilWrapperClass.addConstructor(RangeToString(", ", "", "")(
558  StencilWrapperConstructorTemplates, [](const std::string& str) { return "class " + str; }));
559 
560  StencilWrapperConstructor.addArg("const " + gtc() + "domain& dom");
561  for(int i = 0; i < SIRFieldsWithoutTemps.size(); ++i)
562  StencilWrapperConstructor.addArg(StencilWrapperConstructorTemplates[i] + "& " +

644 std::string GTClangCodeGen::generateGlobals(const SIR* Sir) {
645  using namespace codegen;
646 
647  const auto& globalsMap = *Sir->GlobalVariableMap;
648  if(globalsMap.empty())
649  return "";
650 
651  std::stringstream ss;
652 
653  std::string StructName = "globals";
654  std::string BaseName = "gridtools::clang::globals_impl<" + StructName + ">";
655 
656  Struct GlobalsStruct(StructName + ": public " + BaseName, ss);
657  GlobalsStruct.addTypeDef("base_t").addType("gridtools::clang::globals_impl<globals>");
658 
659  for(const auto& globalsPair : globalsMap) {
660  sir::Value& value = *globalsPair.second;
661  std::string Name = globalsPair.first;
662  std::string Type = sir::Value::typeToString(value.getType());
663  std::string AdapterBase = std::string("base_t::variable_adapter_impl") + "<" + Type + ">";
664 
665  Structure AdapterStruct = GlobalsStruct.addStructMember(Name + "_adapter", Name, AdapterBase);
666  AdapterStruct.addConstructor().addArg("").addInit(
667  AdapterBase + "(" + Type + "(" + (value.empty() ? std::string() : value.toString()) + "))");
668 
669  auto AssignmentOperator =
670  AdapterStruct.addMemberFunction(Name + "_adapter&", "operator=", "class ValueType");
671  AssignmentOperator.addArg("ValueType&& value");
672  if(value.isConstexpr())
673  AssignmentOperator.addStatement(
674  "throw std::runtime_error(\"invalid assignment to constant variable '" + Name + "'\")");
675  else
676  AssignmentOperator.addStatement("get_value() = value");
677  AssignmentOperator.addStatement("return *this");
678  AssignmentOperator.commit();
679  }
680 
681  GlobalsStruct.commit();
682 
683  // Add the symbol for the singleton
684  codegen::Statement(ss) << "template<> " << StructName << "* " << BaseName
685  << "::s_instance = nullptr";
686 
687  // Remove trailing ';' as this is retained by Clang's Rewriter
688  std::string str = ss.str();
689  str[str.size() - 2] = ' ';
690 
691  return str;

694 std::unique_ptr<TranslationUnit> GTClangCodeGen::generateCode() {
695  mplContainerMaxSize_ = 20;
696  DAWN_LOG(INFO) << "Starting code generation for GTClang ...";
697 
698  // Generate StencilInstantiations
699  std::map<std::string, std::string> stencils;
700  for(const auto& nameStencilCtxPair : context_->getStencilInstantiationMap()) {
701  std::string code = generateStencilInstantiation(nameStencilCtxPair.second.get());
702  if(code.empty())
703  return nullptr;
704  stencils.emplace(nameStencilCtxPair.first, std::move(code));
705  }
706 
707  // Generate globals
708  std::string globals = generateGlobals(context_->getSIR());
709 
710  std::vector<std::string> ppDefines;
711  auto makeDefine = [](std::string define, int value) {
712  return "#define " + define + " " + std::to_string(value);
713  };
714 
715  auto makeIfNotDefined = [](std::string define, int value) {
716  return "#ifndef " + define + "\n #define " + define + " " + std::to_string(value) + "\n#endif";
717  };
718 
719  ppDefines.push_back(makeDefine("GRIDTOOLS_CLANG_GENERATED", 1));
720  ppDefines.push_back(
721  makeDefine("GRIDTOOLS_CLANG_HALO_EXTEND", context_->getOptions().MaxHaloPoints));
722  ppDefines.push_back(makeIfNotDefined("BOOST_RESULT_OF_USE_TR1", 1));
723  ppDefines.push_back(makeIfNotDefined("BOOST_NO_CXX11_DECLTYPE", 1));
724 
725  // If we need more than 20 elements in boost::mpl containers, we need to increment to the nearest
726  // multiple of ten
727  // http://www.boost.org/doc/libs/1_61_0/libs/mpl/doc/refmanual/limit-vector-size.html
728  if(mplContainerMaxSize_ > 20) {
729  mplContainerMaxSize_ += (10 - mplContainerMaxSize_ % 10);
730  DAWN_LOG(INFO) << "increasing boost::mpl template limit to " << mplContainerMaxSize_;
731  ppDefines.push_back(makeIfNotDefined("BOOST_MPL_CFG_NO_PREPROCESSED_HEADERS", 1));
732  }
733 
734  DAWN_ASSERT_MSG(mplContainerMaxSize_ % 10 == 0,
735  "boost::mpl template limit needs to be multiple of 10");
736 
737  ppDefines.push_back(makeIfNotDefined("FUSION_MAX_VECTOR_SIZE", mplContainerMaxSize_));
738  ppDefines.push_back(makeIfNotDefined("FUSION_MAX_MAP_SIZE", mplContainerMaxSize_));
739  ppDefines.push_back(makeIfNotDefined("BOOST_MPL_LIMIT_VECTOR_SIZE", mplContainerMaxSize_));
740 
741  DAWN_LOG(INFO) << "Done generating code";
742 
743  return make_unique<TranslationUnit>(context_->getSIR()->Filename, std::move(ppDefines),
744  std::move(stencils), std::move(globals));
745 }
