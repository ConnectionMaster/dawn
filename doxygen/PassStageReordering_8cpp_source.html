32  OptimizerContext* context = stencilInstantiation->getOptimizerContext();
33 
34  std::string filenameWE = getFilenameWithoutExtension(context->getSIR()->Filename);
35  if(context->getOptions().ReportPassStageReodering)
36  stencilInstantiation->dumpAsJson(filenameWE + "_before.json", getName());
37 
38  for(auto& stencilPtr : stencilInstantiation->getStencils()) {
39  if(strategy_ == ReorderStrategy::RK_None)
40  continue;
41 
42  std::unique_ptr<ReorderStrategy> strategy;
43  switch(strategy_) {
44  case ReorderStrategy::RK_Greedy:
45  strategy = make_unique<ReoderStrategyGreedy>();
46  break;
47  case ReorderStrategy::RK_Partitioning:
48  strategy = make_unique<ReoderStrategyPartitioning>();
49  break;
50  default:
51  dawn_unreachable("invalid reorder strategy");
52  }
53 
54  stencilPtr = strategy->reorder(stencilPtr);
55  if(!stencilPtr)
56  return false;
57  }
58 
59  if(context->getOptions().ReportPassStageReodering)
60  stencilInstantiation->dumpAsJson(filenameWE + "_after.json", getName());
61 
62  return true;
63 }
