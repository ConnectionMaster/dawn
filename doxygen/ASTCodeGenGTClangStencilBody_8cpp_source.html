123 void ASTCodeGenGTClangStencilBody::visit(const std::shared_ptr<StencilFunCallExpr>& expr) {
124  if(nestingOfStencilFunArgLists_++)
125  ss_ << ", ";
126 
127  const StencilFunctionInstantiation* stencilFun =
128  currentFunction_ ? currentFunction_->getStencilFunctionInstantiation(expr)
129  : instantiation_->getStencilFunctionInstantiation(expr);
130 
131  ss_ << "gridtools::call<" << StencilFunctionInstantiation::makeCodeGenName(*stencilFun) << ", "
132  << intervalToNameMap_.find(stencilFun->getInterval())->second << ">::with_offsets(eval";
133 
134  for(auto& arg : expr->getArguments())
135  arg->accept(*this);
136 
137  nestingOfStencilFunArgLists_--;
138  ss_ << ")";
139 }

143 void ASTCodeGenGTClangStencilBody::visit(const std::shared_ptr<VarAccessExpr>& expr) {
144  std::string name = getName(expr);
145  int AccessID = getAccessID(expr);
146 
147  if(instantiation_->isGlobalVariable(AccessID)) {
148  if(!nestingOfStencilFunArgLists_)
149  ss_ << "eval(";
150  else
151  ss_ << ", ";
152 
153  ss_ << name << "()";
154 
155  if(!nestingOfStencilFunArgLists_)
156  ss_ << ").value";
157 
158  } else {
159  ss_ << name;
160 
161  if(expr->isArrayAccess()) {
162  ss_ << "[";
163  expr->getIndex()->accept(*this);
164  ss_ << "]";
165  }
166  }

173 void ASTCodeGenGTClangStencilBody::visit(const std::shared_ptr<FieldAccessExpr>& expr) {
174  if(!nestingOfStencilFunArgLists_)
175  ss_ << "eval(";
176  else
177  ss_ << ", ";
178 
179  if(currentFunction_) {
180  ss_ << currentFunction_->getOriginalNameFromCallerAccessID(
181  currentFunction_->getAccessIDFromExpr(expr))
182  << offsetPrinter_(currentFunction_->evalOffsetOfFieldAccessExpr(expr, false));
183  } else
184  ss_ << getName(expr) << offsetPrinter_(expr->getOffset());
185 
186  if(!nestingOfStencilFunArgLists_)
187  ss_ << ")";
188 }
