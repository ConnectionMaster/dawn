33 bool PassStageSplitter::run(StencilInstantiation* stencilInstantiation) {
34  int numSplit = 0;
35  std::deque<int> splitterIndices;
36  std::deque<std::shared_ptr<DependencyGraphAccesses>> graphs;
37  OptimizerContext* context = stencilInstantiation->getOptimizerContext();
38 
39  // Iterate over all stages in all multistages of all stencils
40  for(auto& stencil : stencilInstantiation->getStencils()) {
41 
42  int multiStageIndex = 0;
43  int linearStageIndex = 0;
44  for(auto& multiStage : stencil->getMultiStages()) {
45 
46  int stageIndex = 0;
47  for(auto stageIt = multiStage->getStages().begin(); stageIt != multiStage->getStages().end();
48  ++stageIndex, ++linearStageIndex) {
49  Stage& stage = (**stageIt);
50  DoMethod& doMethod = stage.getSingleDoMethod();
51 
52  splitterIndices.clear();
53  graphs.clear();
54 
55  std::shared_ptr<DependencyGraphAccesses> newGraph, oldGraph;
56  newGraph = std::make_shared<DependencyGraphAccesses>(stencilInstantiation);
57 
58  // Build the Dependency graph (bottom to top)
59  for(int stmtIndex = doMethod.getStatementAccessesPairs().size() - 1; stmtIndex >= 0;
60  --stmtIndex) {
61  auto& stmtAccessesPair = doMethod.getStatementAccessesPairs()[stmtIndex];
62 
63  newGraph->insertStatementAccessesPair(stmtAccessesPair);
64 
65  // If we have a horizontal read-before-write conflict, we record the current index for
66  // splitting
67  if(hasHorizontalReadBeforeWriteConflict(newGraph.get())) {
68 
69  if(context->getOptions().DumpSplitGraphs)
70  oldGraph->toDot(
71  format("stmt_hd_ms%i_s%i_%02i.dot", multiStageIndex, stageIndex, numSplit));
72 
73  // Set the splitter index and assign the *old* graph as the stage dependency graph
74  splitterIndices.push_front(stmtIndex);
75  graphs.push_front(std::move(oldGraph));
76 
77  if(context->getOptions().ReportPassStageSplit)
78  std::cout << "\nPASS: " << getName() << ": " << stencilInstantiation->getName()
79  << ": split:"
80  << stmtAccessesPair->getStatement()->ASTStmt->getSourceLocation().Line
81  << "\n";
82 
83  // Clear the new graph an process the current statements again
84  newGraph->clear();
85  newGraph->insertStatementAccessesPair(stmtAccessesPair);
86 
87  numSplit++;
88  }
89 
90  oldGraph = newGraph->clone();
91  }
92 
93  if(context->getOptions().DumpSplitGraphs)
94  newGraph->toDot(
95  format("stmt_hd_ms%i_s%i_%02i.dot", multiStageIndex, stageIndex, numSplit));
96 
97  graphs.push_front(newGraph);
98 
99  // Perform the spliting of the stages and insert the stages *before* the stage we processed.
100  // Note that the "old" stage will be erased (it was consumed in split(...) anyway)
101  if(!splitterIndices.empty()) {
102  auto newStages = stage.split(splitterIndices, &graphs);
103  stageIt = multiStage->getStages().erase(stageIt);
104  multiStage->getStages().insert(stageIt, std::make_move_iterator(newStages.begin()),
105  std::make_move_iterator(newStages.end()));
106  } else {
107  DAWN_ASSERT(graphs.size() == 1);
108  doMethod.setDependencyGraph(graphs.back());
109  stage.update();
110  ++stageIt;
111  }
112  }
113 
114  multiStageIndex += 1;
115  }
116  }
117 
118  if(context->getOptions().ReportPassStageSplit && !numSplit)
119  std::cout << "\nPASS: " << getName() << ": " << stencilInstantiation->getName()
120  << ": no split\n";
121 
122  return true;
123 }
