34 static bool pointerMapComparison(const sir::GlobalVariableMap& map1,
35  const sir::GlobalVariableMap& map2) {
36  if(map1.size() != map2.size()) {
37  return false;
38  } else {
39  for(auto& a : map1) {
40  auto finder = map2.find(a.first);
41  if(finder == map2.end()) {
42  return false;
43  } else if(finder->second == nullptr || a.second == nullptr) {
44  return finder->second == a.second;
45  } else if(!(*(finder->second.get()) == *(a.second.get()))) {
46  return false;
47  }
48  }
49  return true;
50  }
51 }

93 std::ostream& operator<<(std::ostream& os, const SIR& Sir) {
94  const char* indent1 = MakeIndent<1>::value;
95  const char* indent2 = MakeIndent<2>::value;
96 
97  os << "SIR : " << Sir.Filename << "\n{\n";
98  for(const auto& stencilFunction : Sir.StencilFunctions) {
99  os << "\n"
100  << indent1 << "StencilFunction : " << stencilFunction->Name << "\n"
101  << indent1 << "{\n";
102  for(const auto& arg : stencilFunction->Args) {
103  if(sir::Field* field = dyn_cast<sir::Field>(arg.get()))
104  os << indent2 << "Field : " << field->Name << "\n";
105  if(sir::Direction* dir = dyn_cast<sir::Direction>(arg.get()))
106  os << indent2 << "Direction : " << dir->Name << "\n";
107  if(sir::Offset* offset = dyn_cast<sir::Offset>(arg.get()))
108  os << indent2 << "Offset : " << offset->Name << "\n";
109  }
110 
111  if(!stencilFunction->isSpecialized()) {
112  os << "\n" << indent2 << "Do\n";
113  os << ASTStringifer::toString(*stencilFunction->Asts[0], 2 * DAWN_PRINT_INDENT);
114  } else {
115  for(int i = 0; i < stencilFunction->Intervals.size(); ++i) {
116  os << "\n" << indent2 << "Do " << *stencilFunction->Intervals[i].get() << "\n";
117  os << ASTStringifer::toString(*stencilFunction->Asts[i], 2 * DAWN_PRINT_INDENT);
118  }
119  }
120  os << indent1 << "}\n";
121  }
122 
123  for(const auto& stencil : Sir.Stencils) {
124  os << "\n" << indent1 << "Stencil : " << stencil->Name << "\n" << indent1 << "{\n";
125  for(const auto& field : stencil->Fields)
126  os << indent2 << "Field : " << field->Name << "\n";
127  os << "\n";
128 
129  os << indent2 << "Do\n"
130  << ASTStringifer::toString(*stencil->StencilDescAst, 2 * DAWN_PRINT_INDENT);
131  os << indent1 << "}\n";
132  }
133 
134  os << "\n}";
135  return os;
136 }
