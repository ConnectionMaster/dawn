74  std::set<int> levels;
75  for(const Interval& interval : intervals) {
76  levels.insert(interval.lowerLevel());
77  levels.insert(interval.upperLevel());
78  }
79 
80  std::vector<Interval> newIntervals;
81 
82  int lowerLevel = *levels.begin();
83  newIntervals.emplace_back(lowerLevel, lowerLevel);
84  for(auto it = std::next(levels.begin()), end = levels.end(); it != end; ++it) {
85  int upperLevel = *it;
86 
87  if(((upperLevel - 1) - (lowerLevel + 1)) >= 0)
88  newIntervals.emplace_back(lowerLevel + 1, upperLevel - 1);
89 
90  newIntervals.emplace_back(upperLevel, upperLevel);
91  lowerLevel = upperLevel;
92  }

97 std::vector<Interval> Interval::computeGapIntervals(const Interval& axis,
98  const std::vector<Interval>& intervals) {
99  std::vector<Interval> newIntervals;
100 
101  // Insert the intervals in sorted order
102  for(const Interval& interval : intervals) {
103  auto it = std::find_if(newIntervals.begin(), newIntervals.end(), [&](const Interval& I) {
104  DAWN_ASSERT_MSG(I.lowerBound() != interval.lowerBound(),
105  "Intervals have to be non-overlapping");
106  return I.lowerBound() > interval.lowerBound();
107  });
108  newIntervals.insert(it, interval);
109  }
110 
111  // Close the intermediate gaps
112  if(newIntervals.size() > 1) {
113  for(auto curLowIt = newIntervals.begin(), curTopIt = std::next(newIntervals.begin());
114  curTopIt != newIntervals.end();) {
115  const Interval& curLowInterval = *curLowIt;
116  const Interval& curTopInterval = *curTopIt;
117 
118  DAWN_ASSERT_MSG(!curLowInterval.overlaps(curTopInterval),
119  "Intervals have to be non-overlapping");
120 
121  if(!curLowInterval.adjacent(curTopInterval)) {
122  Interval gapFillInterval(curLowInterval.upperLevel(), curTopInterval.lowerLevel(),
123  curLowInterval.upperOffset() + 1,
124  curTopInterval.lowerOffset() - 1);
125  curLowIt = newIntervals.insert(curTopIt, gapFillInterval);
126  curTopIt = std::next(curLowIt);
127  }
128 
129  curLowIt = curTopIt;
130  curTopIt++;
131  }
132  }
133 
134  // Close the lower gap
135  if(axis.lowerBound() != newIntervals.front().lowerBound()) {
136  const Interval& lowestInterval = newIntervals.front();
137  Interval lowerFillInterval(axis.lowerLevel(), lowestInterval.lowerLevel(), axis.lowerOffset(),
138  lowestInterval.lowerOffset() - 1);
139  newIntervals.insert(newIntervals.begin(), lowerFillInterval);
140  }
141 
142  // Close the upper gap
143  if(axis.upperBound() != newIntervals.back().upperBound()) {
144  const Interval& topInterval = newIntervals.back();
145  Interval topFillInterval(topInterval.upperLevel(), axis.upperLevel(),
146  topInterval.upperOffset() + 1, axis.upperOffset());
147  newIntervals.insert(newIntervals.end(), topFillInterval);
148  }
149 
150  return newIntervals;
151 }
