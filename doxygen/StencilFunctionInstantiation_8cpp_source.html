46 Array3i StencilFunctionInstantiation::evalOffsetOfFieldAccessExpr(
47  const std::shared_ptr<FieldAccessExpr>& expr, bool applyInitialOffset) const {
48 
49  // Get the offsets we know so far (i.e the constant offset)
50  Array3i offset = expr->getOffset();
51 
52  // Apply the initial offset (e.g if we call a function `avg(in(i+1))` we have to shift all
53  // accesses of the field `in` by [1, 0, 0])
54  if(applyInitialOffset) {
55  const Array3i& initialOffset = getCallerInitialOffsetFromAccessID(getAccessIDFromExpr(expr));
56  offset[0] += initialOffset[0];
57  offset[1] += initialOffset[1];
58  offset[2] += initialOffset[2];
59  }
60 
61  int sign = expr->negateOffset() ? -1 : 1;
62 
63  // Iterate the argument map (if index is *not* -1, we have to lookup the dimension or offset of
64  // the directional or offset argument)
65  for(int i = 0; i < expr->getArgumentMap().size(); ++i) {
66  const int argIndex = expr->getArgumentMap()[i];
67 
68  if(argIndex != -1) {
69  const int argOffset = expr->getArgumentOffset()[i];
70 
71  // Resolve the directions and offsets
72  if(isArgDirection(argIndex))
73  offset[getCallerDimensionOfArgDirection(argIndex)] += sign * argOffset;
74  else {
75  const auto& instantiatedOffset = getCallerOffsetOfArgOffset(argIndex);
76  offset[instantiatedOffset[0]] += sign * (argOffset + instantiatedOffset[1]);
77  }
78  }
79  }
80 
81  return offset;
82 }

214 void StencilFunctionInstantiation::renameCallerAccessID(int oldAccessID, int newAccessID) {
215  // Update argument maps
216  for(auto& argumentAccessIDPair : ArgumentIndexToCallerAccessIDMap_) {
217  int& AccessID = argumentAccessIDPair.second;
218  if(AccessID == oldAccessID)
219  AccessID = newAccessID;
220  }
221  replaceKeyInMap(CallerAcceessIDToInitialOffsetMap_, oldAccessID, newAccessID);
222 
223  // Update AccessID to name map
224  replaceKeyInMap(AccessIDToNameMap_, oldAccessID, newAccessID);
225 
226  // Update statements
227  renameAccessIDInStmts(this, oldAccessID, newAccessID, statementAccessesPairs_);
228 
229  // Update accesses
230  renameAccessIDInAccesses(this, oldAccessID, newAccessID, statementAccessesPairs_);
231 
232  // Recompute the fields
233  update();
234 }

343 void StencilFunctionInstantiation::update() {
344  callerFields_.clear();
345  calleeFields_.clear();
346  unusedFields_.clear();
347 
348  // Compute the fields and their intended usage. Fields can be in one of three states: `Output`,
349  // `InputOutput` or `Input` which implements the following state machine:
350  //
351  // +-------+ +--------+
352  // | Input | | Output |
353  // +-------+ +--------+
354  // | |
355  // | +-------------+ |
356  // +----------> | InputOutput | <----------+
357  // +-------------+
358  //
359  std::set<int> inputOutputFields;
360  std::set<int> inputFields;
361  std::set<int> outputFields;
362 
363  for(const auto& statementAccessesPair : statementAccessesPairs_) {
364  auto& access = statementAccessesPair->getAccesses();
365 
366  for(const auto& accessPair : access->getWriteAccesses()) {
367  int AccessID = accessPair.first;
368 
369  // Does this AccessID correspond to a field access?
370  if(!isProvidedByStencilFunctionCall(AccessID) && !stencilInstantiation_->isField(AccessID))
371  continue;
372 
373  // Field was recorded as `InputOutput`, state can't change ...
374  if(inputOutputFields.count(AccessID))
375  continue;
376 
377  // Field was recorded as `Input`, change it's state to `InputOutput`
378  if(inputFields.count(AccessID)) {
379  inputOutputFields.insert(AccessID);
380  inputFields.erase(AccessID);
381  continue;
382  }
383 
384  // Field not yet present, record it as output
385  outputFields.insert(AccessID);
386  }
387 
388  for(const auto& accessPair : access->getReadAccesses()) {
389  int AccessID = accessPair.first;
390 
391  // Does this AccessID correspond to a field access?
392  if(!isProvidedByStencilFunctionCall(AccessID) && !stencilInstantiation_->isField(AccessID))
393  continue;
394 
395  // Field was recorded as `InputOutput`, state can't change ...
396  if(inputOutputFields.count(AccessID))
397  continue;
398 
399  // Field was recorded as `Output`, change it's state to `InputOutput`
400  if(outputFields.count(AccessID)) {
401  inputOutputFields.insert(AccessID);
402  outputFields.erase(AccessID);
403  continue;
404  }
405 
406  // Field not yet present, record it as input
407  inputFields.insert(AccessID);
408  }
409  }
410 
411  // Add AccessIDs of unused fields i.e fields which are passed as arguments but never referenced.
412  for(const auto& argIdxCallerAccessIDPair : ArgumentIndexToCallerAccessIDMap_) {
413  int AccessID = argIdxCallerAccessIDPair.second;
414  if(!inputFields.count(AccessID) && !outputFields.count(AccessID) &&
415  !inputOutputFields.count(AccessID)) {
416  inputFields.insert(AccessID);
417  unusedFields_.insert(AccessID);
418  }
419  }
420 
421  std::vector<Field> calleeFieldsUnordered;
422  std::vector<Field> callerFieldsUnordered;
423 
424  // Merge inputFields, outputFields and fields. Note that caller and callee fields are the same,
425  // the only difference is that in the caller fields we apply the inital offset to the extents
426  // while in the callee fields we do not.
427  for(int AccessID : outputFields) {
428  calleeFieldsUnordered.emplace_back(AccessID, Field::IK_Output);
429  callerFieldsUnordered.emplace_back(AccessID, Field::IK_Output);
430  }
431 
432  for(int AccessID : inputOutputFields) {
433  calleeFieldsUnordered.emplace_back(AccessID, Field::IK_InputOutput);
434  callerFieldsUnordered.emplace_back(AccessID, Field::IK_InputOutput);
435  }
436 
437  for(int AccessID : inputFields) {
438  calleeFieldsUnordered.emplace_back(AccessID, Field::IK_Input);
439  callerFieldsUnordered.emplace_back(AccessID, Field::IK_Input);
440  }
441 
442  if(calleeFieldsUnordered.empty() || callerFieldsUnordered.empty()) {
443  DAWN_LOG(WARNING) << "no fields referenced in this stencil function";
444  } else {
445 
446  // Accumulate the extent of the fields (note that here the callee and caller fields differ
447  // as the caller fields have the *initial* extent (e.g in `avg(u(i+1))` u has an initial extent
448  // of [1, 0, 0])
449  auto computeAccesses = [&](std::vector<Field>& fields, bool callerAccesses) {
450 
451  // Index to speedup lookup into fields map
452  std::unordered_map<int, std::vector<Field>::iterator> AccessIDToFieldMap;
453  for(auto it = fields.begin(), end = fields.end(); it != end; ++it)
454  AccessIDToFieldMap.insert(std::make_pair(it->AccessID, it));
455 
456  // Accumulate the extents of each field in this stage
457  for(const auto& statementAccessesPair : statementAccessesPairs_) {
458  const auto& access = callerAccesses ? statementAccessesPair->getCallerAccesses()
459  : statementAccessesPair->getCalleeAccesses();
460 
461  // first => AccessID, second => Extent
462  for(auto& accessPair : access->getWriteAccesses()) {
463  if(!isProvidedByStencilFunctionCall(accessPair.first) &&
464  !stencilInstantiation_->isField(accessPair.first))
465  continue;
466 
467  AccessIDToFieldMap[accessPair.first]->Extent.merge(accessPair.second);
468  }
469 
470  for(const auto& accessPair : access->getReadAccesses()) {
471  if(!isProvidedByStencilFunctionCall(accessPair.first) &&
472  !stencilInstantiation_->isField(accessPair.first))
473  continue;
474 
475  AccessIDToFieldMap[accessPair.first]->Extent.merge(accessPair.second);
476  }
477  }
478  };
479 
480  computeAccesses(callerFieldsUnordered, true);
481  computeAccesses(calleeFieldsUnordered, false);
482  }
483 
484  // Reorder the fields s.t they match the order in which they were decalred in the stencil-function
485  for(int argIdx = 0; argIdx < getArguments().size(); ++argIdx) {
486  if(isArgField(argIdx)) {
487  int AccessID = getCallerAccessIDOfArgField(argIdx);
488 
489  auto insertField = [&](std::vector<Field>& fieldsOrdered,
490  std::vector<Field>& fieldsUnordered) {
491  auto it = std::find_if(fieldsUnordered.begin(), fieldsUnordered.end(),
492  [&](const Field& field) { return field.AccessID == AccessID; });
493  DAWN_ASSERT(it != fieldsUnordered.end());
494  fieldsOrdered.push_back(*it);
495  };
496 
497  insertField(callerFields_, callerFieldsUnordered);
498  insertField(calleeFields_, calleeFieldsUnordered);
499  }
500  }
501 }

511 std::string
512 StencilFunctionInstantiation::makeCodeGenName(const StencilFunctionInstantiation& stencilFun) {
513  std::string name = stencilFun.getName();
514 
515  for(std::size_t argIdx = 0; argIdx < stencilFun.getStencilFunction()->Args.size(); ++argIdx) {
516  if(stencilFun.isArgOffset(argIdx)) {
517  const auto& offset = stencilFun.getCallerOffsetOfArgOffset(argIdx);
518  name += "_" + dim2str(offset[0]) + "_";
519  if(offset[1] != 0)
520  name += (offset[1] > 0 ? "plus_" : "minus_");
521  name += std::to_string(std::abs(offset[1]));
522 
523  } else if(stencilFun.isArgDirection(argIdx)) {
524  name += "_" + dim2str(stencilFun.getCallerDimensionOfArgDirection(argIdx));
525  }
526  }
527 
528  name += "_" + Interval::makeCodeGenName(stencilFun.getInterval());
529  return name;
530 }

538 void StencilFunctionInstantiation::dump() {
539  std::cout << "\nStencilFunction : " << getName() << " " << getInterval() << "\n";
540  std::cout << MakeIndent<1>::value << "Arguments:\n";
541 
542  for(std::size_t argIdx = 0; argIdx < function_->Args.size(); ++argIdx) {
543 
544  std::cout << MakeIndent<2>::value << "arg(" << argIdx << ") : ";
545 
546  if(isArgOffset(argIdx)) {
547  int dim = getCallerOffsetOfArgOffset(argIdx)[0];
548  int offset = getCallerOffsetOfArgOffset(argIdx)[1];
549  std::cout << "Offset : " << dim2str(dim);
550  if(offset != 0)
551  std::cout << (offset > 0 ? "+" : "") << offset;
552  } else if(isArgField(argIdx)) {
553  sir::Field* field = dyn_cast<sir::Field>(function_->Args[argIdx].get());
554  std::cout << "Field : " << field->Name << " -> ";
555  if(isArgStencilFunctionInstantiation(argIdx)) {
556  std::cout << "stencil-function-call:"
557  << getFunctionInstantiationOfArgField(argIdx)->getName();
558  } else {
559  int callerAccessID = getCallerAccessIDOfArgField(argIdx);
560  std::cout << stencilInstantiation_->getNameFromAccessID(callerAccessID) << " "
561  << getCallerInitialOffsetFromAccessID(callerAccessID);
562  }
563 
564  } else {
565  std::cout << "Direction : " << dim2str(getCallerDimensionOfArgDirection(argIdx));
566  }
567  std::cout << "\n";
568  }
569 
570  std::cout << MakeIndent<1>::value << "Accesses (including initial offset):\n";
571 
572  const auto& statements = getAST()->getRoot()->getStatements();
573  for(std::size_t i = 0; i < statements.size(); ++i) {
574  std::cout << "\e[1m" << ASTStringifer::toString(statements[i], 2 * DAWN_PRINT_INDENT)
575  << "\e[0m";
576  if(statementAccessesPairs_[i]->getCallerAccesses())
577  std::cout << statementAccessesPairs_[i]->getCallerAccesses()->toString(this,
578  3 * DAWN_PRINT_INDENT)
579  << "\n";
580  }
581  std::cout.flush();
582 }
