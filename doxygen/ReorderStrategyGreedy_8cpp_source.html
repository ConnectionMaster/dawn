32 template <typename ReturnType = std::pair<std::shared_ptr<DependencyGraphAccesses>, LoopOrderKind>>
33 ReturnType isMergable(const Stage& stage, LoopOrderKind stageLoopOrder,
34  const MultiStage& multiStage) {
35  LoopOrderKind multiStageLoopOrder = multiStage.getLoopOrder();
36  auto multiStageDependencyGraph =
37  multiStage.getDependencyGraphOfInterval(stage.getEnclosingExtendedInterval());
38 
39  // Merge stage into dependency graph
40  const DoMethod& doMethod = stage.getSingleDoMethod();
41  multiStageDependencyGraph->merge(doMethod.getDependencyGraph().get());
42 
43  // Try all possible loop orders while *favoring* a parallel loop order. Note that a parallel loop
44  // order can be changed to forward or backward.
45  //
46  // MULTI-STAGE
47  //
48  // | P | F | B | P = Parallel
49  // -----+-----+-----+-----+ F = Foward
50  // S P | PFB F B | B = Backward
51  // T -----+ + X = Incompatible
52  // A F | F F X |
53  // G -----+ +
54  // E B | B X B |
55  // -----+-----------------+
56  //
57  std::vector<LoopOrderKind> possibleLoopOrders;
58 
59  if(multiStageLoopOrder == LoopOrderKind::LK_Parallel &&
60  stageLoopOrder == LoopOrderKind::LK_Parallel)
61  possibleLoopOrders = {LoopOrderKind::LK_Parallel, LoopOrderKind::LK_Forward,
62  LoopOrderKind::LK_Backward};
63  else if(stageLoopOrder == LoopOrderKind::LK_Parallel)
64  possibleLoopOrders.push_back(multiStageLoopOrder);
65  else
66  possibleLoopOrders.push_back(stageLoopOrder);
67 
68  if(multiStageDependencyGraph->empty())
69  return ReturnType(multiStageDependencyGraph, possibleLoopOrders.front());
70 
71  // If the resulting graph isn't a DAG anymore that isn't gonna work
72  if(!multiStageDependencyGraph->isDAG())
73  return ReturnType(nullptr, multiStageLoopOrder);
74 
75  // Check all possible loop orders if there aren't any vertical conflicts
76  for(auto loopOrder : possibleLoopOrders) {
77  auto conflict = hasVerticalReadBeforeWriteConflict(multiStageDependencyGraph.get(), loopOrder);
78  if(!conflict.CounterLoopOrderConflict)
79  return ReturnType(multiStageDependencyGraph, loopOrder);
80  }
81 
82  return ReturnType(nullptr, multiStageLoopOrder);
83 }

85 std::shared_ptr<Stencil> ReoderStrategyGreedy::reorder(const std::shared_ptr<Stencil>& stencilPtr) {
86  Stencil& stencil = *stencilPtr;
87 
88  DependencyGraphStage& stageDAG = *stencil.getStageDependencyGraph();
89  StencilInstantiation* instantiation = stencil.getStencilInstantiation();
90 
91  std::shared_ptr<Stencil> newStencil =
92  std::make_shared<Stencil>(instantiation, stencil.getSIRStencil(), stencilPtr->getStencilID(),
93  stencil.getStageDependencyGraph());
94  int newNumStages = 0;
95  int newNumMultiStages = 0;
96 
97  const int maxBoundaryExtent = instantiation->getOptimizerContext()->getOptions().MaxHaloPoints;
98 
99  auto pushBackNewMultiStage = [&](LoopOrderKind loopOrder) -> void {
100  newStencil->getMultiStages().push_back(std::make_shared<MultiStage>(instantiation, loopOrder));
101  newNumMultiStages++;
102  };
103 
104  for(const auto& multiStagePtr : stencil.getMultiStages()) {
105 
106  // First time we encounter this multi-stage, create an empty multi-stage
107  pushBackNewMultiStage(LoopOrderKind::LK_Parallel);
108 
109  for(const auto& stagePtr : multiStagePtr->getStages()) {
110  const Stage& stage = *stagePtr;
111  int stageIdx = newNumStages - 1;
112 
113  // Compute the best possible position to where we can move this stage without violating
114  // any dependencies
115  for(; stageIdx >= 0; --stageIdx) {
116  if(stageDAG.depends(stage.getStageID(), newStencil->getStage(stageIdx)->getStageID()))
117  break;
118  }
119 
120  Stencil::StagePosition pos = newStencil->getPositionFromStageIndex(stageIdx);
121  LoopOrderKind stageLoopOrder = multiStagePtr->getLoopOrder();
122 
123  // Find the first available multi-stage
124  bool lastChance = false;
125  while(true) {
126  const auto& MS = newStencil->getMultiStageFromMultiStageIndex(pos.MultiStageIndex);
127 
128  // 1) Are the loop orders compatible?
129  if(loopOrdersAreCompatible(stageLoopOrder, MS->getLoopOrder())) {
130 
131  // 2) Can we merge the stage wihtout violating vertical dependencies?
132  auto dependencyGraphLoopOrderPair = isMergable(stage, stageLoopOrder, *MS);
133  auto multiStageDependencyGraph = dependencyGraphLoopOrderPair.first;
134 
135  if(multiStageDependencyGraph) {
136 
137  // 3) Do we not exceed the maximum allowed boundary extents?
138  if(!exceedsMaxBoundaryPoints(multiStageDependencyGraph.get(), maxBoundaryExtent)) {
139 
140  // Yes, Yes and Yes ... stop and insert the stage!
141  MS->setLoopOrder(dependencyGraphLoopOrderPair.second);
142  break;
143  } else if(lastChance) {
144  // Our stage exceeds the maximum allowed boundary extents... nothing we can do
145  DiagnosticsBuilder diag(DiagnosticsKind::Error, SourceLocation());
146  diag << "stencil '" << instantiation->getName()
147  << "' exceeds maximum number of allowed halo lines (" << maxBoundaryExtent
148  << ")";
149  instantiation->getOptimizerContext()->getDiagnostics().report(diag);
150  return nullptr;
151  }
152  }
153  DAWN_ASSERT_MSG(!lastChance,
154  "merging stage in empty multi-stage failed (this probably means the "
155  "stage graph contains cycles - i.e is not a DAG!)");
156  }
157 
158  // Advance to the next multi-stage
159  pos.MultiStageIndex++;
160  pos.StageOffset = -1;
161 
162  // The last available multi-stage wasn't legal, we push-back a new multistage with parallel
163  // loop order (this will guarantee a success the next check if our stage does not exceed the
164  // maximum boundary lines in which case we abort)
165  if(pos.MultiStageIndex == newNumMultiStages) {
166  pushBackNewMultiStage(LoopOrderKind::LK_Parallel);
167  lastChance = true;
168  }
169  }
170 
171  newNumStages++;
172  newStencil->insertStage(pos, stagePtr);
173  }
174  }
175 
176  // Remove empty multi-stages
177  for(auto it = newStencil->getMultiStages().begin(); it != newStencil->getMultiStages().end();) {
178  if((*it)->getStages().empty())
179  it = newStencil->getMultiStages().erase(it);
180  else
181  it++;
182  }
183 
184  return newStencil;
185 }
