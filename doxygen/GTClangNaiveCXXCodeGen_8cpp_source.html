64 std::string GTClangNaiveCXXCodeGen::generateStencilInstantiation(
65  const StencilInstantiation* stencilInstantiation) {
66  using namespace codegen;
67 
68  std::stringstream ssSW, tss;
69 
70  Class StencilWrapperClass(stencilInstantiation->getName(), ssSW);
71  StencilWrapperClass.changeAccessibility("private");
72 
73  // Generate stencils
74  auto& stencils = stencilInstantiation->getStencils();
75  for(std::size_t stencilIdx = 0; stencilIdx < stencils.size(); ++stencilIdx) {
76  const Stencil& stencil = *stencilInstantiation->getStencils()[stencilIdx];
77 
78  if(stencil.isEmpty())
79  continue;
80 
81  //
82  // Stencil definition
83  //
84  const auto& StencilFields = stencil.getFields();
85  std::vector<std::string> StencilTemplates;
86  for(int i = 0; i < StencilFields.size(); ++i)
87  StencilTemplates.push_back("StorageType" + std::to_string(i + 1));
88 
89  Structure StencilClass = StencilWrapperClass.addStruct(
90  Twine("stencil_") + Twine(stencilIdx),
91  RangeToString(", ", "", "")(StencilTemplates,
92  [](const std::string& type) { return "class " + type; }));
93  std::string StencilName = StencilClass.getName();
94 
95  //
96  // Members
97  //
98  StencilClass.addMember("gridtools::clang::domain", "dom");
99  for(int i = 0; i < StencilFields.size(); ++i)
100  StencilClass.addMember(StencilTemplates[i] + "&", StencilFields[i].Name);
101 
102  StencilClass.changeAccessibility("public");
103 
104  //
105  // Constructor
106  //
107  auto StencilConstructor = StencilClass.addConstructor(RangeToString(", ", "", "")(
108  StencilTemplates, [](const std::string& str) { return "class " + str; }));
109 
110  // Arguments
111  StencilConstructor.addArg("const gridtools::clang::domain& dom_");
112  for(int i = 0; i < StencilFields.size(); ++i)
113  StencilConstructor.addArg(StencilTemplates[i] + "& " + StencilFields[i].Name + "_");
114 
115  // Initializers
116  StencilConstructor.addInit("dom(dom_)");
117  for(int i = 0; i < StencilFields.size(); ++i)
118  StencilConstructor.addInit(StencilFields[i].Name + "(" + StencilFields[i].Name + "_)");
119 
120  StencilConstructor.commit();
121 
122  //
123  // Do-Method
124  //
125  MemberFunction StencilDoMethod = StencilClass.addMemberFunction("void", "Do");
126 
127  for(const auto& multiStagePtr : stencil.getMultiStages()) {
128  const MultiStage& multiStage = *multiStagePtr;
129  StencilDoMethod.addBlockStatement(
130  makeKLoop(multiStage.getLoopOrder() == LoopOrderKind::LK_Backward), [&]() {
131  for(const auto& stagePtr : multiStage.getStages()) {
132  const Stage& stage = *stagePtr;
133 
134  // Check Interval
135  stage.getIntervals();
136 
137  StencilDoMethod.addBlockStatement(makeIJLoop("i"), [&]() {
138  StencilDoMethod.addBlockStatement(makeIJLoop("j"), [&]() {
139 
140  });
141  });
142  }
143  });
144  }
145  StencilDoMethod.commit();
146  }
147 
148  StencilWrapperClass.commit();
149 
150  // Remove trailing ';' as this is retained by Clang's Rewriter
151  std::string str = ssSW.str();
152  str[str.size() - 2] = ' ';
153 
154  return str;
155 }
