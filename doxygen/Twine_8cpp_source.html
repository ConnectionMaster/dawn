39 StringRef Twine::toNullTerminatedStringRef(SmallVectorImpl<char>& Out) const {
40  if(isUnary()) {
41  switch(getLHSKind()) {
42  case CStringKind:
43  // Already null terminated, yay!
44  return StringRef(lhs_.cString);
45  case StdStringKind: {
46  const std::string* str = lhs_.stdString;
47  return StringRef(str->c_str(), str->size());
48  }
49  default:
50  break;
51  }
52  }
53  toVector(Out);
54  Out.push_back(0);
55  Out.pop_back();
