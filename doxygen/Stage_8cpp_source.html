74 bool Stage::overlaps(const Stage& other) const {
75  // This is a more conservative test.. if it fails we are certain nothing overlaps
76  if(!getEnclosingExtendedInterval().overlaps(other.getEnclosingExtendedInterval()))
77  return false;
78 
79  for(const auto& otherDoMethodPtr : other.getDoMethods())
80  if(overlaps(otherDoMethodPtr->getInterval(), other.getFields()))
81  return true;
82  return false;
83 }

85 bool Stage::overlaps(const Interval& interval, ArrayRef<Field> fields) const {
86  for(const auto& doMethodPtr : DoMethods_) {
87  const Interval& thisInterval = doMethodPtr->getInterval();
88 
89  for(const Field& thisField : getFields()) {
90  for(const Field& field : fields) {
91  if(thisField.AccessID != field.AccessID)
92  continue;
93 
94  if(thisInterval.extendInterval(thisField.Extent[2])
95  .overlaps(interval.extendInterval(field.Extent[2])))
96  return true;
97  }
98  }
99  }
100 
101  return false;
102 }

104 void Stage::update() {
105  fields_.clear();
106  globalVariables_.clear();
107 
108  // Compute the fields and their intended usage. Fields can be in one of three states: `Output`,
109  // `InputOutput` or `Input` which implements the following state machine:
110  //
111  // +-------+ +--------+
112  // | Input | | Output |
113  // +-------+ +--------+
114  // | |
115  // | +-------------+ |
116  // +----------> | InputOutput | <----------+
117  // +-------------+
118  //
119  std::set<int> inputOutputFields;
120  std::set<int> inputFields;
121  std::set<int> outputFields;
122 
123  for(const auto& doMethodPtr : DoMethods_) {
124  const DoMethod& doMethod = *doMethodPtr;
125  for(const auto& statementAccessesPair : doMethod.getStatementAccessesPairs()) {
126  const auto& access = statementAccessesPair->getAccesses();
127 
128  for(const auto& accessPair : access->getWriteAccesses()) {
129  int AccessID = accessPair.first;
130 
131  // Does this AccessID correspond to a field access?
132  if(!stencilInstantiation_->isField(AccessID)) {
133  if(stencilInstantiation_->isGlobalVariable(AccessID))
134  globalVariables_.insert(AccessID);
135  continue;
136  }
137 
138  // Field was recorded as `InputOutput`, state can't change ...
139  if(inputOutputFields.count(AccessID))
140  continue;
141 
142  // Field was recorded as `Input`, change it's state to `InputOutput`
143  if(inputFields.count(AccessID)) {
144  inputOutputFields.insert(AccessID);
145  inputFields.erase(AccessID);
146  continue;
147  }
148 
149  // Field not yet present, record it as output
150  outputFields.insert(AccessID);
151  }
152 
153  for(const auto& accessPair : access->getReadAccesses()) {
154  int AccessID = accessPair.first;
155 
156  // Does this AccessID correspond to a field access?
157  if(!stencilInstantiation_->isField(AccessID)) {
158  if(stencilInstantiation_->isGlobalVariable(AccessID))
159  globalVariables_.insert(AccessID);
160  continue;
161  }
162 
163  // Field was recorded as `InputOutput`, state can't change ...
164  if(inputOutputFields.count(AccessID))
165  continue;
166 
167  // Field was recorded as `Output`, change it's state to `InputOutput`
168  if(outputFields.count(AccessID)) {
169  inputOutputFields.insert(AccessID);
170  outputFields.erase(AccessID);
171  continue;
172  }
173 
174  // Field not yet present, record it as input
175  inputFields.insert(AccessID);
176  }
177  }
178  }
179 
180  // Merge inputFields, outputFields and fields
181  for(int AccessID : outputFields)
182  fields_.emplace_back(AccessID, Field::IK_Output);
183 
184  for(int AccessID : inputOutputFields)
185  fields_.emplace_back(AccessID, Field::IK_InputOutput);
186 
187  for(int AccessID : inputFields)
188  fields_.emplace_back(AccessID, Field::IK_Input);
189 
190  if(fields_.empty()) {
191  DAWN_LOG(WARNING) << "no fields referenced in stage";
192  return;
193  }
194 
195  // Index to speedup lookup into fields map
196  std::unordered_map<int, std::vector<Field>::iterator> AccessIDToFieldMap;
197  for(auto it = fields_.begin(), end = fields_.end(); it != end; ++it)
198  AccessIDToFieldMap.insert(std::make_pair(it->AccessID, it));
199 
200  // Accumulate the extents of each field in this stage
201  for(const auto& doMethodPtr : DoMethods_) {
202  const DoMethod& doMethod = *doMethodPtr;
203 
204  for(const auto& statementAccessesPair : doMethod.getStatementAccessesPairs()) {
205  const auto& access = statementAccessesPair->getAccesses();
206 
207  // first => AccessID, second => Extent
208  for(auto& accessPair : access->getWriteAccesses()) {
209  if(!stencilInstantiation_->isField(accessPair.first))
210  continue;
211 
212  AccessIDToFieldMap[accessPair.first]->Extent.merge(accessPair.second);
213  }
214 
215  for(const auto& accessPair : access->getReadAccesses()) {
216  if(!stencilInstantiation_->isField(accessPair.first))
217  continue;
218 
219  AccessIDToFieldMap[accessPair.first]->Extent.merge(accessPair.second);
220  }
221  }
222  }
223 }

227 void Stage::addDoMethod(std::unique_ptr<DoMethod>& doMethod) {
228  DAWN_ASSERT_MSG(std::find_if(DoMethods_.begin(), DoMethods_.end(),
229  [&](const std::unique_ptr<DoMethod>& doMethodPtr) {
230  return doMethodPtr->getInterval() == doMethod->getInterval();
231  }) == DoMethods_.end(),
232  "Do-Method with given interval already exists!");
233  DoMethods_.emplace_back(std::move(doMethod));
234  update();
235 }

237 void Stage::appendDoMethod(std::unique_ptr<DoMethod>& from, std::unique_ptr<DoMethod>& to,
238  const std::shared_ptr<DependencyGraphAccesses>& dependencyGraph) {
239  DAWN_ASSERT_MSG(std::find(DoMethods_.begin(), DoMethods_.end(), to) != DoMethods_.end(),
240  "'to' DoMethod does not exists");
241  DAWN_ASSERT_MSG(from->getInterval() == to->getInterval(),
242  "DoMethods have incompatible intervals!");
243 
244  to->setDependencyGraph(dependencyGraph);
245  to->getStatementAccessesPairs().insert(
246  to->getStatementAccessesPairs().end(),
247  std::make_move_iterator(from->getStatementAccessesPairs().begin()),
248  std::make_move_iterator(from->getStatementAccessesPairs().end()));
249  update();
250 }

254 std::vector<std::shared_ptr<Stage>>
255 Stage::split(std::deque<int>& splitterIndices,
256  const std::deque<std::shared_ptr<DependencyGraphAccesses>>* graphs) {
257  DAWN_ASSERT_MSG(hasSingleDoMethod(), "Stage::split does not support multiple Do-Methods");
258  DoMethod& thisDoMethod = getSingleDoMethod();
259  auto& thisStatementAccessesPairs = thisDoMethod.getStatementAccessesPairs();
260 
261  DAWN_ASSERT(thisStatementAccessesPairs.size() >= 2);
262  DAWN_ASSERT(!graphs || splitterIndices.size() == graphs->size() - 1);
263 
264  std::vector<std::shared_ptr<Stage>> newStages;
265 
266  splitterIndices.push_back(thisStatementAccessesPairs.size() - 1);
267  std::size_t prevSplitterIndex = 0;
268 
269  // Create new stages
270  for(std::size_t i = 0; i < splitterIndices.size(); ++i) {
271  std::size_t nextSplitterIndex = splitterIndices[i] + 1;
272 
273  newStages.push_back(std::make_shared<Stage>(stencilInstantiation_, multiStage_,
274  stencilInstantiation_->nextUID(),
275  thisDoMethod.getInterval()));
276  Stage& newStage = *newStages.back();
277  DoMethod& doMethod = newStage.getSingleDoMethod();
278 
279  if(graphs)
280  doMethod.setDependencyGraph((*graphs)[i]);
281 
282  // The new stage contains the statements in the range [prevSplitterIndex , nextSplitterIndex)
283  for(std::size_t idx = prevSplitterIndex; idx < nextSplitterIndex; ++idx)
284  doMethod.getStatementAccessesPairs().emplace_back(std::move(thisStatementAccessesPairs[idx]));
285 
286  // Update the fields of the new stage
287  newStage.update();
288 
289  prevSplitterIndex = nextSplitterIndex;
290  }
291 
292  return newStages;
293 }
