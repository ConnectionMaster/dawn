86 void DependencyGraphAccesses::merge(const DependencyGraphAccesses* other) {
87  // Insert the nodes of `other`
88  for(const auto& AccessIDVertexIDPair : other->getVertices()) {
89  int AccessID = AccessIDVertexIDPair.first;
90  insertNode(AccessID);
91  }
92 
93  // Insert the edges of `other`
94  for(std::size_t VertexID = 0; VertexID < other->getAdjacencyList().size(); ++VertexID) {
95  for(const Edge& edge : *(other->getAdjacencyList()[VertexID])) {
96  insertEdge(other->getIDFromVertexID(edge.FromVertexID),
97  other->getIDFromVertexID(edge.ToVertexID), edge.Data);
98  }
99  }

111 std::vector<std::set<std::size_t>> DependencyGraphAccesses::partitionInSubGraphs() const {
112 
113  // Each Vertex is assigned to a partition (-1 means not yet assigned)
114  std::vector<int> partition(adjacencyList_.size(), -1);
115  int numPartitions = 0;
116 
117  std::vector<std::size_t> nodesToVisit;
118 
119  // Perform a BFS
120  for(std::size_t VertexID = 0; VertexID < adjacencyList_.size(); ++VertexID) {
121  int currentPartitionIdx = partition[VertexID];
122  if(currentPartitionIdx != -1)
123  continue;
124 
125  // Create a new partition
126  currentPartitionIdx = numPartitions++;
127 
128  nodesToVisit.push_back(VertexID);
129  while(!nodesToVisit.empty()) {
130 
131  // Process next node
132  std::size_t curNode = nodesToVisit.back();
133  nodesToVisit.pop_back();
134 
135  int nodePartitionIdx = partition[curNode];
136 
137  // If we already visited this node, we have to merge our current partition (given by
138  // `currentPartitionIdx`) with the other partition already visited.
139  if(nodePartitionIdx != -1) {
140  if(nodePartitionIdx == currentPartitionIdx)
141  continue;
142 
143  int otherPartitionIdx = nodePartitionIdx;
144  DAWN_ASSERT(otherPartitionIdx != -1);
145 
146  // Merge the current partition into other
147  std::for_each(partition.begin(), partition.end(), [&](int& partitionIdx) {
148  if(partitionIdx == currentPartitionIdx)
149  partitionIdx = otherPartitionIdx;
150  });
151 
152  // Current partition is now equal to the other partition
153  currentPartitionIdx = otherPartitionIdx;
154  continue;
155  } else {
156  partition[curNode] = currentPartitionIdx;
157  }
158 
159  for(const Edge& edge : *adjacencyList_[curNode])
160  nodesToVisit.push_back(edge.ToVertexID);
161  }
162  }
163 
164  // Assemble the final partitions. We use a map between the index in `finalPartitions` and the
165  // partitionIdx.
166  std::vector<std::set<std::size_t>> finalPartitions;
167  std::unordered_map<int, std::size_t> PartitionIndexToIndexInFinalPartitionsMap;
168 
169  for(std::size_t VertexID = 0; VertexID < adjacencyList_.size(); ++VertexID) {
170  int partitionIdx = partition[VertexID];
171  auto it = PartitionIndexToIndexInFinalPartitionsMap.find(partitionIdx);
172 
173  if(it != PartitionIndexToIndexInFinalPartitionsMap.end())
174  finalPartitions[it->second].insert(VertexID);
175  else {
176  PartitionIndexToIndexInFinalPartitionsMap.emplace(partitionIdx, finalPartitions.size());
177  finalPartitions.push_back(std::set<std::size_t>{});
178  finalPartitions.back().insert(VertexID);
179  }
180  }
181  return finalPartitions;
182 }

188 template <class VertexListType, class GetVertexIDFromVertexListElemenFuncType>
189 void getInputVertexIDsImpl(
190  const DependencyGraphAccesses& graph, const VertexListType& vertexList,
191  GetVertexIDFromVertexListElemenFuncType&& getVertexIDFromVertexListElemenFunc,
192  std::vector<std::size_t>& inputVertexIDs) {
193 
194  const auto& adjacencyList = graph.getAdjacencyList();
195  for(const auto& vertex : vertexList) {
196  std::size_t VertexID = getVertexIDFromVertexListElemenFunc(vertex);
197  if(adjacencyList[VertexID]->empty())
198  inputVertexIDs.push_back(VertexID);
199  else if(adjacencyList[VertexID]->size() == 1) {
200  // We allow self-dependencies!
201  const auto& edge = adjacencyList[VertexID]->front();
202  if(edge.FromVertexID == edge.ToVertexID)
203  inputVertexIDs.push_back(VertexID);
204  }
205  }
206 }

212 template <class VertexListType, class GetVertexIDFromVertexListElemenFuncType>
213 void getOutputVertexIDsImpl(
214  const DependencyGraphAccesses& graph, const VertexListType& vertexList,
215  GetVertexIDFromVertexListElemenFuncType&& getVertexIDFromVertexListElemenFunc,
216  std::vector<std::size_t>& outputVertexIDs) {
217 
218  std::set<std::size_t> dependentNodes;
219  const auto& adjacencyList = graph.getAdjacencyList();
220 
221  // Construct a set of dependent nodes i.e nodes with edges from other nodes pointing to them
222  for(const auto& edgeList : adjacencyList)
223  for(const auto& edge : *edgeList)
224  // We allow self-dependencies!
225  if(edge.FromVertexID != edge.ToVertexID)
226  dependentNodes.insert(edge.ToVertexID);
227 
228  for(const auto& vertex : vertexList) {
229  std::size_t VertexID = getVertexIDFromVertexListElemenFunc(vertex);
230  if(!dependentNodes.count(VertexID))
231  outputVertexIDs.push_back(VertexID);
232  }
233 }

235 bool DependencyGraphAccesses::isDAG() const {
236  auto partitions = partitionInSubGraphs();
237  std::vector<std::size_t> vertices;
238 
239  for(std::set<std::size_t>& partition : partitions) {
240  vertices.empty();
241  getInputVertexIDsImpl(*this, partition, [](std::size_t VertexID) { return VertexID; },
242  vertices);
243  if(vertices.empty())
244  return false;
245 
246  vertices.clear();
247  getOutputVertexIDsImpl(*this, partition, [](std::size_t VertexID) { return VertexID; },
248  vertices);
249  if(vertices.empty())
250  return false;
251  }
252  return true;
253 }

278 class StronglyConnectedComponents {
279  using VertexType = DependencyGraphAccesses::Vertex;
280  using EdgeType = DependencyGraphAccesses::Edge;
281 
283  struct VertexData {
286  int Index;
287 
293  int LowLink;
294 
296  bool OnStack;
297 
299  const VertexType* Vertex;
300  };
301 
302  std::unordered_map<std::size_t, VertexData> vertexData_;
303  std::stack<std::size_t> vertexStack_;
304 
305  int index_;
306  bool hasMultiNodeSCC_;
307 
308  const DependencyGraphAccesses* graph_;
309  std::vector<std::set<int>>* scc_;
310 
311 public:
312  StronglyConnectedComponents(const DependencyGraphAccesses* graph, std::vector<std::set<int>>* scc)
313  : index_(0), hasMultiNodeSCC_(false), graph_(graph), scc_(scc) {}
314 
316  bool find() {
317  if(scc_)
318  scc_->clear();
319 
320  // Initialize the vertex Data
321  for(const auto& AccessIDVertexPair : graph_->getVertices()) {
322  const VertexType& vertex = AccessIDVertexPair.second;
323  vertexData_.emplace(vertex.VertexID, VertexData{-1, -1, false, &vertex});
324  }
325  index_ = 0;
326  hasMultiNodeSCC_ = false;
327 
328  // Iterate all Vertices
329  for(const auto& AccessIDVertexPair : graph_->getVertices()) {
330  const VertexType& vertex = AccessIDVertexPair.second;
331  findImpl(vertex.VertexID);
332  }
333 
334  return hasMultiNodeSCC_;
335  }
336 
337 private:
338  void findImpl(std::size_t FromVertexID) {
339  VertexData& FromVertexData = vertexData_[FromVertexID];
340 
341  if(FromVertexData.Index != -1)
342  return;
343 
344  // Set the depth index for the `FromVertex` to the smallest unused index
345  FromVertexData.Index = index_;
346  FromVertexData.LowLink = index_;
347  FromVertexData.OnStack = true;
348  vertexStack_.push(FromVertexID);
349  index_++;
350 
351  // Consider successors of the `FromVertex`
352  for(const EdgeType& edge : *graph_->getAdjacencyList()[FromVertexID]) {
353 
354  VertexData& ToVertexData = vertexData_[edge.ToVertexID];
355 
356  if(ToVertexData.Index == -1) {
357  // Successor `ToVertex` has not yet been visited; recurse on it
358  findImpl(edge.ToVertexID);
359  FromVertexData.LowLink = std::min(FromVertexData.LowLink, ToVertexData.LowLink);
360  } else if(ToVertexData.OnStack) {
361  // Successor `ToVertex` is in stack and hence in the current SCC
362  FromVertexData.LowLink = std::min(FromVertexData.LowLink, ToVertexData.LowLink);
363  }
364  }
365 
366  // If `FromVertex` is a root node, pop the stack and generate an SCC
367  if(FromVertexData.LowLink == FromVertexData.Index) {
368  std::set<int> SCC;
369  std::size_t VertexID;
370 
371  do {
372  VertexID = vertexStack_.top();
373  vertexStack_.pop();
374 
375  VertexData& vertexData = vertexData_[VertexID];
376  vertexData.OnStack = false;
377  SCC.insert(vertexData.Vertex->ID);
378 
379  } while(VertexID != FromVertexID);
380 
381  if(SCC.size() > 1) {
382  hasMultiNodeSCC_ = true;
383  if(scc_)
384  scc_->emplace_back(std::move(SCC));
385  }
386  }
387  }
388 };
389 
390 } // anonymous namespace

404 class GreedyColoring {
405  const DependencyGraphAccesses* graph_;
406  std::unordered_map<int, int>& coloring_;
407 
408 public:
409  GreedyColoring(const DependencyGraphAccesses* graph, std::unordered_map<int, int>& coloring)
410  : graph_(graph), coloring_(coloring) {}
411 
412  using Vertex = DependencyGraphAccesses::Vertex;
413  using Edge = DependencyGraphAccesses::Edge;
414 
415  void compute() {
416  coloring_.clear();
417 
418  std::size_t numVertices = graph_->getNumVertices();
419  const auto& adjacencyList = graph_->getAdjacencyList();
420 
421  // Compute the neighbor-list
422  std::vector<std::set<std::size_t>> neighborList(numVertices);
423  for(std::size_t FromVertexID = 0; FromVertexID < numVertices; ++FromVertexID) {
424  for(const Edge& edge : *adjacencyList[FromVertexID]) {
425  neighborList[edge.FromVertexID].insert(edge.ToVertexID);
426  neighborList[edge.ToVertexID].insert(edge.FromVertexID);
427  }
428  }
429 
430  // True value of `available[VertexID]` means that the color `VertexID` is assigned to one of
431  // its adjacent vertices
432  std::vector<bool> assigned(numVertices, false);
433 
434  // Colors of the VertexID
435  std::vector<int> colors(numVertices, -1);
436  colors[0] = 0;
437 
438  // Assign colors to the remaining vertices
439  for(std::size_t FromVertexID = 1; FromVertexID < numVertices; ++FromVertexID) {
440 
441  auto setAssignmentOfNeigbors = [&](bool isAssigned) {
442  for(std::size_t ToVertexID : neighborList[FromVertexID])
443  if(colors[ToVertexID] != -1)
444  assigned[colors[ToVertexID]] = isAssigned;
445  };
446 
447  // Process all neighbor vertices and flag their colors as assigned
448  setAssignmentOfNeigbors(true);
449 
450  // Find the first available color
451  auto it = std::find(assigned.begin(), assigned.end(), false);
452  colors[FromVertexID] =
453  it == assigned.end() ? assigned.size() - 1 : std::distance(assigned.begin(), it);
454 
455  // Reset the values back to false for the next iteration
456  setAssignmentOfNeigbors(false);
457  }
458 
459  for(std::size_t VertexID = 0; VertexID < numVertices; ++VertexID)
460  coloring_.emplace(graph_->getIDFromVertexID(VertexID), colors[VertexID]);
461  }
462 };
463 
464 } // anonymous namespace

475 void DependencyGraphAccesses::toJSON(const std::string& file) const {
476  std::unordered_map<std::size_t, Extents> extentMap = *computeBoundaryExtents(this);
477  json::json jgraph;
478 
479  auto extentsToVec = [&](const Extents& extents) {
480  std::vector<int> extentsVec;
481  for(const Extent& extent : extents.getExtents()) {
482  extentsVec.push_back(extent.Minus);
483  extentsVec.push_back(extent.Plus);
484  }
485  return extentsVec;
486  };
487 
488  std::size_t EdgeID = 0;
489  for(std::size_t VertexID = 0; VertexID < getNumVertices(); ++VertexID) {
490  json::json jvertex;
491 
492  jvertex["name"] = getVertexNameByVertexID(VertexID);
493  jvertex["extent"] = extentsToVec(extentMap[VertexID]);
494 
495  int AccessID = getIDFromVertexID(VertexID);
496  if(instantiation_->isTemporaryField(AccessID))
497  jvertex["type"] = "field_temporary";
498  else if(instantiation_->isField(AccessID))
499  jvertex["type"] = "field";
500  else if(instantiation_->isVariable(AccessID) || instantiation_->isGlobalVariable(AccessID))
501  jvertex["type"] = "variable";
502  else if(instantiation_->isLiteral(AccessID))
503  jvertex["type"] = "literal";
504  else
505  dawn_unreachable("invalid vertex type");
506 
507  jgraph["vertices"][std::to_string(VertexID)] = jvertex;
508 
509  for(const Edge& edge : *getAdjacencyList()[VertexID]) {
510  json::json jedge;
511 
512  jedge["from"] = edge.FromVertexID;
513  jedge["to"] = edge.ToVertexID;
514  jedge["extent"] = extentsToVec(edge.Data);
515  jgraph["edges"][std::to_string(EdgeID)] = jedge;
516 
517  EdgeID++;
518  }
519  }
520 
521  jgraph["num_vertices"] = getNumVertices();
522  jgraph["num_edges"] = EdgeID;
523 
524  std::ofstream ofs;
525  ofs.open(file);
526  if(!ofs.is_open()) {
527  DiagnosticsBuilder diag(DiagnosticsKind::Error);
528  diag << "failed to open file: \"" << file << "\"";
529  instantiation_->getOptimizerContext()->getDiagnostics().report(diag);
530  }
531 
532  ofs << jgraph.dump(2);
533  ofs.close();
534 }
