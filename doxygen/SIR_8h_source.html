40 class Attr {
41  unsigned attrBits_;
42 
43 public:
44  Attr() : attrBits_(0) {}
45 
47  enum AttrKind : unsigned {
48  AK_NoCodeGen = 1 << 0,
49  AK_MergeStages = 1 << 1,
50  AK_MergeDoMethods = 1 << 2,
51  AK_MergeTemporaries = 1 << 3,
52  AK_UseKCaches = 1 << 4
53  };
54 
56  bool has(AttrKind attr) const { return (attrBits_ >> attr) & 1; }
57 
60  bool hasOneOf(AttrKind attr1, AttrKind attr2) const { return has(attr1) || has(attr2); }
61 
62  template <typename... AttrTypes>
63  bool hasOneOf(AttrKind attr1, AttrKind attr2, AttrTypes... attrs) const {
64  return has(attr1) || hasOneOf(attr2, attrs...);
65  }
67 
69  unsigned getBits() const { return attrBits_; }
71  void set(AttrKind attr) { attrBits_ |= 1 << attr; }
72 
74  void unset(AttrKind attr) { attrBits_ &= ~(1 << attr); }
75 
77  void clear() { attrBits_ = 0; }
78 
79  bool operator==(const Attr& rhs) const { return getBits() == rhs.getBits(); }
80  bool operator!=(const Attr& rhs) const { return getBits() != rhs.getBits(); }
81 };

96 struct Interval {
97  enum LevelKind : int { Start = 0, End = (1 << 20) };
98 
99  Interval(int lowerLevel, int upperLevel, int lowerOffset = 0, int upperOffset = 0)
100  : LowerLevel(lowerLevel), UpperLevel(upperLevel), LowerOffset(lowerOffset),
101  UpperOffset(upperOffset) {
102  DAWN_ASSERT(lowerLevel >= LevelKind::Start && upperLevel <= LevelKind::End);
103  DAWN_ASSERT((lowerLevel + lowerOffset) <= (upperLevel + upperOffset));
104  }
105 
106  int LowerLevel;
107  int UpperLevel;
108  int LowerOffset;
109  int UpperOffset;
110 
113  bool operator==(const Interval& other) const {
114  return LowerLevel == other.LowerLevel && UpperLevel == other.UpperLevel &&
115  LowerOffset == other.LowerOffset && UpperOffset == other.UpperOffset;
116  }
117  bool operator!=(const Interval& other) const { return !(*this == other); }
118 
119  std::pair<std::string, bool> comparison(const Interval& rhs) const;
121 
124  std::string toString() const;
125  friend std::ostream& operator<<(std::ostream& os, const Interval& interval);
127 };
