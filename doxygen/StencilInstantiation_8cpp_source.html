258  void visit(const std::shared_ptr<StencilFunCallExpr>& expr) override {
259  // Find the referenced stencil function
260  StencilFunctionInstantiation* stencilFun = nullptr;
261  const Interval& interval = scope_.top()->VerticalInterval;
262 
263  for(auto& SIRStencilFun : instantiation_->getSIR()->StencilFunctions) {
264  if(SIRStencilFun->Name == expr->getCallee()) {
265 
266  std::shared_ptr<AST> ast = nullptr;
267  if(SIRStencilFun->isSpecialized()) {
268  // Select the correct overload
269  ast = SIRStencilFun->getASTOfInterval(interval.asSIRInterval());
270  if(ast == nullptr) {
271  DiagnosticsBuilder diag(DiagnosticsKind::Error, expr->getSourceLocation());
272  diag << "no viable Do-Method overload for stencil function call '" << expr->getCallee()
273  << "'";
274  instantiation_->getOptimizerContext()->getDiagnostics().report(diag);
275  return;
276  }
277  } else {
278  ast = SIRStencilFun->Asts.front();
279  }
280 
281  // Clone the AST s.t each stencil function has their own AST which is modifiable
282  ast = ast->clone();
283 
284  stencilFun = instantiation_->makeStencilFunctionInstantiation(
285  expr, SIRStencilFun.get(), ast, interval, scope_.top()->FunctionInstantiation);
286  break;
287  }
288  }
289  DAWN_ASSERT(stencilFun);
290 
291  // If this is a nested function call (e.g the `bar` in `foo(bar(i+1, u))`) register the new
292  // stencil function in the current stencil function
293  if(Scope* candiateScope = getCurrentCandidateScope()) {
294  candiateScope->FunctionInstantiation->setFunctionInstantiationOfArgField(
295  candiateScope->ArgumentIndex, stencilFun);
296  candiateScope->ArgumentIndex += 1;
297  }
298 
299  // Create the scope of the stencil function
300  scope_.top()->CandiateScopes.push(std::make_shared<Scope>(
301  expr->getCallee(), stencilFun->getStatementAccessesPairs(), stencilFun->getInterval()));
302 
303  getCurrentCandidateScope()->FunctionInstantiation = stencilFun;
304 
305  // Resolve the arguments
306  for(auto& arg : expr->getArguments())
307  arg->accept(*this);
308 
309  // Assign the AccessIDs of the fields in the stencil function
310  Scope* candiateScope = getCurrentCandidateScope();
311  const auto& arguments = candiateScope->FunctionInstantiation->getArguments();
312  for(std::size_t argIdx = 0; argIdx < arguments.size(); ++argIdx)
313  if(sir::Field* field = dyn_cast<sir::Field>(arguments[argIdx].get())) {
314  if(candiateScope->FunctionInstantiation->isArgStencilFunctionInstantiation(argIdx)) {
315 
316  // The field is provided by a stencil function call, we create a new AccessID for this
317  // "temporary" field
318  int AccessID = instantiation_->nextUID();
319  candiateScope->FunctionInstantiation->setIsProvidedByStencilFunctionCall(AccessID);
320  candiateScope->FunctionInstantiation->setCallerAccessIDOfArgField(argIdx, AccessID);
321  candiateScope->FunctionInstantiation->setCallerInitialOffsetFromAccessID(
322  AccessID, Array3i{{0, 0, 0}});
323  candiateScope->LocalFieldnameToAccessIDMap.emplace(field->Name, AccessID);
324 
325  } else {
326  int AccessID = candiateScope->FunctionInstantiation->getCallerAccessIDOfArgField(argIdx);
327  candiateScope->LocalFieldnameToAccessIDMap.emplace(field->Name, AccessID);
328  }
329  }
330 
331  // Resolve the function
332  scope_.push(scope_.top()->CandiateScopes.top());
333  scope_.top()->FunctionInstantiation->getAST()->accept(*this);
334  scope_.pop();
335 
336  // We resolved the candiate function, move on ...
337  scope_.top()->CandiateScopes.pop();
338  }

340  virtual void visit(const std::shared_ptr<StencilFunArgExpr>& expr) override {
341  DAWN_ASSERT(!scope_.top()->CandiateScopes.empty());
342 
343  auto& function = scope_.top()->FunctionInstantiation;
344  auto* stencilFun = getCurrentCandidateScope()->FunctionInstantiation;
345  auto& argumentIndex = getCurrentCandidateScope()->ArgumentIndex;
346  bool needsLazyEval = expr->getArgumentIndex() != -1;
347 
348  if(stencilFun->isArgOffset(argumentIndex)) {
349  // Argument is an offset
350  stencilFun->setCallerOffsetOfArgOffset(
351  argumentIndex, needsLazyEval
352  ? function->getCallerOffsetOfArgOffset(expr->getArgumentIndex())
353  : Array2i{{expr->getDimension(), expr->getOffset()}});
354  } else {
355  // Argument is a direction
356  stencilFun->setCallerDimensionOfArgDirection(
357  argumentIndex, needsLazyEval
358  ? function->getCallerDimensionOfArgDirection(expr->getArgumentIndex())
359  : expr->getDimension());
360  }
361 
362  argumentIndex += 1;
363  }

365  void visit(const std::shared_ptr<VarAccessExpr>& expr) override {
366  auto& function = scope_.top()->FunctionInstantiation;
367  const auto& varname = expr->getName();
368 
369  if(expr->isExternal()) {
370  DAWN_ASSERT(!function);
371  DAWN_ASSERT_MSG(!expr->isArrayAccess(), "global array access is not supported");
372 
373  const auto& value = instantiation_->getGlobalVariableValue(varname);
374  if(value.isConstexpr()) {
375  // Replace the variable access with the actual value
376  DAWN_ASSERT_MSG(!value.empty(), "constant global variable with no value");
377 
378  auto newExpr = std::make_shared<dawn::LiteralAccessExpr>(
379  value.toString(), sir::Value::typeToBuiltinTypeID(value.getType()));
380  replaceOldExprWithNewExprInStmt(
381  scope_.top()->StatementAccessesPairs.back()->getStatement()->ASTStmt, expr, newExpr);
382 
383  int AccessID = instantiation_->nextUID();
384  instantiation_->getLiteralAccessIDToNameMap().emplace(AccessID, newExpr->getValue());
385  instantiation_->getExprToAccessIDMap().emplace(newExpr, AccessID);
386 
387  } else {
388  int AccessID = 0;
389  if(!instantiation_->isGlobalVariable(varname)) {
390  AccessID = instantiation_->nextUID();
391  instantiation_->setAccessIDNamePairOfGlobalVariable(AccessID, varname);
392  } else {
393  AccessID = instantiation_->getAccessIDFromName(varname);
394  }
395  instantiation_->getExprToAccessIDMap().emplace(expr, AccessID);
396  }
397 
398  } else {
399  // Register the mapping between VarAccessExpr and AccessID.
400  if(function)
401  function->getExprToCallerAccessIDMap().emplace(
402  expr, scope_.top()->LocalVarNameToAccessIDMap[varname]);
403  else
404  instantiation_->getExprToAccessIDMap().emplace(
405  expr, scope_.top()->LocalVarNameToAccessIDMap[varname]);
406 
407  // Resolve the index if this is an array access
408  if(expr->isArrayAccess())
409  expr->getIndex()->accept(*this);
410  }
411  }
412 
413  void visit(const std::shared_ptr<LiteralAccessExpr>& expr) override {
414  // Register a literal access (Note: the negative AccessID we assign!)
415  int AccessID = -instantiation_->nextUID();
416 
417  auto& function = scope_.top()->FunctionInstantiation;
418  if(function) {
419  function->getLiteralAccessIDToNameMap().emplace(AccessID, expr->getValue());
420  function->getExprToCallerAccessIDMap().emplace(expr, AccessID);
421  } else {
422  instantiation_->getLiteralAccessIDToNameMap().emplace(AccessID, expr->getValue());
423  instantiation_->getExprToAccessIDMap().emplace(expr, AccessID);
424  }
425  }
426 
427  void visit(const std::shared_ptr<FieldAccessExpr>& expr) override {
428  // Register the mapping between FieldAccessExpr and AccessID
429  int AccessID = scope_.top()->LocalFieldnameToAccessIDMap[expr->getName()];
430 
431  auto& function = scope_.top()->FunctionInstantiation;
432  if(function) {
433  function->getExprToCallerAccessIDMap().emplace(expr, AccessID);
434  } else {
435  instantiation_->getExprToAccessIDMap().emplace(expr, AccessID);
436  }
437 
438  if(Scope* candiateScope = getCurrentCandidateScope()) {
439  // We are currently traversing an argument list of a stencil function (create the mapping of
440  // the arguments and compute the initial offset of the field)
441  candiateScope->FunctionInstantiation->setCallerAccessIDOfArgField(
442  candiateScope->ArgumentIndex, AccessID);
443  candiateScope->FunctionInstantiation->setCallerInitialOffsetFromAccessID(
444  AccessID, function ? function->evalOffsetOfFieldAccessExpr(expr) : expr->getOffset());
445  candiateScope->ArgumentIndex += 1;
446  }
447  }

652  if(evalExprAsBoolean(stmt->getCondExpr(), result, scope_.top()->VariableMap)) {

655  // The condition is known at compile time, we can remove this If statement completely by

662  } else if(stmt->hasElse()) {
663  BlockStmt* elseBody = dyn_cast<BlockStmt>(stmt->getElseStmt().get());
664  DAWN_ASSERT_MSG(elseBody, "else-body of if-statment should be a BlockStmt!");
665  for(auto& s : elseBody->getStatements())
666  s->accept(*this);
667  }
668  } else {
669  // We are inside a nested statement and we need to remove this if-statment and replace it
670  // with either the then-block or the else-block or in case we evaluted to `false` and there
671  // is no else-block we insert a `0` void statement.
672 
673  if(result) {
674  // Replace the if-statement with the then-block
675  replaceOldStmtWithNewStmtInStmt(scope_.top()->Statements.back()->ASTStmt, stmt,
676  stmt->getThenStmt());
677  stmt->getThenStmt()->accept(*this);
678  } else if(stmt->hasElse()) {
679  // Replace the if-statement with the else-block
680  replaceOldStmtWithNewStmtInStmt(scope_.top()->Statements.back()->ASTStmt, stmt,
681  stmt->getElseStmt());
682  stmt->getElseStmt()->accept(*this);
683  } else {
684  // Replace the if-statement with a void `0`
685  auto voidExpr = std::make_shared<LiteralAccessExpr>("0", BuiltinTypeID::Float);
686  auto voidStmt = std::make_shared<ExprStmt>(voidExpr);
687  int AccessID = -instantiation_->nextUID();
688  instantiation_->getLiteralAccessIDToNameMap().emplace(AccessID, "0");
689  instantiation_->getExprToAccessIDMap().emplace(voidExpr, AccessID);
690  replaceOldStmtWithNewStmtInStmt(scope_.top()->Statements.back()->ASTStmt, stmt, voidStmt);
691  }
692  }
693 
694  } else {
695  if(scope_.top()->ScopeDepth == 1)
696  pushBackStatement(stmt);
697 
698  stmt->getCondExpr()->accept(*this);
699 
700  // The then-part needs to go into a separate stencil ...
701  makeNewStencil();
702  stmt->getThenStmt()->accept(*this);
703 
704  if(stmt->hasElse()) {
705  // ... same for the else-part
706  makeNewStencil();
707  stmt->getElseStmt()->accept(*this);
708  }
709 
710  // Everything that follows needs to go into a new stencil as well
711  makeNewStencil();
712  }
713  }

800  sir::StencilCall* stencilCall = stmt->getStencilCall().get();

808  std::shared_ptr<Scope>& curScope = scope_.top();

810  std::make_shared<Scope>(curScope->Name, curScope->Statements);
811 
812  // Variables are inherited from the parent scope (note that this *needs* to be a copy as we
813  // cannot modify the parent scope)
814  candiateScope->VariableMap = curScope->VariableMap;
815 
816  // Record the call
817  if(!curScope->StackTrace)
818  candiateScope->StackTrace = std::make_shared<std::vector<sir::StencilCall*>>();

823

825  auto stencilIt = std::find_if(
826  instantiation_->getSIR()->Stencils.begin(), instantiation_->getSIR()->Stencils.end(),
827  [&](const std::shared_ptr<sir::Stencil>& s) { return s->Name == stencilCall->Callee; });
828  DAWN_ASSERT(stencilIt != instantiation_->getSIR()->Stencils.end());

830 
831  // We need less or an equal amount of args as temporaries are added implicitly
832  DAWN_ASSERT(stencilCall->Args.size() <= stencil.Fields.size());

848  curScope->LocalFieldnameToAccessIDMap.find(stencilCall->Args[stencilCallArgIdx]->Name)
849  ->second;
850  stencilCallArgIdx++;
851  }
852 
853  candiateScope->LocalFieldnameToAccessIDMap.emplace(stencil.Fields[stencilArgIdx]->Name,
854  AccessID);
855  }

858  scope_.push(candiateScope);
859 
860  // As we *may* modify the AST we better make a copy here otherwise we get funny surprises if we
861  // call this stencil multiple times ...
862  stencil.StencilDescAst->clone()->accept(*this);
863 
864  scope_.pop();
865 
866  DAWN_LOG(INFO) << "Done processing stencil call to `" << stencilCall->Callee << "`";
867  }

1110 int StencilInstantiation::createVersionAndRename(int AccessID, Stencil* stencil, int curStageIdx,
1111  int curStmtIdx, std::shared_ptr<Expr>& expr,
1112  RenameDirection dir) {
1113  int newAccessID = nextUID();
1114 
1115  if(isField(AccessID)) {
1116  auto it = FieldVersionsMap_.find(AccessID);
1117  if(it != FieldVersionsMap_.end()) {
1118  // Field is already multi-versioned, append a new version
1119  std::vector<int>& versions = *it->second;
1120 
1121  // Set the second to last field to be a temporary (only the first and the last field will be
1122  // real storages, all other versions will be temporaries)
1123  int lastAccessID = versions.back();
1124  TemporaryFieldAccessIDSet_.insert(lastAccessID);
1125  AllocatedFieldAccessIDSet_.erase(lastAccessID);
1126 
1127  // The field with version 0 contains the original name
1128  const std::string& originalName = getNameFromAccessID(versions.front());
1129 
1130  // Register the new field
1131  setAccessIDNamePairOfField(newAccessID, originalName + "_" + std::to_string(versions.size()),
1132  false);
1133  AllocatedFieldAccessIDSet_.insert(newAccessID);
1134 
1135  versions.push_back(newAccessID);
1136  FieldVersionsMap_.emplace(newAccessID, it->second);
1137 
1138  } else {
1139  const std::string& originalName = getNameFromAccessID(AccessID);
1140 
1141  // Register the new *and* old field as being multi-versioned and indicate code-gen it has to
1142  // allocate the second version
1143  auto versionsVecPtr = std::make_shared<std::vector<int>>();
1144  *versionsVecPtr = {AccessID, newAccessID};
1145 
1146  setAccessIDNamePairOfField(newAccessID, originalName + "_1", false);
1147  AllocatedFieldAccessIDSet_.insert(newAccessID);
1148 
1149  FieldVersionsMap_.emplace(AccessID, versionsVecPtr);
1150  FieldVersionsMap_.emplace(newAccessID, versionsVecPtr);
1151  }
1152  } else {
1153  auto it = VariableVersionsMap_.find(AccessID);
1154  if(it != VariableVersionsMap_.end()) {
1155  // Variable is already multi-versioned, append a new version
1156  std::vector<int>& versions = *it->second;
1157 
1158  // The variable with version 0 contains the original name
1159  const std::string& originalName = getNameFromAccessID(versions.front());
1160 
1161  // Register the new variable
1162  setAccessIDNamePair(newAccessID, originalName + "_" + std::to_string(versions.size()));
1163  versions.push_back(newAccessID);
1164  VariableVersionsMap_.emplace(newAccessID, it->second);
1165 
1166  } else {
1167  const std::string& originalName = getNameFromAccessID(AccessID);
1168 
1169  // Register the new *and* old variable as being multi-versioned
1170  auto versionsVecPtr = std::make_shared<std::vector<int>>();
1171  *versionsVecPtr = {AccessID, newAccessID};
1172 
1173  setAccessIDNamePair(newAccessID, originalName + "_1");
1174  VariableVersionsMap_.emplace(AccessID, versionsVecPtr);
1175  VariableVersionsMap_.emplace(newAccessID, versionsVecPtr);
1176  }
1177  }
1178 
1179  // Rename the Expression
1180  renameAccessIDInExpr(this, AccessID, newAccessID, expr);
1181 
1182  // Recompute the accesses of the current statement (only works with single Do-Methods - for now)
1183  computeAccesses(
1184  this,
1185  stencil->getStage(curStageIdx)->getSingleDoMethod().getStatementAccessesPairs()[curStmtIdx]);
1186 
1187  // Rename the statement and accesses
1188  for(int stageIdx = curStageIdx;
1189  dir == RD_Above ? (stageIdx >= 0) : (stageIdx < stencil->getNumStages());
1190  dir == RD_Above ? stageIdx-- : stageIdx++) {
1191  Stage& stage = *stencil->getStage(stageIdx);
1192  DoMethod& doMethod = stage.getSingleDoMethod();
1193 
1194  if(stageIdx == curStageIdx) {
1195  for(int i = dir == RD_Above ? (curStmtIdx - 1) : (curStmtIdx + 1);
1196  dir == RD_Above ? (i >= 0) : (i < doMethod.getStatementAccessesPairs().size());
1197  dir == RD_Above ? (--i) : (++i)) {
1198  renameAccessIDInStmts(this, AccessID, newAccessID, doMethod.getStatementAccessesPairs()[i]);
1199  renameAccessIDInAccesses(this, AccessID, newAccessID,
1200  doMethod.getStatementAccessesPairs()[i]);
1201  }
1202 
1203  } else {
1204  renameAccessIDInStmts(this, AccessID, newAccessID, doMethod.getStatementAccessesPairs());
1205  renameAccessIDInAccesses(this, AccessID, newAccessID, doMethod.getStatementAccessesPairs());
1206  }
1207 
1208  // Updat the fields of the stage
1209  stage.update();
1210  }
1211 
1212  return newAccessID;
1213 }
