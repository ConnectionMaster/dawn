68 reportAccessesImpl(FieldAccessIDToStringFunctionType&& fieldAccessIDToStringFunction,
69  IsFieldFunctionType&& isField,
70  LiteralAccessIDToStringFunctionFunctionType&& literalAccessIDToStringFunction,
71  VarAccessIDToStringFunctionFunctionType&& varAccessIDToStringFunction,
72  const MapType& writeAccessMap, const MapType& readAccessMap) {
73  std::stringstream ss;
74 
75  auto printMap = [&](const MapType& map, const char* intent) {
76  for(auto it = map.begin(), end = map.end(); it != end; ++it) {
77  ss << (it != map.begin() ? " " : "") << intent << ":";
78  int AccessID = it->first;
79  if(AccessID < 0)
80  ss << literalAccessIDToStringFunction(AccessID);
81  else {
82  if(isField(AccessID))
83  ss << fieldAccessIDToStringFunction(AccessID);
84  else
85  ss << varAccessIDToStringFunction(AccessID);
86  }
87  ss << ":<";
88  const auto& extents = it->second.getExtents();
89  for(std::size_t i = 0; i < extents.size(); ++i)
90  ss << extents[i].Minus << "," << extents[i].Plus << (i != extents.size() - 1 ? "," : ">");
91  }
92  };
93 
94  printMap(writeAccessMap, "W");
95  ss << " ";
96  printMap(readAccessMap, "R");
97  return ss.str();
98 }
