215  void mergeExtentWithAllFields(const Extents& extent,
216  StencilFunctionInstantiation* curStencilFunCall,
217  std::set<int>& appliedAccessIDs) {
218  for(const Field& field : curStencilFunCall->getCallerFields()) {
219  int AccessID = field.AccessID;
220 
221  if(appliedAccessIDs.count(AccessID))
222  continue;
223 
224  if(curStencilFunCall->isProvidedByStencilFunctionCall(AccessID)) {
225  // The field is provided by a stencil function, forward to the callee
226  mergeExtentWithAllFields(
227  extent, curStencilFunCall->getFunctionInstantiationOfArgField(
228  curStencilFunCall->getArgumentIndexFromCallerAccessID(AccessID)),
229  appliedAccessIDs);
230 
231  } else {
232  appliedAccessIDs.insert(AccessID);
233 
234  if(field.Intend == Field::IK_Input || field.Intend == Field::IK_InputOutput)
235  for(auto& callerAccesses : callerAccessesList_)
236  callerAccesses->addReadExtent(AccessID, extent);
237 
238  if(field.Intend == Field::IK_Output || field.Intend == Field::IK_InputOutput)
239  for(auto& callerAccesses : callerAccessesList_)
240  callerAccesses->addWriteExtent(AccessID, extent);
241  }
242  }

245  virtual void visit(const std::shared_ptr<BlockStmt>& stmt) override {
246  // If we are inside the else block of an if-statement we need to continue iterating
247  // the children as the if/then/else block of the if-statement has been collapsed into one single
248  // vector of children
249  if(!curStatementAccessPairStack_.back()->IfCondExpr)
250  curStatementAccessPairStack_.back()->ChildIndex = 0;
251

253  curStatementAccessPairStack_.push_back(make_unique<CurrentStatementAccessPair>(
254  curStatementAccessPairStack_.back()
255  ->Pair->getChildren()[curStatementAccessPairStack_.back()->ChildIndex]));
256 
257  // Process the statement

262  }

320  virtual void visit(const std::shared_ptr<StencilFunCallExpr>& expr) override {
321  StencilFunctionCallScope* previousStencilFunCallScope = nullptr;
322  if(!stencilFunCalls_.empty())
323  previousStencilFunCallScope = stencilFunCalls_.top().get();
324 
325  // Compute the accesses of the stencil function
326  stencilFunCalls_.push(
327  make_unique<StencilFunctionCallScope>(getStencilFunctionInstantiation(expr)));
328 
329  StencilFunctionInstantiation* curStencilFunCall = stencilFunCalls_.top()->FunctionInstantiation;
330  computeAccesses(curStencilFunCall, curStencilFunCall->getStatementAccessesPairs());
331 
332  // Compute the fields to get the IOPolicy of the arguments
333  curStencilFunCall->update();
334 
335  // Traverse the Arguments
336  for(const auto& arg : expr->getArguments())
337  arg->accept(*this);
338 
339  // If the current stencil function is called within the argument list of another stencil
340  // function, we need to merge the extents
341  if(previousStencilFunCallScope) {
342  auto& prevArgumentIndex = previousStencilFunCallScope->ArgumentIndex;
343  auto& prevStencilFunCall = previousStencilFunCallScope->FunctionInstantiation;
344 
345  const Field& field = prevStencilFunCall->getCallerFieldFromArgumentIndex(prevArgumentIndex);
346  DAWN_ASSERT(prevStencilFunCall->isProvidedByStencilFunctionCall(field.AccessID));
347 
348  // Add the extent of the field mapping to the stencil function to all fields
349  std::set<int> appliedAccessIDs;
350  mergeExtentWithAllFields(field.Extent, curStencilFunCall, appliedAccessIDs);
351 
352  prevArgumentIndex += 1;
353  }
354 
355  // Done with the current stencil function call
356  stencilFunCalls_.pop();
357  }

363  void visit(const std::shared_ptr<AssignmentExpr>& expr) override {
364  // LHS is a write, we resolve this manually as we only care about FieldAccessExpr and
365  // VarAccessExpr. However, if we have an expression `a += 5` we need to register the access as
366  // write and read!
367  bool readAndWrite = StringRef(expr->getOp()) == "+=" || StringRef(expr->getOp()) == "-=" ||
368  StringRef(expr->getOp()) == "/=" || StringRef(expr->getOp()) == "*=" ||
369  StringRef(expr->getOp()) == "|=" || StringRef(expr->getOp()) == "&=";
370 
371  if(isa<FieldAccessExpr>(expr->getLeft().get())) {
372  auto field = std::static_pointer_cast<FieldAccessExpr>(expr->getLeft());
373  mergeWriteOffset(field);
374  if(readAndWrite)
375  mergeReadOffset(field);
376  } else if(isa<VarAccessExpr>(expr->getLeft().get())) {
377  auto var = std::static_pointer_cast<VarAccessExpr>(expr->getLeft());
378  mergeWriteOffset(var);
379  if(readAndWrite)
380  mergeReadOffset(var);
381  }
382 
383  // RHS are read accesses
384  expr->getRight()->accept(*this);
385  }

417  void visit(const std::shared_ptr<FieldAccessExpr>& expr) override {
418  if(!stencilFunCalls_.empty()) {
419 
420  StencilFunctionInstantiation* functionInstantiation =
421  stencilFunCalls_.top()->FunctionInstantiation;
422  int& ArgumentIndex = stencilFunCalls_.top()->ArgumentIndex;
423 
424  // Get the extent of the field corresponding to the argument index
425  const Field& field = functionInstantiation->getCallerFieldFromArgumentIndex(ArgumentIndex);
426 
427  if(field.Intend == Field::IK_Input || field.Intend == Field::IK_InputOutput)
428  mergeReadExtent(expr, field.Extent);
429 
430  if(field.Intend == Field::IK_Output || field.Intend == Field::IK_InputOutput)
431  mergeWriteExtent(expr, field.Extent);
432 
433  ArgumentIndex += 1;
434 
435  } else {
436  // This is always a read access (writes are resolved in the the assignment)
437  mergeReadOffset(expr);
438  }
439  }
