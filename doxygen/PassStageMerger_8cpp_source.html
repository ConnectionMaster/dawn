29 bool PassStageMerger::run(StencilInstantiation* stencilInstantiation) {
30  OptimizerContext* context = stencilInstantiation->getOptimizerContext();
31 
32  // Do we need to run this Pass?
33  bool stencilNeedsMergePass = false;
34  for(const auto& stencilPtr : stencilInstantiation->getStencils())
35  stencilNeedsMergePass |= stencilPtr->getSIRStencil()->Attributes.hasOneOf(
36  sir::Attr::AK_MergeStages, sir::Attr::AK_MergeDoMethods);
37 
38  bool MergeStages = context->getOptions().MergeStages;
39  bool MergeDoMethods = context->getOptions().MergeDoMethods;
40 
41  // ... Nope
42  if(!MergeStages && !MergeDoMethods && !stencilNeedsMergePass)
43  return true;
44 
45  std::string filenameWE = getFilenameWithoutExtension(context->getSIR()->Filename);
46  if(context->getOptions().ReportPassStageMerger)
47  stencilInstantiation->dumpAsJson(filenameWE + "_before.json", getName());
48 
49  for(const auto& stencilPtr : stencilInstantiation->getStencils()) {
50  Stencil& stencil = *stencilPtr;
51 
52  // Do we need to run the analysis for thist stencil?
53  const std::shared_ptr<DependencyGraphStage>& stageDAG = stencil.getStageDependencyGraph();
54  bool MergeDoMethodsOfStencil =
55  stencil.getSIRStencil()->Attributes.has(sir::Attr::AK_MergeDoMethods) || MergeDoMethods;
56  bool MergeStagesOfStencil =
57  stencil.getSIRStencil()->Attributes.has(sir::Attr::AK_MergeStages) || MergeStages;
58 
59  // Nope
60  if(!MergeStagesOfStencil && !MergeDoMethodsOfStencil)
61  continue;
62 
63  // Note that the underyling assumption is that stages in the same multi-stage are guaranteed to
64  // have no counter loop-oorder vertical dependencies. We can thus treat each multi-stage in
65  // isolation!
66  for(const auto& multiStagePtr : stencil.getMultiStages()) {
67  MultiStage& multiStage = *multiStagePtr;
68 
69  // Iterate stages backwards (bottom -> top)
70  for(auto curStageIt = multiStage.getStages().rbegin();
71  curStageIt != multiStage.getStages().rend();) {
72 
73  Stage& curStage = **curStageIt;
74  std::vector<std::unique_ptr<DoMethod>>& curDoMethods = curStage.getDoMethods();
75 
76  bool updateFields = false;
77 
78  // If our Do-Methods already spans the entire axis, we don't want to destroy that property
79  bool MergeDoMethodsOfStage = curStage.getEnclosingInterval() != stencil.getAxis(false);
80  if(!MergeDoMethodsOfStage && !MergeDoMethodsOfStencil) {
81  continue;
82  }
83 
84  // Try to merge each Do-Method of our `curStage`
85  for(auto curDoMethodIt = curDoMethods.begin(); curDoMethodIt != curDoMethods.end();) {
86  DoMethod& curDoMethod = **curDoMethodIt;
87 
88  bool mergedDoMethod = false;
89 
90  // Start from the next stage and iterate upwards (i.e backwards) to find a suitable
91  // stage to merge
92  for(auto candiateStageIt = std::next(curStageIt);
93  candiateStageIt != multiStage.getStages().rend(); ++candiateStageIt) {
94  Stage& candiateStage = **candiateStageIt;
95  std::vector<std::unique_ptr<DoMethod>>& candiateDoMethods =
96  candiateStage.getDoMethods();
97 
98  // Does the interval of `curDoMethod` overlap with any DoMethod interval in
99  // `candiateStage`?
100  auto candiateDoMethodIt = std::find_if(
101  candiateDoMethods.begin(), candiateDoMethods.end(),
102  [&](const std::unique_ptr<DoMethod>& doMethodPtr) {
103  return doMethodPtr->getInterval().overlaps(curDoMethod.getInterval());
104  });
105 
106  if(candiateDoMethodIt != candiateDoMethods.end()) {
107 
108  // Check if our interval exists (Note that if we overlap with a DoMethod but our
109  // interval does not exists, we cannot merge ourself into this stage).
110  candiateDoMethodIt =
111  std::find_if(candiateDoMethods.begin(), candiateDoMethods.end(),
112  [&](const std::unique_ptr<DoMethod>& doMethodPtr) {
113  return doMethodPtr->getInterval() == curDoMethod.getInterval();
114  });
115 
116  if(candiateDoMethodIt != candiateDoMethods.end()) {
117 
118  // Check if we can append our `curDoMethod` to that `candiateDoMethod`. We need
119  // to check if the resulting dep. graph is a DAG and does not contain any horizontal
120  // dependencies
121  DoMethod& candiateDoMethod = **candiateDoMethodIt;
122 
123  auto& candiateDepGraph = candiateDoMethod.getDependencyGraph();
124  auto& curDepGraph = curDoMethod.getDependencyGraph();
125 
126  auto newDepGraph = std::make_shared<DependencyGraphAccesses>(
127  stencilInstantiation, candiateDepGraph, curDepGraph);
128 
129  if(newDepGraph->isDAG() &&
130  !hasHorizontalReadBeforeWriteConflict(newDepGraph.get())) {
131  if(MergeStagesOfStencil) {
132  candiateStage.appendDoMethod(*curDoMethodIt, *candiateDoMethodIt, newDepGraph);
133  mergedDoMethod = true;
134 
135  // We moved one Do-Method away and thus broke our full axis
136  MergeDoMethodsOfStage = true;
137  break;
138  }
139  }
140  }
141  } else {
142  // Interval does not exists in `candiateStage`, just insert our DoMethod
143  if(MergeDoMethodsOfStencil && MergeDoMethodsOfStage) {
144  candiateStage.addDoMethod(*curDoMethodIt);
145  mergedDoMethod = true;
146  break;
147  }
148  }
149 
150  // The `curStage` depends on `candiateStage`, we thus cannot go further upwards
151  if(stageDAG->depends(curStage.getStageID(), candiateStage.getStageID())) {
152  break;
153  }
154  }
155 
156  if(mergedDoMethod) {
157  curDoMethodIt = curDoMethods.erase(curDoMethodIt);
158  updateFields = true;
159  } else
160  curDoMethodIt++;
161  }
162 
163  // Stage is empty, remove it (the wirdness here stems from the fact that we have a reverse
164  // iterator and erase expects a normal iterator ...)
165  if(curDoMethods.empty())
166  curStageIt = decltype(curStageIt)(multiStage.getStages().erase(--curStageIt.base()));
167  else {
168  if(updateFields)
169  curStage.update();
170  curStageIt++;
171  }
172  }
173  }
174  }
175 
176  if(context->getOptions().ReportPassStageMerger)
177  stencilInstantiation->dumpAsJson(filenameWE + "_after.json", getName());
178 
179  return true;
180 }
